40 QDataStream &operator>>(QDataStream &stream, QList<QNetworkCookie> &list)
41 {
42  list.clear();
43 
44  quint32 version;
45  stream >> version;
46 
47  if (version != JAR_VERSION)
48  return stream;
49 
50  quint32 count;
51  stream >> count;
52  for(quint32 i = 0; i < count; ++i)
53  {
54  QByteArray value;
55  stream >> value;
56  QList<QNetworkCookie> newCookies = QNetworkCookie::parseCookies(value);
57  if (newCookies.count() == 0 && value.length() != 0) {
58  qWarning() << "CookieJar: Unable to parse saved cookie:" << value;
59  }
60  for (int j = 0; j < newCookies.count(); ++j)
61  list.append(newCookies.at(j));
62  if (stream.atEnd())
63  break;
64  }
65  return stream;
66 }

123 QList<QNetworkCookie> CookieJar::removeExpired(const QList<QNetworkCookie> &cookies)
124 {
125  QList<QNetworkCookie> updatedList;
126  foreach(const QNetworkCookie &cookie, cookies) {
127  if (cookie.expirationDate() > QDateTime::currentDateTime() && !cookie.isSessionCookie()) {
128  updatedList << cookie;
