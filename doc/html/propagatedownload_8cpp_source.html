40 // Always coming in with forward slashes.
41 // In csync_excluded_no_ctx we ignore all files with longer than 254 chars
42 // This function also adds a dot at the beginning of the filename to hide the file on OS X and Linux
43 QString OWNCLOUDSYNC_EXPORT createDownloadTmpFileName(const QString &previous) {
44  QString tmpFileName;
45  QString tmpPath;
46  int slashPos = previous.lastIndexOf('/');
47  // work with both pathed filenames and only filenames
48  if (slashPos == -1) {
49  tmpFileName = previous;
50  tmpPath = QString();
51  } else {
52  tmpFileName = previous.mid(slashPos+1);
53  tmpPath = previous.left(slashPos);
54  }
55  int overhead = 1 + 1 + 2 + 8; // slash dot dot-tilde ffffffff"
56  int spaceForFileName = qMin(254, tmpFileName.length() + overhead) - overhead;
57  if (tmpPath.length() > 0) {
58  return tmpPath + '/' + '.' + tmpFileName.left(spaceForFileName) + ".~" + (QString::number(uint(qrand() % 0xFFFFFFFF), 16));
59  } else {
60  return '.' + tmpFileName.left(spaceForFileName) + ".~" + (QString::number(uint(qrand() % 0xFFFFFFFF), 16));
61  }
62 }

89 void GETFileJob::start() {
90  if (_resumeStart > 0) {
91  _headers["Range"] = "bytes=" + QByteArray::number(_resumeStart) +'-';
92  _headers["Accept-Ranges"] = "bytes";
93  qDebug() << "Retry with range " << _headers["Range"];
94  }
95 
96  QNetworkRequest req;
97  for(QMap<QByteArray, QByteArray>::const_iterator it = _headers.begin(); it != _headers.end(); ++it) {
98  req.setRawHeader(it.key(), it.value());
99  }
100 
101  if (_directDownloadUrl.isEmpty()) {
102  setReply(davRequest("GET", path(), req));
103  } else {
104  // Use direct URL
105  setReply(davRequest("GET", _directDownloadUrl, req));
106  }
107  setupConnections(reply());
108 
109  reply()->setReadBufferSize(16 * 1024); // keep low so we can easier limit the bandwidth
110  qDebug() << Q_FUNC_INFO << _bandwidthManager << _bandwidthChoked << _bandwidthLimited;
111  if (_bandwidthManager) {
112  _bandwidthManager->registerDownloadJob(this);
113  }
114 
115  if( reply()->error() != QNetworkReply::NoError ) {
116  qWarning() << Q_FUNC_INFO << " Network error: " << reply()->errorString();
117  }
118 
119  connect(reply(), SIGNAL(metaDataChanged()), this, SLOT(slotMetaDataChanged()));
120  connect(reply(), SIGNAL(readyRead()), this, SLOT(slotReadyRead()));
121  connect(reply(), SIGNAL(downloadProgress(qint64,qint64)), this, SIGNAL(downloadProgress(qint64,qint64)));
122  connect(this, SIGNAL(networkActivity()), account().data(), SIGNAL(propagatorNetworkActivity()));
123 
124  AbstractNetworkJob::start();
125 }

127 void GETFileJob::slotMetaDataChanged()
128 {
129  // For some reason setting the read buffer in GETFileJob::start doesn't seem to go
130  // through the HTTP layer thread(?)
131  reply()->setReadBufferSize(16 * 1024);
132 
133  int httpStatus = reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
134 
135  // If the status code isn't 2xx, don't write the reply body to the file.
136  // For any error: handle it when the job is finished, not here.
137  if (httpStatus / 100 != 2) {
138  _device->close();
139  return;
140  }
141  if (reply()->error() != QNetworkReply::NoError) {
142  return;
143  }
144  _etag = getEtagFromReply(reply());
145 
146  if (!_directDownloadUrl.isEmpty() && !_etag.isEmpty()) {
147  qDebug() << Q_FUNC_INFO << "Direct download used, ignoring server ETag" << _etag;
148  _etag = QByteArray(); // reset received ETag
149  } else if (!_directDownloadUrl.isEmpty()) {
150  // All fine, ETag empty and directDownloadUrl used
151  } else if (_etag.isEmpty()) {
152  qDebug() << Q_FUNC_INFO << "No E-Tag reply by server, considering it invalid";
153  _errorString = tr("No E-Tag received from server, check Proxy/Gateway");
154  _errorStatus = SyncFileItem::NormalError;
155  reply()->abort();
156  return;
157  } else if (!_expectedEtagForResume.isEmpty() && _expectedEtagForResume != _etag) {
158  qDebug() << Q_FUNC_INFO << "We received a different E-Tag for resuming!"
159  << _expectedEtagForResume << "vs" << _etag;
160  _errorString = tr("We received a different E-Tag for resuming. Retrying next time.");
161  _errorStatus = SyncFileItem::NormalError;
162  reply()->abort();
163  return;
164  }
165 
166  quint64 start = 0;
167  QByteArray ranges = reply()->rawHeader("Content-Range");
168  if (!ranges.isEmpty()) {
169  QRegExp rx("bytes (\\d+)-");
170  if (rx.indexIn(ranges) >= 0) {
171  start = rx.cap(1).toULongLong();
172  }
173  }
174  if (start != _resumeStart) {
175  qDebug() << Q_FUNC_INFO << "Wrong content-range: "<< ranges << " while expecting start was" << _resumeStart;
176  if (ranges.isEmpty()) {
177  // device doesn't support range, just try again from scratch
178  _device->close();
179  if (!_device->open(QIODevice::WriteOnly)) {
180  _errorString = _device->errorString();
181  _errorStatus = SyncFileItem::NormalError;
182  reply()->abort();
183  return;
184  }
185  _resumeStart = 0;
186  } else {
187  _errorString = tr("Server returned wrong content-range");
188  _errorStatus = SyncFileItem::NormalError;
189  reply()->abort();
190  return;
191  }
192  }
193 
194  auto lastModified = reply()->header(QNetworkRequest::LastModifiedHeader);
195  if (!lastModified.isNull()) {
196  _lastModified = Utility::qDateTimeToTime_t(lastModified.toDateTime());
197  }
198 }

232 void GETFileJob::slotReadyRead()
233 {
234  int bufferSize = qMin(1024*8ll , reply()->bytesAvailable());
235  QByteArray buffer(bufferSize, Qt::Uninitialized);
236 
237  //qDebug() << Q_FUNC_INFO << reply()->bytesAvailable() << reply()->isOpen() << reply()->isFinished();
238 
239  while(reply()->bytesAvailable() > 0) {
240  if (_bandwidthChoked) {
241  qDebug() << Q_FUNC_INFO << "Download choked";
242  break;
243  }
244  qint64 toRead = bufferSize;
245  if (_bandwidthLimited) {
246  toRead = qMin(qint64(bufferSize), _bandwidthQuota);
247  if (toRead == 0) {
248  //qDebug() << Q_FUNC_INFO << "Out of quota";
249  break;
250  }
251  _bandwidthQuota -= toRead;
252  //qDebug() << Q_FUNC_INFO << "Reading" << toRead << "remaining" << _bandwidthQuota;
253  }
254 
255  qint64 r = reply()->read(buffer.data(), toRead);
256  if (r < 0) {
257  _errorString = reply()->errorString();
258  _errorStatus = SyncFileItem::NormalError;
259  qDebug() << "Error while reading from device: " << _errorString;
260  reply()->abort();
261  return;
262  }
263 
264  if (_device->isOpen()) {
265  qint64 w = _device->write(buffer.constData(), r);
266  if (w != r) {
267  _errorString = _device->errorString();
268  _errorStatus = SyncFileItem::NormalError;
269  qDebug() << "Error while writing to file" << w << r << _errorString;
270  reply()->abort();
271  return;
272  }
273  }
274  }
275 
276  //qDebug() << Q_FUNC_INFO << "END" << reply()->isFinished() << reply()->bytesAvailable() << _hasEmittedFinishedSignal;
277  if (reply()->isFinished() && reply()->bytesAvailable() == 0) {
278  qDebug() << Q_FUNC_INFO << "Actually finished!";
279  if (_bandwidthManager) {
280  _bandwidthManager->unregisterDownloadJob(this);
281  }
282  if (!_hasEmittedFinishedSignal) {
283  emit finishedSignal();
284  }
285  _hasEmittedFinishedSignal = true;
286  deleteLater();
287  }

311 void PropagateDownloadFile::start()
312 {
313  if (_propagator->_abortRequested.fetchAndAddRelaxed(0))
314  return;
315 
316  qDebug() << Q_FUNC_INFO << _item->_file << _propagator->_activeJobList.count();
317  _stopwatch.start();
318 
319  if (_deleteExisting) {
320  deleteExistingFolder();
321 
322  // check for error with deletion
323  if (_state == Finished) {
324  return;
325  }
326  }
327 
328  // do a klaas' case clash check.
329  if( _propagator->localFileNameClash(_item->_file) ) {
330  done( SyncFileItem::NormalError, tr("File %1 can not be downloaded because of a local file name clash!")
331  .arg(QDir::toNativeSeparators(_item->_file)) );
332  return;
333  }
334 
335  emit progress(*_item, 0);
336 
337  QString tmpFileName;
338  QByteArray expectedEtagForResume;
339  const SyncJournalDb::DownloadInfo progressInfo = _propagator->_journal->getDownloadInfo(_item->_file);
340  if (progressInfo._valid) {
341  // if the etag has changed meanwhile, remove the already downloaded part.
342  if (progressInfo._etag != _item->_etag) {
343  FileSystem::remove(_propagator->getFilePath(progressInfo._tmpfile));
344  _propagator->_journal->setDownloadInfo(_item->_file, SyncJournalDb::DownloadInfo());
345  } else {
346  tmpFileName = progressInfo._tmpfile;
347  expectedEtagForResume = progressInfo._etag;
348  }
349 
350  }
351 
352  if (tmpFileName.isEmpty()) {
353  tmpFileName = createDownloadTmpFileName(_item->_file);
354  }
355 
356  _tmpFile.setFileName(_propagator->getFilePath(tmpFileName));
357  if (!_tmpFile.open(QIODevice::Append | QIODevice::Unbuffered)) {
358  done(SyncFileItem::NormalError, _tmpFile.errorString());
359  return;
360  }
361 
362  FileSystem::setFileHidden(_tmpFile.fileName(), true);
363 
364  _resumeStart = _tmpFile.size();
365  if (_resumeStart > 0) {
366  if (_resumeStart == _item->_size) {
367  qDebug() << "File is already complete, no need to download";
368  _tmpFile.close();
369  downloadFinished();
370  return;
371  }
372  }
373 
374  // If there's not enough space to fully download this file, stop.
375  const auto diskSpaceResult = _propagator->diskSpaceCheck();
376  if (diskSpaceResult == OwncloudPropagator::DiskSpaceFailure) {
377  _item->_errorMayBeBlacklisted = true;
378  done(SyncFileItem::NormalError,
379  tr("The download would reduce free disk space below %1").arg(
380  Utility::octetsToString(freeSpaceLimit())));
381  return;
382  } else if (diskSpaceResult == OwncloudPropagator::DiskSpaceCritical) {
383  done(SyncFileItem::FatalError,
384  tr("Free space on disk is less than %1").arg(
385  Utility::octetsToString(criticalFreeSpaceLimit())));
386  return;
387  }
388 
389  {
390  SyncJournalDb::DownloadInfo pi;
391  pi._etag = _item->_etag;
392  pi._tmpfile = tmpFileName;
393  pi._valid = true;
394  _propagator->_journal->setDownloadInfo(_item->_file, pi);
395  _propagator->_journal->commit("download file start");
396  }
397 
398  QMap<QByteArray, QByteArray> headers;
399 
400  if (_item->_directDownloadUrl.isEmpty()) {
401  // Normal job, download from oC instance
402  _job = new GETFileJob(_propagator->account(),
403  _propagator->_remoteFolder + _item->_file,
404  &_tmpFile, headers, expectedEtagForResume, _resumeStart, this);
405  } else {
406  // We were provided a direct URL, use that one
407  qDebug() << Q_FUNC_INFO << "directDownloadUrl given for " << _item->_file << _item->_directDownloadUrl;
408 
409  if (!_item->_directDownloadCookies.isEmpty()) {
410  headers["Cookie"] = _item->_directDownloadCookies.toUtf8();
411  }
412 
413  QUrl url = QUrl::fromUserInput(_item->_directDownloadUrl);
414  _job = new GETFileJob(_propagator->account(),
415  url,
416  &_tmpFile, headers, expectedEtagForResume, _resumeStart, this);
417  }
418  _job->setBandwidthManager(&_propagator->_bandwidthManager);
419  connect(_job, SIGNAL(finishedSignal()), this, SLOT(slotGetFinished()));
420  connect(_job, SIGNAL(downloadProgress(qint64,qint64)), this, SLOT(slotDownloadProgress(qint64,qint64)));
421  _propagator->_activeJobList.append(this);
422  _job->start();
423 }

438 const char owncloudCustomSoftErrorStringC[] = "owncloud-custom-soft-error-string";
439 void PropagateDownloadFile::slotGetFinished()
440 {
441  _propagator->_activeJobList.removeOne(this);
442 
443  GETFileJob *job = qobject_cast<GETFileJob *>(sender());
444  Q_ASSERT(job);
445 
446  qDebug() << Q_FUNC_INFO << job->reply()->request().url() << "FINISHED WITH STATUS"
447  << job->reply()->error()
448  << (job->reply()->error() == QNetworkReply::NoError ? QLatin1String("") : job->reply()->errorString())
449  << _item->_httpErrorCode
450  << _tmpFile.size() << _item->_size << job->resumeStart()
451  << job->reply()->rawHeader("Content-Range") << job->reply()->rawHeader("Content-Length");
452 
453  QNetworkReply::NetworkError err = job->reply()->error();
454  if (err != QNetworkReply::NoError) {
455  _item->_httpErrorCode = job->reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
456 
457  // If we sent a 'Range' header and get 416 back, we want to retry
458  // without the header.
459  const bool badRangeHeader = job->resumeStart() > 0 && _item->_httpErrorCode == 416;
460  if (badRangeHeader) {
461  qDebug() << Q_FUNC_INFO << "server replied 416 to our range request, trying again without";
462  _propagator->_anotherSyncNeeded = true;
463  }
464 
465  // Getting a 404 probably means that the file was deleted on the server.
466  const bool fileNotFound = _item->_httpErrorCode == 404;
467  if (fileNotFound) {
468  qDebug() << Q_FUNC_INFO << "server replied 404, assuming file was deleted";
469  }
470 
471  // Don't keep the temporary file if it is empty or we
472  // used a bad range header or the file's not on the server anymore.
473  if (_tmpFile.size() == 0 || badRangeHeader || fileNotFound) {
474  _tmpFile.close();
475  FileSystem::remove(_tmpFile.fileName());
476  _propagator->_journal->setDownloadInfo(_item->_file, SyncJournalDb::DownloadInfo());
477  }
478 
479  if(!_item->_directDownloadUrl.isEmpty() && err != QNetworkReply::OperationCanceledError) {
480  // If this was with a direct download, retry without direct download
481  qWarning() << "Direct download of" << _item->_directDownloadUrl << "failed. Retrying through owncloud.";
482  _item->_directDownloadUrl.clear();
483  start();
484  return;
485  }
486 
487  // This gives a custom QNAM (by the user of libowncloudsync) to abort() a QNetworkReply in its metaDataChanged() slot and
488  // set a custom error string to make this a soft error. In contrast to the default hard error this won't bring down
489  // the whole sync and allows for a custom error message.
490  QNetworkReply *reply = job->reply();
491  if (err == QNetworkReply::OperationCanceledError && reply->property(owncloudCustomSoftErrorStringC).isValid()) {
492  job->setErrorString(reply->property(owncloudCustomSoftErrorStringC).toString());
493  job->setErrorStatus(SyncFileItem::SoftError);
494  } else if (badRangeHeader) {
495  // Can't do this in classifyError() because 416 without a
496  // Range header should result in NormalError.
497  job->setErrorStatus(SyncFileItem::SoftError);
498  } else if (fileNotFound) {
499  job->setErrorString(tr("File was deleted from server"));
500  job->setErrorStatus(SyncFileItem::SoftError);
501  }
502 
503  SyncFileItem::Status status = job->errorStatus();
504  if (status == SyncFileItem::NoStatus) {
505  status = classifyError(err, _item->_httpErrorCode,
506  &_propagator->_anotherSyncNeeded);
507  }
508 
509  done(status, job->errorString());
510  return;
511  }
512 
513  if (!job->etag().isEmpty()) {
514  // The etag will be empty if we used a direct download URL.
515  // (If it was really empty by the server, the GETFileJob will have errored
516  _item->_etag = parseEtag(job->etag());
517  }
518  if (job->lastModified()) {
519  // It is possible that the file was modified on the server since we did the discovery phase
520  // so make sure we have the up-to-date time
521  _item->_modtime = job->lastModified();
522  }
523  _item->_requestDuration = job->duration();
524  _item->_responseTimeStamp = job->responseTimestamp();
525 
526  _tmpFile.close();
527  _tmpFile.flush();
528 
529  /* Check that the size of the GET reply matches the file size. There have been cases
530  * reported that if a server breaks behind a proxy, the GET is still a 200 but is
531  * truncated, as described here: https://github.com/owncloud/mirall/issues/2528
532  */
533  const QByteArray sizeHeader("Content-Length");
534  quint64 bodySize = job->reply()->rawHeader(sizeHeader).toULongLong();
535 
536  if (!job->reply()->rawHeader(sizeHeader).isEmpty() && _tmpFile.size() > 0 && bodySize == 0) {
537  // Strange bug with broken webserver or webfirewall https://github.com/owncloud/client/issues/3373#issuecomment-122672322
538  // This happened when trying to resume a file. The Content-Range header was files, Content-Length was == 0
539  qDebug() << bodySize << _item->_size << _tmpFile.size() << job->resumeStart();
540  FileSystem::remove(_tmpFile.fileName());
541  done(SyncFileItem::SoftError, QLatin1String("Broken webserver returning empty content length for non-empty file on resume"));
542  return;
543  }
544 
545  if(bodySize > 0 && bodySize != _tmpFile.size() - job->resumeStart() ) {
546  qDebug() << bodySize << _tmpFile.size() << job->resumeStart();
547  _propagator->_anotherSyncNeeded = true;
548  done(SyncFileItem::SoftError, tr("The file could not be downloaded completely."));
549  return;
550  }
551 
552  if (_tmpFile.size() == 0 && _item->_size > 0) {
553  FileSystem::remove(_tmpFile.fileName());
554  done(SyncFileItem::NormalError,
555  tr("The downloaded file is empty despite the server announced it should have been %1.")
556  .arg(Utility::octetsToString(_item->_size)));
557  return;
558  }
559 
560  // Do checksum validation for the download. If there is no checksum header, the validator
561  // will also emit the validated() signal to continue the flow in slot transmissionChecksumValidated()
562  // as this is (still) also correct.
563  ValidateChecksumHeader *validator = new ValidateChecksumHeader(this);
564  connect(validator, SIGNAL(validated(QByteArray,QByteArray)),
565  SLOT(transmissionChecksumValidated(QByteArray,QByteArray)));
566  connect(validator, SIGNAL(validationFailed(QString)),
567  SLOT(slotChecksumFail(QString)));
568  auto checksumHeader = job->reply()->rawHeader(checkSumHeaderC);
569  validator->start(_tmpFile.fileName(), checksumHeader);
570 }

579 void PropagateDownloadFile::deleteExistingFolder()
580 {
581  QString existingDir = _propagator->getFilePath(_item->_file);
582  if (!QFileInfo(existingDir).isDir()) {
583  return;
584  }
585 
586  // Delete the directory if it is empty!
587  QDir dir(existingDir);
588  if (dir.entryList(QDir::NoDotAndDotDot|QDir::AllEntries).count() == 0) {
589  if (dir.rmdir(existingDir)) {
590  return;
591  }
592  // on error, just try to move it away...
593  }
594 
595  QString conflictDir = FileSystem::makeConflictFileName(
596  existingDir, Utility::qDateTimeFromTime_t(FileSystem::getModTime(existingDir)));
597 
598  emit _propagator->touchedFile(existingDir);
599  emit _propagator->touchedFile(conflictDir);
600  QString renameError;
601  if (!FileSystem::rename(existingDir, conflictDir, &renameError)) {
602  done(SyncFileItem::NormalError, renameError);
603  }
604 }

606 namespace { // Anonymous namespace for the recall feature
607 static QString makeRecallFileName(const QString &fn)
608 {
609  QString recallFileName(fn);
610  // Add _recall-XXXX before the extension.

624 {

627  FileSystem::setFileHidden(filePath, true);

707 void PropagateDownloadFile::downloadFinished()
708 {
709  QString fn = _propagator->getFilePath(_item->_file);
710 
711  // In case of file name clash, report an error
712  // This can happen if another parallel download saved a clashing file.
713  if (_propagator->localFileNameClash(_item->_file)) {
714  done( SyncFileItem::NormalError, tr("File %1 cannot be saved because of a local file name clash!")
715  .arg(QDir::toNativeSeparators(_item->_file)) );
716  return;
717  }
718 
719  // In case of conflict, make a backup of the old file
720  // Ignore conflicts where both files are binary equal
721  bool isConflict = _item->_instruction == CSYNC_INSTRUCTION_CONFLICT
722  && !FileSystem::fileEquals(fn, _tmpFile.fileName());
723  if (isConflict) {
724  QString renameError;
725  QString conflictFileName = FileSystem::makeConflictFileName(
726  fn, Utility::qDateTimeFromTime_t(FileSystem::getModTime(fn)));
727  if (!FileSystem::rename(fn, conflictFileName, &renameError)) {
728  // If the rename fails, don't replace it.
729 
730  // If the file is locked, we want to retry this sync when it
731  // becomes available again.
732  if (FileSystem::isFileLocked(fn)) {
733  emit _propagator->seenLockedFile(fn);
734  }
735 
736  done(SyncFileItem::SoftError, renameError);
737  return;
738  }
739  qDebug() << "Created conflict file" << fn << "->" << conflictFileName;
740  }
741 
742  FileSystem::setModTime(_tmpFile.fileName(), _item->_modtime);
743  // We need to fetch the time again because some file systems such as FAT have worse than a second
744  // Accuracy, and we really need the time from the file system. (#3103)
745  _item->_modtime = FileSystem::getModTime(_tmpFile.fileName());
746 
747  if (FileSystem::fileExists(fn)) {
748  // Preserve the existing file permissions.
749  QFileInfo existingFile(fn);
750  if (existingFile.permissions() != _tmpFile.permissions()) {
751  _tmpFile.setPermissions(existingFile.permissions());
752  }
753  preserveGroupOwnership(_tmpFile.fileName(), existingFile);
754 
755  // Check whether the existing file has changed since the discovery
756  // phase by comparing size and mtime to the previous values. This
757  // is necessary to avoid overwriting user changes that happened between
758  // the discovery phase and now.
759  const qint64 expectedSize = _item->log._other_size;
760  const time_t expectedMtime = _item->log._other_modtime;
761  if (! FileSystem::verifyFileUnchanged(fn, expectedSize, expectedMtime)) {
762  _propagator->_anotherSyncNeeded = true;
763  done(SyncFileItem::SoftError, tr("File has changed since discovery"));
764  return;
765  }
766  }
767 
768  // Apply the remote permissions
769  // Older server versions sometimes provide empty remote permissions
770  // see #4450 - don't adjust the write permissions there.
771  const int serverVersionGoodRemotePerm = 0x070000; // 7.0.0
772  if (_propagator->account()->serverVersionInt() >= serverVersionGoodRemotePerm) {
773  FileSystem::setFileReadOnlyWeak(_tmpFile.fileName(),
774  !_item->_remotePerm.contains('W'));
775  }
776 
777  QString error;
778  emit _propagator->touchedFile(fn);
779  // The fileChanged() check is done above to generate better error messages.
780  if (!FileSystem::uncheckedRenameReplace(_tmpFile.fileName(), fn, &error)) {
781  qDebug() << Q_FUNC_INFO << QString("Rename failed: %1 => %2").arg(_tmpFile.fileName()).arg(fn);
782 
783  // If we moved away the original file due to a conflict but can't
784  // put the downloaded file in its place, we are in a bad spot:
785  // If we do nothing the next sync run will assume the user deleted
786  // the file!
787  // To avoid that, the file is removed from the metadata table entirely
788  // which makes it look like we're just about to initially download
789  // it.
790  if (isConflict) {
791  _propagator->_journal->deleteFileRecord(fn);
792  _propagator->_journal->commit("download finished");
793  }
794 
795  // If the file is locked, we want to retry this sync when it
796  // becomes available again, otherwise try again directly
797  if (FileSystem::isFileLocked(fn)) {
798  emit _propagator->seenLockedFile(fn);
799  } else {
800  _propagator->_anotherSyncNeeded = true;
801  }
802 
803  done(SyncFileItem::SoftError, error);
804  return;
805  }
806  FileSystem::setFileHidden(fn, false);
807 
808  // Maybe we downloaded a newer version of the file than we thought we would...
809  // Get up to date information for the journal.
810  _item->_size = FileSystem::getSize(fn);
811 
812  if (!_propagator->_journal->setFileRecord(SyncJournalFileRecord(*_item, fn))) {
813  done(SyncFileItem::FatalError, tr("Error writing metadata to the database"));
814  return;
815  }
816  _propagator->_journal->setDownloadInfo(_item->_file, SyncJournalDb::DownloadInfo());
817  _propagator->_journal->commit("download file start2");
818  done(isConflict ? SyncFileItem::Conflict : SyncFileItem::Success);
819 
820  // handle the special recall file
821  if(!_item->_remotePerm.contains("S")
822  && (_item->_file == QLatin1String(".sys.admin#recall#")
823  || _item->_file.endsWith("/.sys.admin#recall#"))) {
824  handleRecallFile(fn, _propagator->_localDir, *_propagator->_journal);
825  }
826 
827  qint64 duration = _stopwatch.elapsed();
828  if (isLikelyFinishedQuickly() && duration > 5*1000) {
829  qDebug() << "WARNING: Unexpectedly slow connection, took" << duration << "msec for" << _item->_size - _resumeStart << "bytes for" << _item->_file;
830  }
831 }
