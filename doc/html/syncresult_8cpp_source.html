38 {
39  QString re;
40  Status stat = status();
41 
42  switch( stat ){
43  case Undefined:
44  re = QLatin1String("Undefined");
45  break;
46  case NotYetStarted:
47  re = QLatin1String("Not yet Started");
48  break;
49  case SyncRunning:
50  re = QLatin1String("Sync Running");
51  break;
52  case Success:
53  re = QLatin1String("Success");
54  break;
55  case Error:
56  re = QLatin1String("Error");
57  break;
58  case SetupError:
59  re = QLatin1String("SetupError");
60  break;
61  case SyncPrepare:
62  re = QLatin1String("SyncPrepare");
63  break;
64  case Problem:
65  re = QLatin1String("Success, some files were ignored.");
66  break;
67  case SyncAbortRequested:
68  re = QLatin1String("Sync Request aborted by user");
69  break;
70  case Paused:

73  }
