69 void PropagateRemoteMkdir::slotMkcolJobFinished()
70 {
71  _propagator->_activeJobList.removeOne(this);
72 
73  Q_ASSERT(_job);
74 
75  qDebug() << Q_FUNC_INFO << _job->reply()->request().url() << "FINISHED WITH STATUS"
76  << _job->reply()->error()
77  << (_job->reply()->error() == QNetworkReply::NoError ? QLatin1String("") : _job->reply()->errorString());
78 
79  QNetworkReply::NetworkError err = _job->reply()->error();
80  _item->_httpErrorCode = _job->reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
81 
82  if (_item->_httpErrorCode == 405) {
83  // This happens when the directory already exists. Nothing to do.
84  } else if (err != QNetworkReply::NoError) {
85  SyncFileItem::Status status = classifyError(err, _item->_httpErrorCode,
86  &_propagator->_anotherSyncNeeded);
87  auto errorString = _job->reply()->errorString();
88  if (_job->reply()->hasRawHeader("OC-ErrorString")) {
89  errorString = _job->reply()->rawHeader("OC-ErrorString");
90  }
91  done(status, errorString);
92  return;
93  } else if (_item->_httpErrorCode != 201) {
94  // Normally we expect "201 Created"
95  // If it is not the case, it might be because of a proxy or gateway intercepting the request, so we must
96  // throw an error.
97  done(SyncFileItem::NormalError, tr("Wrong HTTP code returned by server. Expected 201, but received \"%1 %2\".")
98  .arg(_item->_httpErrorCode).arg(_job->reply()->attribute(QNetworkRequest::HttpReasonPhraseAttribute).toString()));
99  return;
100  }
101 
102  _item->_requestDuration = _job->duration();
103  _item->_responseTimeStamp = _job->responseTimestamp();
104  _item->_fileId = _job->reply()->rawHeader("OC-FileId");
105 
106  if (_item->_fileId.isEmpty()) {
107  // Owncloud 7.0.0 and before did not have a header with the file id.
108  // (https://github.com/owncloud/core/issues/9000)
109  // So we must get the file id using a PROPFIND
110  // This is required so that we can detect moves even if the folder is renamed on the server
111  // while files are still uploading
112  _propagator->_activeJobList.append(this);
113  auto propfindJob = new PropfindJob(_job->account(), _job->path(), this);
114  propfindJob->setProperties(QList<QByteArray>() << "getetag" << "http://owncloud.org/ns:id");
115  QObject::connect(propfindJob, SIGNAL(result(QVariantMap)), this, SLOT(propfindResult(QVariantMap)));
116  QObject::connect(propfindJob, SIGNAL(finishedWithError()), this, SLOT(propfindError()));
117  propfindJob->start();
118  _job = propfindJob;
119  return;
120  }
121  success();
122 }

124 void PropagateRemoteMkdir::propfindResult(const QVariantMap &result)
125 {
126  _propagator->_activeJobList.removeOne(this);
127  if (result.contains("getetag")) {
128  _item->_etag = result["getetag"].toByteArray();
129  }
130  if (result.contains("id")) {
131  _item->_fileId = result["id"].toByteArray();
132  }
133  success();
134 }

143 void PropagateRemoteMkdir::success()
144 {
145  // save the file id already so we can detect rename or remove
146  SyncJournalFileRecord record(*_item, _propagator->_localDir + _item->destination());
147  if (!_propagator->_journal->setFileRecord(record)) {
148  done(SyncFileItem::FatalError, tr("Error writing metadata to the database"));
149  return;
150  }
151 
152  done(SyncFileItem::Success);
