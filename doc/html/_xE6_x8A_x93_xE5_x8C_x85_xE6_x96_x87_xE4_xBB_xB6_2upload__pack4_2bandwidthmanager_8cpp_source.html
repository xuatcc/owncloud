164 void BandwidthManager::relativeUploadMeasuringTimerExpired()
165 {
166  if (!usingRelativeUploadLimit() || _relativeUploadDeviceList.count() == 0) {
167  // Not in this limiting mode, just wait 1 sec to continue the cycle
168  _relativeUploadDelayTimer.setInterval(1000);
169  _relativeUploadDelayTimer.start();
170  return;
171  }
172  if (_relativeLimitCurrentMeasuredDevice == 0) {
173  qDebug() << Q_FUNC_INFO << "No device set, just waiting 1 sec";
174  _relativeUploadDelayTimer.setInterval(1000);
175  _relativeUploadDelayTimer.start();
176  return;
177  }
178 
179 // qDebug() << Q_FUNC_INFO << _relativeUploadDeviceList.count() << "Starting Delay";
180 
181  qint64 relativeLimitProgressMeasured = (_relativeLimitCurrentMeasuredDevice->_readWithProgress
182  + _relativeLimitCurrentMeasuredDevice->_read) / 2;
183  qint64 relativeLimitProgressDifference = relativeLimitProgressMeasured - _relativeUploadLimitProgressAtMeasuringRestart;
184 // qDebug() << Q_FUNC_INFO << _relativeUploadLimitProgressAtMeasuringRestart
185 // << relativeLimitProgressMeasured << relativeLimitProgressDifference;
186 
187 // qint64 speedkBPerSec = (relativeLimitProgressDifference / relativeLimitMeasuringTimerIntervalMsec*1000.0) / 1024.0;
188 // qDebug() << Q_FUNC_INFO << relativeLimitProgressDifference/1024 <<"kB =>" << speedkBPerSec << "kB/sec on full speed ("
189 // << _relativeLimitCurrentMeasuredDevice->_readWithProgress << _relativeLimitCurrentMeasuredDevice->_read
190 // << qAbs(_relativeLimitCurrentMeasuredDevice->_readWithProgress
191 // - _relativeLimitCurrentMeasuredDevice->_read) << ")";
192 
193  qint64 uploadLimitPercent = -_currentUploadLimit;
194  // don't use too extreme values
195  uploadLimitPercent = qMin(uploadLimitPercent, qint64(90));
196  uploadLimitPercent = qMax(qint64(10), uploadLimitPercent);
197  qint64 wholeTimeMsec = (100.0 / uploadLimitPercent) * relativeLimitMeasuringTimerIntervalMsec;
198  qint64 waitTimeMsec = wholeTimeMsec - relativeLimitMeasuringTimerIntervalMsec;
199  qint64 realWaitTimeMsec = waitTimeMsec + wholeTimeMsec;
200 // qDebug() << Q_FUNC_INFO << waitTimeMsec << " - "<< realWaitTimeMsec <<
201 // " msec for " << uploadLimitPercent << "%";
202 // qDebug() << Q_FUNC_INFO << "XXXX" << uploadLimitPercent << relativeLimitMeasuringTimerIntervalMsec;
203 
204  // We want to wait twice as long since we want to give all
205  // devices the same quota we used now since we don't want
206  // any upload to timeout
207  _relativeUploadDelayTimer.setInterval(realWaitTimeMsec);
208  _relativeUploadDelayTimer.start();
209 
210  int deviceCount = _relativeUploadDeviceList.count();
211  qint64 quotaPerDevice = relativeLimitProgressDifference * (uploadLimitPercent / 100.0) / deviceCount + 1.0;
212 // qDebug() << Q_FUNC_INFO << "YYYY" << relativeLimitProgressDifference << uploadLimitPercent << deviceCount;
213  Q_FOREACH(UploadDevice *ud, _relativeUploadDeviceList) {
214  ud->setBandwidthLimited(true);
215  ud->setChoked(false);
216  ud->giveBandwidthQuota(quotaPerDevice);
217 // qDebug() << Q_FUNC_INFO << "Gave" << quotaPerDevice/1024.0 << "kB to" << ud;
218  }
219  _relativeLimitCurrentMeasuredDevice = 0;
220 }

222 void BandwidthManager::relativeUploadDelayTimerExpired()
223 {
224  // Switch to measuring state
225  _relativeUploadMeasuringTimer.start(); // always start to continue the cycle
226 
227  if (!usingRelativeUploadLimit()) {
228  return; // oh, not actually needed
229  }
230 
231  if (_relativeUploadDeviceList.isEmpty()) {
232  return;
233  }
234 
235 // qDebug() << Q_FUNC_INFO << _relativeUploadDeviceList.count() << "Starting measuring";
236 
237  // Take first device and then append it again (= we round robin all devices)
238  _relativeLimitCurrentMeasuredDevice = _relativeUploadDeviceList.takeFirst();
239  _relativeUploadDeviceList.append(_relativeLimitCurrentMeasuredDevice);
240 
241  _relativeUploadLimitProgressAtMeasuringRestart = (_relativeLimitCurrentMeasuredDevice->_readWithProgress
242  + _relativeLimitCurrentMeasuredDevice->_read) / 2;
243  _relativeLimitCurrentMeasuredDevice->setBandwidthLimited(false);
244  _relativeLimitCurrentMeasuredDevice->setChoked(false);
245 
246  // choke all other UploadDevices
247  Q_FOREACH(UploadDevice *ud, _relativeUploadDeviceList) {
248  if (ud != _relativeLimitCurrentMeasuredDevice) {
249  ud->setBandwidthLimited(true);
250  ud->setChoked(true);
251  }
252  }
253 
254  // now we're in measuring state
255 }

355 void BandwidthManager::switchingTimerExpired() {
356  qint64 newUploadLimit = _propagator->_uploadLimit.fetchAndAddAcquire(0);
357  if (newUploadLimit != _currentUploadLimit) {
358  qDebug() << Q_FUNC_INFO << "Upload Bandwidth limit changed" << _currentUploadLimit << newUploadLimit;
359  _currentUploadLimit = newUploadLimit;
360  Q_FOREACH(UploadDevice *ud, _relativeUploadDeviceList) {
361  if (newUploadLimit == 0) {
362  ud->setBandwidthLimited(false);
363  ud->setChoked(false);
364  } else if (newUploadLimit > 0) {
365  ud->setBandwidthLimited(true);
366  ud->setChoked(false);
367  } else if (newUploadLimit < 0) {
368  ud->setBandwidthLimited(true);
369  ud->setChoked(true);
370  }
371  }
372  }
373  qint64 newDownloadLimit = _propagator->_downloadLimit.fetchAndAddAcquire(0);
374  if (newDownloadLimit != _currentDownloadLimit) {
375  qDebug() << Q_FUNC_INFO << "Download Bandwidth limit changed" << _currentDownloadLimit << newDownloadLimit;
376  _currentDownloadLimit = newDownloadLimit;
377  Q_FOREACH(GETFileJob *j, _downloadJobList) {
378  if (usingAbsoluteDownloadLimit()) {
379  j->setBandwidthLimited(true);
380  j->setChoked(false);
381  } else if (usingRelativeDownloadLimit()) {
382  j->setBandwidthLimited(true);
383  j->setChoked(true);
384  } else {
385  j->setBandwidthLimited(false);
386  j->setChoked(false);
387  }
388  }
389  }
