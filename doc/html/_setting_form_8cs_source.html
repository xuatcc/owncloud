127  void btn_Click_Event(object obj, EventArgs ea)
128  {
129  if (obj.Equals(pictureBox_buttonClose)) this.Close();
130  if (obj.Equals(pictureBox_buttonMinimize)) this.WindowState = FormWindowState.Minimized;
131  if (obj.Equals(button_confirm))
132  {
133 
134  /* 可能不合法的配置 */
135  string sync_ip = settingSyncForm.ServerAddress;
136  //string a = "";
137  if (!CodeAnalysis.IsValidIP(sync_ip))
138  {
139  label_error.Visible = true;
140  label_error.Text = "服务器地址不合法";
141  return;
142  }
143  string sync_port_str = settingSyncForm.Port;
144  if (!CodeAnalysis.IsInteger(sync_port_str))
145  {
146  label_error.Visible = true;
147  label_error.Text = "端口不合法";
148  return;
149  }
150  /* 保存配置 */
151  MyConfig.ConfigFile configFile;
152  if (File.Exists(MyConfig.CONFIG_FILE_PATH)) configFile = MyConfig.readConfig(MyConfig.CONFIG_FILE_PATH);
153  else configFile = new MyConfig.ConfigFile();
154  if (configFile == null) return;
155 
156  /* 保存服务器地址 */
157  configFile.createOrModifyItem(MyConfig.ConfigFile.TABLE_NAME_SYNC,
158  MyConfig.ConfigFile.Sync.KEY_SERVER_URI, sync_ip);
159  /* 保存端口 */
160  configFile.createOrModifyItem(MyConfig.ConfigFile.TABLE_NAME_SYNC,
161  MyConfig.ConfigFile.Sync.KEY_SERVER_PORT, int.Parse(sync_port_str));
162  /* 是否自动启动 */
163  configFile.createOrModifyItem(MyConfig.ConfigFile.TABLE_NAME_PRIMARY,
164  MyConfig.ConfigFile.Primary.KEY_AUTO_START, settingPrimaryForm.isAutoStart);
165  /* 是否自动同步 */
166  configFile.createOrModifyItem(MyConfig.ConfigFile.TABLE_NAME_SYNC,
167  MyConfig.ConfigFile.Sync.KEY_AUTO_SYNC, settingSyncForm.isAutoSync);
168 
169  /* 保存外观 */
170  /* 大图标大小 */
171  configFile.createOrModifyItem(MyConfig.ConfigFile.TABLE_NAME_SKIN,
172  MyConfig.ConfigFile.Skin.KEY_LARGE_ICON_SIZE, settingSkinForm.SizeLargeIcon);
173  /* 小图标大小 */
174  configFile.createOrModifyItem(MyConfig.ConfigFile.TABLE_NAME_SKIN,
175  MyConfig.ConfigFile.Skin.KEY_SMALL_ICON_SIZE, settingSkinForm.SizeSmallIcon);
176 
177  MyConfig.saveConfig(MyConfig.CONFIG_FILE_PATH, configFile);
