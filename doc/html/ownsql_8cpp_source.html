45 bool SqlDatabase::openHelper( const QString& filename, int sqliteFlags )
46 {
47  if( isOpen() ) {
48  return true;
49  }
50 
51  sqliteFlags |= SQLITE_OPEN_NOMUTEX;
52 
53  SQLITE_DO( sqlite3_open_v2(filename.toUtf8().constData(), &_db, sqliteFlags, 0) );
54 
55  if( _errId != SQLITE_OK ) {
56  qDebug() << "Error:" << _error << "for" << filename;
57  close();
58  return false;
59  }
60 
61  if( !_db ) {
62  qDebug() << "Error: no database for" << filename;
63  return false;
64  }
65 
66  sqlite3_busy_timeout(_db, 5000);
67 
68  return true;
69 }

89 bool SqlDatabase::openOrCreateReadWrite( const QString& filename )
90 {
91  if( isOpen() ) {
92  return true;
93  }
94 
95  if( !openHelper(filename, SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE) ) {
96  return false;
97  }
98 
99  if( !checkDb() ) {
100  // When disk space is low, checking the db may fail even though it's fine.
101  qint64 freeSpace = Utility::freeDiskSpace(filename);
102  if (freeSpace < 1000000) {
103  qDebug() << "Consistency check failed, disk space is low, aborting" << freeSpace;
104  close();
105  return false;
106  }
107 
108  qDebug() << "Consistency check failed, removing broken db" << filename;
109  close();
110  QFile::remove(filename);
111 
112  return openHelper(filename, SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE);
113  }
114 
115  return true;
116 }

202 int SqlQuery::prepare( const QString& sql)
203 {
204  QString s(sql);
205  _sql = s.trimmed();
206  if(_stmt ) {
207  finish();
208  }
209  if(!_sql.isEmpty() ) {
210  int n = 0;
211  int rc;
212  do {
213  rc = sqlite3_prepare_v2(_db, _sql.toUtf8().constData(), -1, &_stmt, 0);
214  if( (rc == SQLITE_BUSY) || (rc == SQLITE_LOCKED) ) {
215  n++;
216  OCC::Utility::usleep(SQLITE_SLEEP_TIME_USEC);
217  }
218  } while( (n < SQLITE_REPEAT_COUNT) && ((rc == SQLITE_BUSY) || (rc == SQLITE_LOCKED)));
219  _errId = rc;
220 
221  if( _errId != SQLITE_OK ) {
222  _error = QString::fromUtf8(sqlite3_errmsg(_db));
223  qWarning() << "Sqlite prepare statement error:" << _error << "in" <<_sql;
224  Q_ASSERT(!"SQLITE Prepare error");
225  }
226  }
227  return _errId;
228 }

240 bool SqlQuery::exec()
241 {
242  if (!_stmt) {
243  return false;
244  }
245 
246  // Don't do anything for selects, that is how we use the lib :-|
247  if( !isSelect() && !isPragma() ) {
248  int rc, n = 0;
249  do {
250  rc = sqlite3_step(_stmt);
251  if( rc == SQLITE_LOCKED ) {
252  rc = sqlite3_reset(_stmt); /* This will also return SQLITE_LOCKED */
253  n++;
254  OCC::Utility::usleep(SQLITE_SLEEP_TIME_USEC);
255  } else if( rc == SQLITE_BUSY ) {
256  OCC::Utility::usleep(SQLITE_SLEEP_TIME_USEC);
257  n++;
258  }
259  } while( (n < SQLITE_REPEAT_COUNT) && ((rc == SQLITE_BUSY) || (rc == SQLITE_LOCKED)));
260  _errId = rc;
261 
262  if (_errId != SQLITE_DONE && _errId != SQLITE_ROW) {
263  _error = QString::fromUtf8(sqlite3_errmsg(_db));
264  qDebug() << "Sqlite exec statement error:" << _errId << _error << "in" <<_sql;
265  }
266  return (_errId == SQLITE_DONE); // either SQLITE_ROW or SQLITE_DONE
267  }
268 
269  return true;
270 }

278 void SqlQuery::bindValue(int pos, const QVariant& value)
279 {
280  int res = -1;
281  Q_ASSERT(_stmt);
282  if( _stmt ) {
283  switch (value.type()) {
284  case QVariant::Int:
285  case QVariant::Bool:
286  res = sqlite3_bind_int(_stmt, pos, value.toInt());
287  break;
288  case QVariant::Double:
289  res = sqlite3_bind_double(_stmt, pos, value.toDouble());
290  break;
291  case QVariant::UInt:
292  case QVariant::LongLong:
293  res = sqlite3_bind_int64(_stmt, pos, value.toLongLong());
294  break;
295  case QVariant::DateTime: {
296  const QDateTime dateTime = value.toDateTime();
297  const QString str = dateTime.toString(QLatin1String("yyyy-MM-ddThh:mm:ss.zzz"));
298  res = sqlite3_bind_text16(_stmt, pos, str.utf16(),
299  str.size() * sizeof(ushort), SQLITE_TRANSIENT);
300  break;
301  }
302  case QVariant::Time: {
303  const QTime time = value.toTime();
304  const QString str = time.toString(QLatin1String("hh:mm:ss.zzz"));
305  res = sqlite3_bind_text16(_stmt, pos, str.utf16(),
306  str.size() * sizeof(ushort), SQLITE_TRANSIENT);
307  break;
308  }
309  case QVariant::String: {
310  if( !value.toString().isNull() ) {
311  // lifetime of string == lifetime of its qvariant
312  const QString *str = static_cast<const QString*>(value.constData());
313  res = sqlite3_bind_text16(_stmt, pos, str->utf16(),
314  (str->size()) * sizeof(QChar), SQLITE_TRANSIENT);
315  } else {
316  res = sqlite3_bind_null(_stmt, pos);
317  }
318  break; }
319  case QVariant::ByteArray: {
320  auto ba = value.toByteArray();
321  res = sqlite3_bind_text(_stmt, pos, ba.constData(), ba.size(), SQLITE_TRANSIENT);
322  break;
323  }
324  default: {
325  QString str = value.toString();
326  // SQLITE_TRANSIENT makes sure that sqlite buffers the data
327  res = sqlite3_bind_text16(_stmt, pos, str.utf16(),
328  (str.size()) * sizeof(QChar), SQLITE_TRANSIENT);
329  break; }
330  }
331  }
332  if (res != SQLITE_OK) {
333  qDebug() << Q_FUNC_INFO << "ERROR" << value << res;
334  }
335  Q_ASSERT( res == SQLITE_OK );
336 }
