94 bool ConfigFile::setConfDir(const QString &value)
95 {
96  QString dirPath = value;
97  if( dirPath.isEmpty() ) return false;
98 
99  QFileInfo fi(dirPath);
100  if ( !fi.exists() ) {
101  QDir().mkpath(dirPath);
102  fi.setFile(dirPath);
103  }
104  if( fi.exists() && fi.isDir() ) {
105  dirPath = fi.absoluteFilePath();
106  qDebug() << "** Using custom config dir " << dirPath;
107  _confDir=dirPath;
108  return true;
109  }
110  return false;
111 }

181 QVariant ConfigFile::getPolicySetting(const QString &setting, const QVariant& defaultValue) const
182 {
183  if (Utility::isWindows()) {
184  // check for policies first and return immediately if a value is found.
185  QSettings userPolicy(QString::fromLatin1("HKEY_CURRENT_USER\\Software\\Policies\\%1\\%2")
186  .arg(APPLICATION_VENDOR).arg(Theme::instance()->appName()),
187  QSettings::NativeFormat);
188  if(userPolicy.contains(setting)) {
189  return userPolicy.value(setting);
190  }
191 
192  QSettings machinePolicy(QString::fromLatin1("HKEY_LOCAL_MACHINE\\Software\\Policies\\%1\\%2")
193  .arg(APPLICATION_VENDOR).arg(APPLICATION_NAME),
194  QSettings::NativeFormat);
195  if(machinePolicy.contains(setting)) {
196  return machinePolicy.value(setting);
197  }
198  }
199  return defaultValue;

225 QString ConfigFile::excludeFile(Scope scope) const
226 {
227  // prefer sync-exclude.lst, but if it does not exist, check for
228  // exclude.lst for compatibility reasons in the user writeable
229  // directories.
230  QFileInfo fi;
231 
232  if (scope != SystemScope) {
233  QFileInfo fi;
234  fi.setFile( configPath(), exclFile );
235 
236  if( ! fi.isReadable() ) {
237  fi.setFile( configPath(), QLatin1String("exclude.lst") );
238  }
239  if( ! fi.isReadable() ) {
240  fi.setFile( configPath(), exclFile );
241  }
242  return fi.absoluteFilePath();
243  } else if (scope != UserScope) {
244  return ConfigFile::excludeFileFromSystem();
245  } else {
246  Q_ASSERT(false);
247  return QString(); // unreachable
248  }
249 }

251 QString ConfigFile::excludeFileFromSystem()
252 {
253  QFileInfo fi;
254 #ifdef Q_OS_WIN
255  fi.setFile( QCoreApplication::applicationDirPath(), exclFile );
256 #endif
257 #ifdef Q_OS_UNIX
258  fi.setFile( QString( SYSCONFDIR "/%1").arg(Theme::instance()->appName()), exclFile );
259  if ( ! fi.exists() ) {
260  // Prefer to return the preferred path! Only use the fallback location
261  // if the other path does not exist and the fallback is valid.
262  QFileInfo nextToBinary( QCoreApplication::applicationDirPath(), exclFile );
263  if (nextToBinary.exists()) {
264  fi = nextToBinary;
265  }
266  }
267 #endif
268 #ifdef Q_OS_MAC
269  // exec path is inside the bundle
270  fi.setFile( QCoreApplication::applicationDirPath(),
271  QLatin1String("../Resources/") + exclFile );
272 #endif
273 
274  return fi.absoluteFilePath();
275 }

330 int ConfigFile::remotePollInterval( const QString& connection ) const
331 {
332  QString con( connection );
333  if( connection.isEmpty() ) con = defaultConnection();
334 
335  QSettings settings(configFile(), QSettings::IniFormat);
336  settings.beginGroup( con );
337 
338  int remoteInterval = settings.value( QLatin1String(remotePollIntervalC), DEFAULT_REMOTE_POLL_INTERVAL ).toInt();
339  if( remoteInterval < 5000) {
340  qDebug() << "Remote Interval is less than 5 seconds, reverting to" << DEFAULT_REMOTE_POLL_INTERVAL;
341  remoteInterval = DEFAULT_REMOTE_POLL_INTERVAL;
342  }
343  return remoteInterval;

472 QVariant ConfigFile::getValue(const QString& param, const QString& group,
473  const QVariant& defaultValue) const
474 {
475  QVariant systemSetting;
476  if (Utility::isMac()) {
477  QSettings systemSettings(QLatin1String("/Library/Preferences/" APPLICATION_REV_DOMAIN ".plist"), QSettings::NativeFormat);
478  if (!group.isEmpty()) {
479  systemSettings.beginGroup(group);
480  }
481  systemSetting = systemSettings.value(param, defaultValue);
482  } else if (Utility::isUnix()) {
483  QSettings systemSettings(QString( SYSCONFDIR "/%1/%1.conf").arg(Theme::instance()->appName()), QSettings::NativeFormat);
484  if (!group.isEmpty()) {
485  systemSettings.beginGroup(group);
486  }
487  systemSetting = systemSettings.value(param, defaultValue);
488  } else { // Windows
489  QSettings systemSettings(QString::fromLatin1("HKEY_LOCAL_MACHINE\\Software\\%1\\%2")
490  .arg(APPLICATION_VENDOR).arg(Theme::instance()->appName()),
491  QSettings::NativeFormat);
492  if (!group.isEmpty()) {
493  systemSettings.beginGroup(group);
494  }
495  systemSetting = systemSettings.value(param, defaultValue);
496  }
497 
498  QSettings settings(configFile(), QSettings::IniFormat);
499  if (!group.isEmpty()) settings.beginGroup(group);
500 
501  return settings.value(param, systemSetting);
502 }
