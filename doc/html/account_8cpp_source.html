115 void Account::setCredentials(AbstractCredentials *cred)
116 {
117  // set active credential manager
118  QNetworkCookieJar *jar = 0;
119  if (_am) {
120  jar = _am->cookieJar();
121  jar->setParent(0);
122 
123  _am = QSharedPointer<QNetworkAccessManager>();
124  }
125 
126  // The order for these two is important! Reading the credential's
127  // settings accesses the account as well as account->_credentials,
128  // so deleteLater must be used.
129  _credentials = QSharedPointer<AbstractCredentials>(cred, &QObject::deleteLater);
130  cred->setAccount(this);
131 
132  _am = QSharedPointer<QNetworkAccessManager>(_credentials->getQNAM(), &QObject::deleteLater);
133 
134  if (jar) {
135  _am->setCookieJar(jar);
136  }
137  connect(_am.data(), SIGNAL(sslErrors(QNetworkReply*,QList<QSslError>)),
138  SLOT(slotHandleSslErrors(QNetworkReply*,QList<QSslError>)));
139  connect(_am.data(), SIGNAL(proxyAuthenticationRequired(QNetworkProxy,QAuthenticator*)),
140  SIGNAL(proxyAuthenticationRequired(QNetworkProxy,QAuthenticator*)));
141  connect(_credentials.data(), SIGNAL(fetched()),
142  SLOT(slotCredentialsFetched()));
143  connect(_credentials.data(), SIGNAL(asked()),
144  SLOT(slotCredentialsAsked()));
145 }

169 void Account::resetNetworkAccessManager()
170 {
171  if (!_credentials || !_am) {
172  return;
173  }
174 
175  qDebug() << "Resetting QNAM";
176  QNetworkCookieJar* jar = _am->cookieJar();
177 
178  // Use a QSharedPointer to allow locking the life of the QNAM on the stack.
179  // Make it call deleteLater to make sure that we can return to any QNAM stack frames safely.
180  _am = QSharedPointer<QNetworkAccessManager>(_credentials->getQNAM(), &QObject::deleteLater);
181 
182  _am->setCookieJar(jar); // takes ownership of the old cookie jar
183  connect(_am.data(), SIGNAL(sslErrors(QNetworkReply*,QList<QSslError>)),
184  SLOT(slotHandleSslErrors(QNetworkReply*,QList<QSslError>)));
185  connect(_am.data(), SIGNAL(proxyAuthenticationRequired(QNetworkProxy,QAuthenticator*)),
186  SIGNAL(proxyAuthenticationRequired(QNetworkProxy,QAuthenticator*)));
187 }

256 QSslConfiguration Account::getOrCreateSslConfig()
257 {
258  if (!_sslConfiguration.isNull()) {
259  // Will be set by CheckServerJob::finished()
260  // We need to use a central shared config to get SSL session tickets
261  return _sslConfiguration;
262  }
263 
264  // if setting the client certificate fails, you will probably get an error similar to this:
265  // "An internal error number 1060 happened. SSL handshake failed, client certificate was requested: SSL error: sslv3 alert handshake failure"
266  QSslConfiguration sslConfig = QSslConfiguration::defaultConfiguration();
267  QSslCertificate sslClientCertificate;
268 
269  ConfigFile cfgFile;
270  if(!cfgFile.certificatePath().isEmpty() && !cfgFile.certificatePasswd().isEmpty()) {
271  resultP12ToPem certif = p12ToPem(cfgFile.certificatePath().toStdString(), cfgFile.certificatePasswd().toStdString());
272  QString s = QString::fromStdString(certif.Certificate);
273  QByteArray ba = s.toLocal8Bit();
274  this->setCertificate(ba, QString::fromStdString(certif.PrivateKey));
275  }
276  if((!_pemCertificate.isEmpty())&&(!_pemPrivateKey.isEmpty())) {
277  // Read certificates
278  QList<QSslCertificate> sslCertificateList = QSslCertificate::fromData(_pemCertificate, QSsl::Pem);
279  if(sslCertificateList.length() != 0) {
280  sslClientCertificate = sslCertificateList.takeAt(0);
281  }
282  // Read key from file
283  QSslKey privateKey(_pemPrivateKey.toLocal8Bit(), QSsl::Rsa, QSsl::Pem, QSsl::PrivateKey , "");
284 
285  // SSL configuration
286  sslConfig.setCaCertificates(QSslSocket::systemCaCertificates());
287  sslConfig.setLocalCertificate(sslClientCertificate);
288  sslConfig.setPrivateKey(privateKey);
289  qDebug() << "Added SSL client certificate to the query";
290  }
291 
292 #if QT_VERSION > QT_VERSION_CHECK(5, 2, 0)
293  // Try hard to re-use session for different requests
294  sslConfig.setSslOption(QSsl::SslOptionDisableSessionTickets, false);
295  sslConfig.setSslOption(QSsl::SslOptionDisableSessionSharing, false);
296  sslConfig.setSslOption(QSsl::SslOptionDisableSessionPersistence, false);
297 #endif
298 
299  return sslConfig;
300 }

348 void Account::slotHandleSslErrors(QNetworkReply *reply , QList<QSslError> errors)
349 {
350  NetworkJobTimeoutPauser pauser(reply);
351  QString out;
352  QDebug(&out) << "SSL-Errors happened for url " << reply->url().toString();
353  foreach(const QSslError &error, errors) {
354  QDebug(&out) << "\tError in " << error.certificate() << ":"
355  << error.errorString() << "("<< error.error() << ")" << "\n";
356  }
357 
358  bool allPreviouslyRejected = true;
359  foreach (const QSslError &error, errors) {
360  if (!_rejectedCertificates.contains(error.certificate())) {
361  allPreviouslyRejected = false;
362  }
363  }
364 
365  // If all certs have previously been rejected by the user, don't ask again.
366  if( allPreviouslyRejected ) {
367  qDebug() << out << "Certs not trusted by user decision, returning.";
368  return;
369  }
370 
371  QList<QSslCertificate> approvedCerts;
372  if (_sslErrorHandler.isNull() ) {
373  qDebug() << out << Q_FUNC_INFO << "called without valid SSL error handler for account" << url();
374  return;
375  }
376 
377  // SslDialogErrorHandler::handleErrors will run an event loop that might execute
378  // the deleteLater() of the QNAM before we have the chance of unwinding our stack.
379  // Keep a ref here on our stackframe to make sure that it doesn't get deleted before
380  // handleErrors returns.
381  QSharedPointer<QNetworkAccessManager> qnamLock = _am;
382  QPointer<QObject> guard = reply;
383 
384  if (_sslErrorHandler->handleErrors(errors, reply->sslConfiguration(), &approvedCerts, sharedFromThis())) {
385  if (!guard) return;
386 
387  QSslSocket::addDefaultCaCertificates(approvedCerts);
388  addApprovedCerts(approvedCerts);
389  emit wantsAccountSaved(this);
390  // all ssl certs are known and accepted. We can ignore the problems right away.
391 // qDebug() << out << "Certs are known and trusted! This is not an actual error.";
392 
393  // Warning: Do *not* use ignoreSslErrors() (without args) here:
394  // it permanently ignores all SSL errors for this host, even
395  // certificate changes.
396  reply->ignoreSslErrors(errors);
397  } else {
398  if (!guard) return;
399 
400  // Mark all involved certificates as rejected, so we don't ask the user again.
401  foreach (const QSslError &error, errors) {
402  if (!_rejectedCertificates.contains(error.certificate())) {
403  _rejectedCertificates.append(error.certificate());
404  }
405  }
406  // if during normal operation, a new certificate was MITM'ed, and the user does not
407  // ACK it, the running request must be aborted and the QNAM must be reset, to not
408  // treat the new cert as granted. See bug #3283
409  reply->abort();
410  resetNetworkAccessManager();
411  return;
412  }
413 }
