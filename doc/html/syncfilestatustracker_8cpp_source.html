23 static SyncFileStatus::SyncFileStatusTag lookupProblem(const QString &pathToMatch, const std::map<QString, SyncFileStatus::SyncFileStatusTag> &problemMap)
24 {
25  auto lower = problemMap.lower_bound(pathToMatch);
26  for (auto it = lower; it != problemMap.cend(); ++it) {
27  const QString &problemPath = it->first;
28  SyncFileStatus::SyncFileStatusTag severity = it->second;
29  // qDebug() << Q_FUNC_INFO << pathToMatch << severity << problemPath;
30  if (problemPath == pathToMatch) {
31  return severity;
32  } else if (severity == SyncFileStatus::StatusError
33  && problemPath.startsWith(pathToMatch)
34  && (pathToMatch.isEmpty() || problemPath.at(pathToMatch.size()) == '/')) {
35  return SyncFileStatus::StatusWarning;
36  } else if (!problemPath.startsWith(pathToMatch)) {
37  // Starting at lower_bound we get the first path that is not smaller,
38  // since: "a/" < "a/aa" < "a/aa/aaa" < "a/ab/aba"
39  // If problemMap keys are ["a/aa/aaa", "a/ab/aba"] and pathToMatch == "a/aa",
40  // lower_bound(pathToMatch) will point to "a/aa/aaa", and the moment that
41  // problemPath.startsWith(pathToMatch) == false, we know that we've looked
42  // at everything that interest us.
43  break;
44  }
45  }
46  return SyncFileStatus::StatusNone;
47 }

87 SyncFileStatus SyncFileStatusTracker::fileStatus(const QString& relativePath)
88 {
89  Q_ASSERT(!relativePath.endsWith(QLatin1Char('/')));
90 
91  if (relativePath.isEmpty()) {
92  // This is the root sync folder, it doesn't have an entry in the database and won't be walked by csync, so resolve manually.
93  return resolveSyncAndErrorStatus(QString(), NotShared);
94  }
95 
96  // The SyncEngine won't notify us at all for CSYNC_FILE_SILENTLY_EXCLUDED
97  // and CSYNC_FILE_EXCLUDE_AND_REMOVE excludes. Even though it's possible
98  // that the status of CSYNC_FILE_EXCLUDE_LIST excludes will change if the user
99  // update the exclude list at runtime and doing it statically here removes
100  // our ability to notify changes through the fileStatusChanged signal,
101  // it's an acceptable compromize to treat all exclude types the same.
102  if( _syncEngine->excludedFiles().isExcluded(_syncEngine->localPath() + relativePath,
103  _syncEngine->localPath(),
104  _syncEngine->ignoreHiddenFiles()) ) {
105  return SyncFileStatus(SyncFileStatus::StatusWarning);
106  }
107 
108  if ( _dirtyPaths.contains(relativePath) )
109  return SyncFileStatus::StatusSync;
110 
111  // First look it up in the database to know if it's shared
112  SyncJournalFileRecord rec = _syncEngine->journal()->getFileRecord(relativePath);
113  if (rec.isValid()) {
114  return resolveSyncAndErrorStatus(relativePath, rec._remotePerm.contains("S") ? Shared : NotShared);
115  }
116 
117  // Must be a new file not yet in the database, check if it's syncing or has an error.
118  return resolveSyncAndErrorStatus(relativePath, NotShared, PathUnknown);
119 }

132 void SyncFileStatusTracker::incSyncCount(const QString &relativePath, EmitStatusChangeFlag emitStatusChange)
133 {
134  // Will return 0 (and increase to 1) if the path wasn't in the map yet
135  int count = _syncCount[relativePath]++;
136  if (!count) {
137  if (emitStatusChange)
138  emit fileStatusChanged(getSystemDestination(relativePath), fileStatus(relativePath));
139 
140  // We passed from OK to SYNC, increment the parent to keep it marked as
141  // SYNC while we propagate ourselves and our own children.
142  Q_ASSERT(!relativePath.endsWith('/'));
143  int lastSlashIndex = relativePath.lastIndexOf('/');
144  if (lastSlashIndex != -1)
145  incSyncCount(relativePath.left(lastSlashIndex), EmitStatusChange);
146  else if (!relativePath.isEmpty())
147  incSyncCount(QString(), EmitStatusChange);
148  }
149 }

151 void SyncFileStatusTracker::decSyncCount(const QString &relativePath, EmitStatusChangeFlag emitStatusChange)
152 {
153  int count = --_syncCount[relativePath];
154  if (!count) {
155  // Remove from the map, same as 0
156  _syncCount.remove(relativePath);
157 
158  if (emitStatusChange)
159  emit fileStatusChanged(getSystemDestination(relativePath), fileStatus(relativePath));
160 
161  // We passed from SYNC to OK, decrement our parent.
162  Q_ASSERT(!relativePath.endsWith('/'));
163  int lastSlashIndex = relativePath.lastIndexOf('/');
164  if (lastSlashIndex != -1)
165  decSyncCount(relativePath.left(lastSlashIndex), EmitStatusChange);
166  else if (!relativePath.isEmpty())
167  decSyncCount(QString(), EmitStatusChange);
168  }
169 }

171 void SyncFileStatusTracker::slotAboutToPropagate(SyncFileItemVector& items)
172 {
173  Q_ASSERT(_syncCount.isEmpty());
174 
175  std::map<QString, SyncFileStatus::SyncFileStatusTag> oldProblems;
176  std::swap(_syncProblems, oldProblems);
177 
178  foreach (const SyncFileItemPtr &item, items) {
179  // qDebug() << Q_FUNC_INFO << "Investigating" << item->destination() << item->_status << item->_instruction;
180 
181  if (showErrorInSocketApi(*item)) {
182  _syncProblems[item->_file] = SyncFileStatus::StatusError;
183  invalidateParentPaths(item->destination());
184  } else if (showWarningInSocketApi(*item)) {
185  _syncProblems[item->_file] = SyncFileStatus::StatusWarning;
186  }
187 
188  // Mark this path as syncing for instructions that will result in propagation,
189  // but DontEmitStatusChange since we're going to emit for ourselves using the
190  // info in the SyncFileItem we received, parents will still be emit if needed.
191  if (item->_instruction != CSYNC_INSTRUCTION_NONE
192  && item->_instruction != CSYNC_INSTRUCTION_UPDATE_METADATA
193  && item->_instruction != CSYNC_INSTRUCTION_IGNORE
194  && item->_instruction != CSYNC_INSTRUCTION_ERROR) {
195  incSyncCount(item->destination(), DontEmitStatusChange);
196  }
197 
198  _dirtyPaths.remove(item->destination());
199  emit fileStatusChanged(getSystemDestination(item->destination()), resolveSyncAndErrorStatus(item->destination(), item->_remotePerm.contains("S") ? Shared : NotShared));
200  }
201 
202  // Some metadata status won't trigger files to be synced, make sure that we
203  // push the OK status for dirty files that don't need to be propagated.
204  // Swap into a copy since fileStatus() reads _dirtyPaths to determine the status
205  QSet<QString> oldDirtyPaths;
206  std::swap(_dirtyPaths, oldDirtyPaths);
207  for (auto it = oldDirtyPaths.constBegin(); it != oldDirtyPaths.constEnd(); ++it)
208  emit fileStatusChanged(getSystemDestination(*it), fileStatus(*it));
209 
210  // Make sure to push any status that might have been resolved indirectly since the last sync
211  // (like an error file being deleted from disk)
212  for (auto it = _syncProblems.begin(); it != _syncProblems.end(); ++it)
213  oldProblems.erase(it->first);
214  for (auto it = oldProblems.begin(); it != oldProblems.end(); ++it) {
215  const QString &path = it->first;
216  SyncFileStatus::SyncFileStatusTag severity = it->second;
217  if (severity == SyncFileStatus::StatusError)
218  invalidateParentPaths(path);
219  emit fileStatusChanged(getSystemDestination(path), fileStatus(path));
220  }
221 }

223 void SyncFileStatusTracker::slotItemCompleted(const SyncFileItem &item)
224 {
225  // qDebug() << Q_FUNC_INFO << item.destination() << item._status << item._instruction;
226 
227  if (showErrorInSocketApi(item)) {
228  _syncProblems[item._file] = SyncFileStatus::StatusError;
229  invalidateParentPaths(item.destination());
230  } else if (showWarningInSocketApi(item)) {
231  _syncProblems[item._file] = SyncFileStatus::StatusWarning;
232  } else {
233  _syncProblems.erase(item._file);
234  }
235 
236  // decSyncCount calls *must* be symetric with incSyncCount calls in slotAboutToPropagate
237  if (item._instruction != CSYNC_INSTRUCTION_NONE
238  && item._instruction != CSYNC_INSTRUCTION_UPDATE_METADATA
239  && item._instruction != CSYNC_INSTRUCTION_IGNORE
240  && item._instruction != CSYNC_INSTRUCTION_ERROR) {
241  decSyncCount(item.destination(), DontEmitStatusChange);
242  }
243  emit fileStatusChanged(getSystemDestination(item.destination()), resolveSyncAndErrorStatus(item.destination(), item._remotePerm.contains("S") ? Shared : NotShared));
244 }

246 void SyncFileStatusTracker::slotSyncFinished()
247 {
248  // Clear the sync counts to reduce the impact of unsymetrical inc/dec calls (e.g. when directory job abort)
249  QHash<QString, int> oldSyncCount;
250  std::swap(_syncCount, oldSyncCount);
251  for (auto it = oldSyncCount.begin(); it != oldSyncCount.end(); ++it)
252  emit fileStatusChanged(getSystemDestination(it.key()), fileStatus(it.key()));

260 SyncFileStatus SyncFileStatusTracker::resolveSyncAndErrorStatus(const QString &relativePath, SharedFlag isShared, PathKnownFlag isPathKnown)
261 {
262  // If it's a new file and that we're not syncing it yet,
263  // don't show any icon and wait for the filesystem watcher to trigger a sync.
264  SyncFileStatus status(isPathKnown ? SyncFileStatus::StatusUpToDate : SyncFileStatus::StatusNone);
265  if (_syncCount.value(relativePath)) {
266  status.set(SyncFileStatus::StatusSync);
267  } else {
268  // After a sync finished, we need to show the users issues from that last sync like the activity list does.
269  // Also used for parent directories showing a warning for an error child.
270  SyncFileStatus::SyncFileStatusTag problemStatus = lookupProblem(relativePath, _syncProblems);
271  if (problemStatus != SyncFileStatus::StatusNone)
272  status.set(problemStatus);
273  }
274 
275  if (isShared)
276  status.setSharedWithMe(true);
277 
278  return status;
279 }

281 void SyncFileStatusTracker::invalidateParentPaths(const QString& path)
282 {
283  QStringList splitPath = path.split('/', QString::SkipEmptyParts);
284  for (int i = 0; i < splitPath.size(); ++i) {
285  QString parentPath = QStringList(splitPath.mid(0, i)).join(QLatin1String("/"));
286  emit fileStatusChanged(getSystemDestination(parentPath), fileStatus(parentPath));
287  }

291 {
