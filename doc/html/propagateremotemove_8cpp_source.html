72 void PropagateRemoteMove::start()
73 {
74  if (_propagator->_abortRequested.fetchAndAddRelaxed(0))
75  return;
76 
77  qDebug() << Q_FUNC_INFO << _item->_file << _item->_renameTarget;
78 
79  QString targetFile(_propagator->getFilePath(_item->_renameTarget));
80 
81  if (_item->_file == _item->_renameTarget) {
82  // The parent has been renamed already so there is nothing more to do.
83  finalize();
84  return;
85  }
86  if (_item->_file == QLatin1String("Shared") ) {
87  // Before owncloud 7, there was no permissions system. At the time all the shared files were
88  // in a directory called "Shared" and were not supposed to be moved, otherwise bad things happened
89 
90  QString versionString = _propagator->account()->serverVersion();
91  if (versionString.contains('.') && versionString.split('.')[0].toInt() < 7) {
92  QString originalFile(_propagator->getFilePath(QLatin1String("Shared")));
93  emit _propagator->touchedFile(originalFile);
94  emit _propagator->touchedFile(targetFile);
95  QString renameError;
96  if( FileSystem::rename(targetFile, originalFile, &renameError) ) {
97  done(SyncFileItem::NormalError, tr("This folder must not be renamed. It is renamed back to its original name."));
98  } else {
99  done(SyncFileItem::NormalError, tr("This folder must not be renamed. Please name it back to Shared."));
100  }
101  return;
102  }
103  }
104 
105  QString destination = QDir::cleanPath(_propagator->account()->url().path() + QLatin1Char('/')
106  + _propagator->account()->davPath() + _propagator->_remoteFolder + _item->_renameTarget);
107  _job = new MoveJob(_propagator->account(),
108  _propagator->_remoteFolder + _item->_file,
109  destination, this);
110  connect(_job, SIGNAL(finishedSignal()), this, SLOT(slotMoveJobFinished()));
111  _propagator->_activeJobList.append(this);
112  _job->start();
113 
114 }

122 void PropagateRemoteMove::slotMoveJobFinished()
123 {
124  _propagator->_activeJobList.removeOne(this);
125 
126  Q_ASSERT(_job);
127 
128  qDebug() << Q_FUNC_INFO << _job->reply()->request().url() << "FINISHED WITH STATUS"
129  << _job->reply()->error()
130  << (_job->reply()->error() == QNetworkReply::NoError ? QLatin1String("") : _job->reply()->errorString());
131 
132  QNetworkReply::NetworkError err = _job->reply()->error();
133  _item->_httpErrorCode = _job->reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
134 
135  if (err != QNetworkReply::NoError) {
136 
137  if( checkForProblemsWithShared(_item->_httpErrorCode,
138  tr("The file was renamed but is part of a read only share. The original file was restored."))) {
139  return;
140  }
141 
142  SyncFileItem::Status status = classifyError(err, _item->_httpErrorCode,
143  &_propagator->_anotherSyncNeeded);
144  done(status, _job->errorString());
145  return;
146  }
147 
148  _item->_requestDuration = _job->duration();
149  _item->_responseTimeStamp = _job->responseTimestamp();
150 
151  if (_item->_httpErrorCode != 201 ) {
152  // Normally we expect "201 Created"
153  // If it is not the case, it might be because of a proxy or gateway intercepting the request, so we must
154  // throw an error.
155  done(SyncFileItem::NormalError, tr("Wrong HTTP code returned by server. Expected 201, but received \"%1 %2\".")
156  .arg(_item->_httpErrorCode).arg(_job->reply()->attribute(QNetworkRequest::HttpReasonPhraseAttribute).toString()));
157  return;
158  }
159 
160  finalize();
161 
162 }

164 void PropagateRemoteMove::finalize()
165 {
166  SyncJournalFileRecord oldRecord =
167  _propagator->_journal->getFileRecord(_item->_originalFile);
168  // if reading from db failed still continue hoping that deleteFileRecord
169  // reopens the db successfully.
170  // The db is only queried to transfer the content checksum from the old
171  // to the new record. It is not a problem to skip it here.
172  _propagator->_journal->deleteFileRecord(_item->_originalFile);
173 
174  SyncJournalFileRecord record(*_item, _propagator->getFilePath(_item->_renameTarget));
175  record._path = _item->_renameTarget;
176  if (oldRecord.isValid()) {
177  record._contentChecksum = oldRecord._contentChecksum;
178  record._contentChecksumType = oldRecord._contentChecksumType;
179  if (record._fileSize != oldRecord._fileSize) {
180  qDebug() << "Warning: file sizes differ on server vs sync journal: " << record._fileSize << oldRecord._fileSize;
181  record._fileSize = oldRecord._fileSize; // server might have claimed different size, we take the old one from the DB
182  }
183  }
184 
185  if (!_propagator->_journal->setFileRecord(record)) {
186  done(SyncFileItem::FatalError, tr("Error writing metadata to the database"));
187  return;
188  }
189 
190  if (_item->_isDirectory) {
191  if (!adjustSelectiveSync(_propagator->_journal, _item->_file, _item->_renameTarget)) {
192  done(SyncFileItem::FatalError, tr("Error writing metadata to the database"));
193  return;
194  }
195  }
196 
197  _propagator->_journal->commit("Remote Rename");
198  done(SyncFileItem::Success);
199 }

201 bool PropagateRemoteMove::adjustSelectiveSync(SyncJournalDb *journal, const QString &from_, const QString &to_)
202 {
203  bool ok;
204  // We only care about preserving the blacklist. The white list should anyway be empty.
205  // And the undecided list will be repopulated on the next sync, if there is anything too big.
206  QStringList list = journal->getSelectiveSyncList(SyncJournalDb::SelectiveSyncBlackList, &ok);
207  if (!ok)
208  return false;
209 
210  bool changed = false;
211  Q_ASSERT(!from_.endsWith(QLatin1String("/")));
212  Q_ASSERT(!to_.endsWith(QLatin1String("/")));
213  QString from = from_ + QLatin1String("/");
214  QString to = to_ + QLatin1String("/");
215 
216  for (auto it = list.begin(); it != list.end(); ++it) {
217  if (it->startsWith(from)) {
218  *it = it->replace(0, from.size(), to);
219  changed = true;
220  }
221  }
222 
223  if (changed) {
224  journal->setSelectiveSyncList(SyncJournalDb::SelectiveSyncBlackList, list);
225  }
226  return true;
227 }
