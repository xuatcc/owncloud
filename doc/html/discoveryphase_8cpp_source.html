27 /* Given a sorted list of paths ending with '/', return whether or not the given path is within one of the paths of the list*/
28 static bool findPathInList(const QStringList &list, const QString &path)
29 {
30  Q_ASSERT(std::is_sorted(list.begin(), list.end()));
31 
32  if (list.size() == 1 && list.first() == QLatin1String("/")) {
33  // Special case for the case "/" is there, it matches everything
34  return true;
35  }
36 
37  QString pathSlash = path + QLatin1Char('/');
38 
39  // Since the list is sorted, we can do a binary search.
40  // If the path is a prefix of another item or right after in the lexical order.
41  auto it = std::lower_bound(list.begin(), list.end(), pathSlash);
42 
43  if (it != list.end() && *it == pathSlash) {
44  return true;
45  }
46 
47  if (it == list.begin()) {
48  return false;
49  }
50  --it;
51  Q_ASSERT(it->endsWith(QLatin1Char('/'))); // Folder::setSelectiveSyncBlackList makes sure of that
52  return pathSlash.startsWith(*it);
53 }

55 bool DiscoveryJob::isInSelectiveSyncBlackList(const char *path) const
56 {
57  if (_selectiveSyncBlackList.isEmpty()) {
58  // If there is no black list, everything is allowed
59  return false;
60  }
61 
62  // Block if it is in the black list
63  if (findPathInList(_selectiveSyncBlackList, QString::fromUtf8(path))) {
64  return true;
65  }
66 
67  // Also try to adjust the path if there was renames
68  if (csync_rename_count(_csync_ctx)) {
69  QScopedPointer<char, QScopedPointerPodDeleter> adjusted(
70  csync_rename_adjust_path_source(_csync_ctx, path));
71  if (strcmp(adjusted.data(), path) != 0) {
72  return findPathInList(_selectiveSyncBlackList, QString::fromUtf8(adjusted.data()));
73  }
74  }
75 
76  return false;
77 }

84 bool DiscoveryJob::checkSelectiveSyncNewFolder(const QString& path)
85 {
86  // If this path or the parent is in the white list, then we do not block this file
87  if (findPathInList(_selectiveSyncWhiteList, path)) {
88  return false;
89  }
90 
91  if (_newBigFolderSizeLimit < 0) {
92  // no limit, everything is allowed;
93  return false;
94  }
95 
96  // Go in the main thread to do a PROPFIND to know the size of this folder
97  qint64 result = -1;
98 
99  {
100  QMutexLocker locker(&_vioMutex);
101  emit doGetSizeSignal(path, &result);
102  _vioWaitCondition.wait(&_vioMutex);
103  }
104 
105  auto limit = _newBigFolderSizeLimit;
106  if (result >= limit) {
107  // we tell the UI there is a new folder
108  emit newBigFolder(path);
109  return true;
110  } else {
111  // it is not too big, put it in the white list (so we will not do more query for the children)
112  // and and do not block.
113  auto p = path;
114  if (!p.endsWith(QLatin1Char('/'))) { p += QLatin1Char('/'); }
115  _selectiveSyncWhiteList.insert(std::upper_bound(_selectiveSyncWhiteList.begin(),
116  _selectiveSyncWhiteList.end(), p), p);
117 
118  return false;
119  }
120 }

152 // Only use for error cases! It will always set an error errno
153 int get_errno_from_http_errcode( int err, const QString & reason ) {
154  int new_errno = EIO;
155 
156  switch(err) {
157  case 401: /* Unauthorized */
158  case 402: /* Payment Required */
159  case 407: /* Proxy Authentication Required */
160  case 405:
161  new_errno = EPERM;
162  break;
163  case 301: /* Moved Permanently */
164  case 303: /* See Other */
165  case 404: /* Not Found */
166  case 410: /* Gone */
167  new_errno = ENOENT;
168  break;
169  case 408: /* Request Timeout */
170  case 504: /* Gateway Timeout */
171  new_errno = EAGAIN;
172  break;
173  case 423: /* Locked */
174  new_errno = EACCES;
175  break;
176  case 403: /* Forbidden */
177  new_errno = ERRNO_FORBIDDEN;
178  break;
179  case 400: /* Bad Request */
180  case 409: /* Conflict */
181  case 411: /* Length Required */
182  case 412: /* Precondition Failed */
183  case 414: /* Request-URI Too Long */
184  case 415: /* Unsupported Media Type */
185  case 424: /* Failed Dependency */
186  case 501: /* Not Implemented */
187  new_errno = EINVAL;
188  break;
189  case 507: /* Insufficient Storage */
190  new_errno = ENOSPC;
191  break;
192  case 206: /* Partial Content */
193  case 300: /* Multiple Choices */
194  case 302: /* Found */
195  case 305: /* Use Proxy */
196  case 306: /* (Unused) */
197  case 307: /* Temporary Redirect */
198  case 406: /* Not Acceptable */
199  case 416: /* Requested Range Not Satisfiable */
200  case 417: /* Expectation Failed */
201  case 422: /* Unprocessable Entity */
202  case 500: /* Internal Server Error */
203  case 502: /* Bad Gateway */
204  case 505: /* HTTP Version Not Supported */
205  new_errno = EIO;
206  break;
207  case 503: /* Service Unavailable */
208  // https://github.com/owncloud/core/pull/26145/files
209  if (reason == "Storage not available" || reason == "Storage is temporarily not available") {
210  new_errno = ERRNO_STORAGE_UNAVAILABLE;
211  } else {
212  new_errno = ERRNO_SERVICE_UNAVAILABLE;
213  }
214  break;
215  case 413: /* Request Entity too Large */
216  new_errno = EFBIG;
217  break;
218  default:
219  new_errno = EIO;
220  }
221  return new_errno;
222 }

261 static csync_vio_file_stat_t* propertyMapToFileStat(const QMap<QString,QString> &map)
262 {
263  csync_vio_file_stat_t* file_stat = csync_vio_file_stat_new();
264 
265  for (auto it = map.constBegin(); it != map.constEnd(); ++it) {
266  //qDebug() << it.key() << it.value();
267  QString property = it.key();
268  QString value = it.value();
269  if (property == "resourcetype") {
270  if (value.contains("collection")) {
271  file_stat->type = CSYNC_VIO_FILE_TYPE_DIRECTORY;
272  } else {
273  file_stat->type = CSYNC_VIO_FILE_TYPE_REGULAR;
274  }
275  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_TYPE;
276  } else if (property == "getlastmodified") {
277  file_stat->mtime = oc_httpdate_parse(value.toUtf8());
278  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_MTIME;
279  } else if (property == "getcontentlength") {
280  bool ok = false;
281  qlonglong ll = value.toLongLong(&ok);
282  if (ok && ll >= 0) {
283  file_stat->size = ll;
284  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_SIZE;
285  }
286  } else if (property == "getetag") {
287  file_stat->etag = csync_normalize_etag(value.toUtf8());
288  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_ETAG;
289  } else if (property == "id") {
290  csync_vio_file_stat_set_file_id(file_stat, value.toUtf8());
291  } else if (property == "downloadURL") {
292  file_stat->directDownloadUrl = strdup(value.toUtf8());
293  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_DIRECTDOWNLOADURL;
294  } else if (property == "dDC") {
295  file_stat->directDownloadCookies = strdup(value.toUtf8());
296  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_DIRECTDOWNLOADCOOKIES;
297  } else if (property == "permissions") {
298  auto v = value.toUtf8();
299  if (value.isEmpty()) {
300  // special meaning for our code: server returned permissions but are empty
301  // meaning only reading is allowed for this resource
302  file_stat->remotePerm[0] = ' ';
303  // see _csync_detect_update()
304  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_PERM;
305  } else if (v.length() < int(sizeof(file_stat->remotePerm))) {
306  strcpy(file_stat->remotePerm, v.constData());
307  file_stat->fields |= CSYNC_VIO_FILE_STAT_FIELDS_PERM;
308  } else {
309  qWarning() << "permissions too large" << v;
310  }
311  }
312  }
313 
314  return file_stat;
315 }

317 void DiscoverySingleDirectoryJob::directoryListingIteratedSlot(QString file, const QMap<QString,QString> &map)
318 {
319  //qDebug() << Q_FUNC_INFO << _subPath << file << map.count() << map.keys() << _account->davPath() << _lsColJob->reply()->request().url().path();
320  if (!_ignoredFirst) {
321  // The first entry is for the folder itself, we should process it differently.
322  _ignoredFirst = true;
323  if (map.contains("permissions")) {
324  emit firstDirectoryPermissions(map.value("permissions"));
325  }
326  if (map.contains("data-fingerprint")) {
327  _dataFingerprint = map.value("data-fingerprint").toUtf8();
328  }
329  } else {
330  // Remove <webDAV-Url>/folder/ from <webDAV-Url>/folder/subfile.txt
331  file.remove(0, _lsColJob->reply()->request().url().path().length());
332  // remove trailing slash
333  while (file.endsWith('/')) {
334  file.chop(1);
335  }
336  // remove leading slash
337  while (file.startsWith('/')) {
338  file = file.remove(0, 1);
339  }
340 
341 
342  FileStatPointer file_stat(propertyMapToFileStat(map));
343  file_stat->name = strdup(file.toUtf8());
344  if (!file_stat->etag || strlen(file_stat->etag) == 0) {
345  qDebug() << "WARNING: etag of" << file_stat->name << "is" << file_stat->etag << " This must not happen.";
346  }
347 
348  QStringRef fileRef(&file);
349  int slashPos = file.lastIndexOf(QLatin1Char('/'));
350  if( slashPos > -1 ) {
351  fileRef = file.midRef(slashPos+1);
352  }
353  //qDebug() << "!!!!" << file_stat << file_stat->name << file_stat->file_id << map.count();
354  _results.append(file_stat);
355  }
356 
357  //This works in concerto with the RequestEtagJob and the Folder object to check if the remote folder changed.
358  if (map.contains("getetag")) {
359  _etagConcatenation += map.value("getetag");
360 
361  if (_firstEtag.isEmpty()) {
362  _firstEtag = map.value("getetag"); // for directory itself
363  }
364  }
365 }

382 void DiscoverySingleDirectoryJob::lsJobFinishedWithErrorSlot(QNetworkReply *r)
383 {
384  QString contentType = r->header(QNetworkRequest::ContentTypeHeader).toString();
385  int httpCode = r->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
386  QString httpReason = r->attribute(QNetworkRequest::HttpReasonPhraseAttribute).toString();
387  QString msg = r->errorString();
388  int errnoCode = EIO; // Something went wrong
389  qDebug() << Q_FUNC_INFO << r->errorString() << httpCode << r->error();
390  if (httpCode != 0 && httpCode != 207) {
391  errnoCode = get_errno_from_http_errcode(httpCode, httpReason);
392  } else if (r->error() != QNetworkReply::NoError) {
393  errnoCode = EIO;
394  } else if (!contentType.contains("application/xml; charset=utf-8")) {
395  msg = QLatin1String("Server error: PROPFIND reply is not XML formatted!");
396  errnoCode = ERRNO_WRONG_CONTENT;
397  } else {
398  // Default keep at EIO, see above
399  }
400 
401  emit finishedWithError(errnoCode == 0 ? EIO : errnoCode, msg);
402  deleteLater();
403 }

419 void DiscoveryMainThread::doOpendirSlot(const QString &subPath, DiscoveryDirectoryResult *r)
420 {
421  QString fullPath = _pathPrefix;
422  if (!_pathPrefix.endsWith('/')) {
423  fullPath += '/';
424  }
425  fullPath += subPath;
426  // remove trailing slash
427  while (fullPath.endsWith('/')) {
428  fullPath.chop(1);
429  }
430 
431  // emit _discoveryJob->folderDiscovered(false, subPath);
432  _discoveryJob->update_job_update_callback (false, subPath.toUtf8(), _discoveryJob);
433 
434  // Result gets written in there
435  _currentDiscoveryDirectoryResult = r;
436  _currentDiscoveryDirectoryResult->path = fullPath;
437 
438  // Schedule the DiscoverySingleDirectoryJob
439  _singleDirJob = new DiscoverySingleDirectoryJob(_account, fullPath, this);
440  QObject::connect(_singleDirJob, SIGNAL(finishedWithResult(const QList<FileStatPointer> &)),
441  this, SLOT(singleDirectoryJobResultSlot(const QList<FileStatPointer> &)));

585 csync_vio_handle_t* DiscoveryJob::remote_vio_opendir_hook (const char *url,
586  void *userdata)
587 {
588  DiscoveryJob *discoveryJob = static_cast<DiscoveryJob*>(userdata);
589  if (discoveryJob) {
590  qDebug() << discoveryJob << url << "Calling into main thread...";
591 
592  QScopedPointer<DiscoveryDirectoryResult> directoryResult(new DiscoveryDirectoryResult());
593  directoryResult->code = EIO;
594 
595  discoveryJob->_vioMutex.lock();
596  const QString qurl = QString::fromUtf8(url);
597  emit discoveryJob->doOpendirSignal(qurl, directoryResult.data());
598  discoveryJob->_vioWaitCondition.wait(&discoveryJob->_vioMutex, ULONG_MAX); // FIXME timeout?
599  discoveryJob->_vioMutex.unlock();
600 
601  qDebug() << discoveryJob << url << "...Returned from main thread";
602 
603  // Upon awakening from the _vioWaitCondition, iterator should be a valid iterator.
604  if (directoryResult->code != 0) {
605  qDebug() << directoryResult->code << "when opening" << url << "msg=" << directoryResult->msg;
606  errno = directoryResult->code;
607  // save the error string to the context
608  discoveryJob->_csync_ctx->error_string = qstrdup( directoryResult->msg.toUtf8().constData() );
609  return NULL;
610  }
611 
612  return directoryResult.take();
613  }
614  return NULL;
615 }
