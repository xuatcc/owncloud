30 bool Capabilities::shareAPI() const
31 {
32  if (_capabilities["files_sharing"].toMap().contains("api_enabled")) {
33  return _capabilities["files_sharing"].toMap()["api_enabled"].toBool();

100 QByteArray Capabilities::uploadChecksumType() const
101 {
102  QByteArray preferred = preferredUploadChecksumType();
103  if (!preferred.isEmpty())
104  return preferred;
105  QList<QByteArray> supported = supportedChecksumTypes();
106  if (!supported.isEmpty())
107  return supported.first();
108  return QByteArray();
