57 void setLaunchOnStartup_private(const QString &appName, const QString& guiName, bool enable)
58 {
59  QString userAutoStartPath = getUserAutostartDir_private();
60  QString desktopFileLocation = userAutoStartPath+appName+QLatin1String(".desktop");
61  if (enable) {
62  if (!QDir().exists(userAutoStartPath) && !QDir().mkpath(userAutoStartPath)) {
63  qDebug() << "Could not create autostart folder";
64  return;
65  }
66  QFile iniFile(desktopFileLocation);
67  if (!iniFile.open(QIODevice::WriteOnly)) {
68  qDebug() << "Could not write auto start entry" << desktopFileLocation;
69  return;
70  }
71  QTextStream ts(&iniFile);
72  ts.setCodec("UTF-8");
73  ts << QLatin1String("[Desktop Entry]") << endl
74  << QLatin1String("Name=") << guiName << endl
75  << QLatin1String("GenericName=") << QLatin1String("File Synchronizer") << endl
76  << QLatin1String("Exec=") << QCoreApplication::applicationFilePath() << endl
77  << QLatin1String("Terminal=") << "false" << endl
78  << QLatin1String("Icon=") << appName.toLower() << endl // always use lowercase for icons
79  << QLatin1String("Categories=") << QLatin1String("Network") << endl
80  << QLatin1String("Type=") << QLatin1String("Application") << endl
81  << QLatin1String("StartupNotify=") << "false" << endl
82  << QLatin1String("X-GNOME-Autostart-enabled=") << "true" << endl
83  ;
84  } else {
85  if (!QFile::remove(desktopFileLocation)) {
86  qDebug() << "Could not remove autostart desktop file";
87  }
88  }
89 }
