21 static void setupFavLink_private(const QString &folder)
22 {
23  // Windows Explorer: Place under "Favorites" (Links)
24  wchar_t path[MAX_PATH];
25  SHGetSpecialFolderPath(0, path, CSIDL_PROFILE, FALSE);
26  QString profile = QDir::fromNativeSeparators(QString::fromWCharArray(path));
27  QDir folderDir(QDir::fromNativeSeparators(folder));
28  QString linkName = profile+QLatin1String("/Links/") + folderDir.dirName() + QLatin1String(".lnk");
29  if (!QFile::link(folder, linkName))
30  qDebug() << Q_FUNC_INFO << "linking" << folder << "to" << linkName << "failed!";
31 }
