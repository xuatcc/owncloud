213 bool SyncEngine::checkErrorBlacklisting( SyncFileItem &item )
214 {
215  if( !_journal ) {
216  qWarning() << "Journal is undefined!";
217  return false;
218  }
219 
220  SyncJournalErrorBlacklistRecord entry = _journal->errorBlacklistEntry(item._file);
221  item._hasBlacklistEntry = false;
222 
223  if( !entry.isValid() ) {
224  return false;
225  }
226 
227  item._hasBlacklistEntry = true;
228 
229  // If duration has expired, it's not blacklisted anymore
230  time_t now = Utility::qDateTimeToTime_t(QDateTime::currentDateTime());
231  if( now > entry._lastTryTime + entry._ignoreDuration ) {
232  qDebug() << "blacklist entry for " << item._file << " has expired!";
233  return false;
234  }
235 
236  // If the file has changed locally or on the server, the blacklist
237  // entry no longer applies
238  if( item._direction == SyncFileItem::Up ) { // check the modtime
239  if(item._modtime == 0 || entry._lastTryModtime == 0) {
240  return false;
241  } else if( item._modtime != entry._lastTryModtime ) {
242  qDebug() << item._file << " is blacklisted, but has changed mtime!";
243  return false;
244  } else if( item._renameTarget != entry._renameTarget) {
245  qDebug() << item._file << " is blacklisted, but rename target changed from" << entry._renameTarget;
246  return false;
247  }
248  } else if( item._direction == SyncFileItem::Down ) {
249  // download, check the etag.
250  if( item._etag.isEmpty() || entry._lastTryEtag.isEmpty() ) {
251  qDebug() << item._file << "one ETag is empty, no blacklisting";
252  return false;
253  } else if( item._etag != entry._lastTryEtag ) {
254  qDebug() << item._file << " is blacklisted, but has changed etag!";
255  return false;
256  }
257  }
258 
259  qDebug() << "Item is on blacklist: " << entry._file
260  << "retries:" << entry._retryCount
261  << "for another" << (entry._lastTryTime + entry._ignoreDuration - now) << "s";
262  item._instruction = CSYNC_INSTRUCTION_ERROR;
263  item._status = SyncFileItem::FileIgnored;
264  item._errorString = tr("The item is not synced because of previous errors: %1").arg(entry._errorString);
265 
266  return true;
267 }

340 int SyncEngine::treewalkFile( TREE_WALK_FILE *file, bool remote )
341 {
342  if( ! file ) return -1;
343 
344  QTextCodec::ConverterState utf8State;
345  static QTextCodec *codec = QTextCodec::codecForName("UTF-8");
346  Q_ASSERT(codec);
347  QString fileUtf8 = codec->toUnicode(file->path, qstrlen(file->path), &utf8State);
348  QString renameTarget;
349  QString key = fileUtf8;
350 
351  auto instruction = file->instruction;
352  if (utf8State.invalidChars > 0 || utf8State.remainingChars > 0) {
353  qWarning() << "File ignored because of invalid utf-8 sequence: " << file->path;
354  instruction = CSYNC_INSTRUCTION_IGNORE;
355  } else {
356  renameTarget = codec->toUnicode(file->rename_path, qstrlen(file->rename_path), &utf8State);
357  if (utf8State.invalidChars > 0 || utf8State.remainingChars > 0) {
358  qWarning() << "File ignored because of invalid utf-8 sequence in the rename_path: " << file->path << file->rename_path;
359  instruction = CSYNC_INSTRUCTION_IGNORE;
360  }
361  if (instruction == CSYNC_INSTRUCTION_RENAME) {
362  key = renameTarget;
363  }
364  }
365 
366  // Gets a default-constructed SyncFileItemPtr or the one from the first walk (=local walk)
367  SyncFileItemPtr item = _syncItemMap.value(key);
368  if (!item)
369  item = SyncFileItemPtr(new SyncFileItem);
370 
371  if (item->_file.isEmpty() || instruction == CSYNC_INSTRUCTION_RENAME) {
372  item->_file = fileUtf8;
373  }
374  item->_originalFile = item->_file;
375 
376  if (item->_instruction == CSYNC_INSTRUCTION_NONE
377  || (item->_instruction == CSYNC_INSTRUCTION_IGNORE && instruction != CSYNC_INSTRUCTION_NONE)) {
378  item->_instruction = instruction;
379  item->_modtime = file->modtime;
380  } else {
381  if (instruction != CSYNC_INSTRUCTION_NONE) {
382  qDebug() << "ERROR: Instruction" << item->_instruction << "vs" << instruction << "for" << fileUtf8;
383  Q_ASSERT(!"Instructions are both unequal NONE");
384  return -1;
385  }
386  }
387 
388  if (file->file_id && file->file_id[0]) {
389  item->_fileId = file->file_id;
390  }
391  if (file->directDownloadUrl) {
392  item->_directDownloadUrl = QString::fromUtf8( file->directDownloadUrl );
393  }
394  if (file->directDownloadCookies) {
395  item->_directDownloadCookies = QString::fromUtf8( file->directDownloadCookies );
396  }
397  if (file->remotePerm && file->remotePerm[0]) {
398  item->_remotePerm = QByteArray(file->remotePerm);
399  }
400 
401  /* The flag "serverHasIgnoredFiles" is true if item in question is a directory
402  * that has children which are ignored in sync, either because the files are
403  * matched by an ignore pattern, or because they are hidden.
404  *
405  * Only the information about the server side ignored files is stored to the
406  * database and thus written to the item here. For the local repository its
407  * generated by the walk through the real file tree by discovery phase.
408  *
409  * It needs to go to the sync journal becasue the stat information about remote
410  * files are often read from database rather than being pulled from remote.
411  */
412  if( remote ) {
413  item->_serverHasIgnoredFiles = (file->has_ignored_files > 0);
414  }
415 
416  // Sometimes the discovery computes checksums for local files
417  if (!remote && file->checksum && file->checksumTypeId) {
418  item->_contentChecksum = QByteArray(file->checksum);
419  item->_contentChecksumType = _journal->getChecksumType(file->checksumTypeId);
420  }
421 
422  // record the seen files to be able to clean the journal later
423  _seenFiles.insert(item->_file);
424  if (!renameTarget.isEmpty()) {
425  // Yes, this records both the rename renameTarget and the original so we keep both in case of a rename
426  _seenFiles.insert(renameTarget);
427  }
428 
429  if (remote && file->remotePerm && file->remotePerm[0]) {
430  _remotePerms[item->_file] = file->remotePerm;
431  }
432 
433  switch(file->error_status) {
434  case CSYNC_STATUS_OK:
435  break;
436  case CSYNC_STATUS_INDIVIDUAL_IS_SYMLINK:
437  item->_errorString = tr("Symbolic links are not supported in syncing.");
438  break;
439  case CSYNC_STATUS_INDIVIDUAL_IGNORE_LIST:
440  item->_errorString = tr("File is listed on the ignore list.");
441  break;
442  case CSYNC_STATUS_INDIVIDUAL_IS_INVALID_CHARS:
443  item->_errorString = tr("Filename contains invalid characters that can not be synced cross platform.");
444  break;
445  case CSYNC_STATUS_INDIVIDUAL_TRAILING_SPACE:
446  item->_errorString = tr("Filename contains trailing spaces.");
447  break;
448  case CSYNC_STATUS_INDIVIDUAL_EXCLUDE_LONG_FILENAME:
449  item->_errorString = tr("Filename is too long.");
450  break;
451  case CSYNC_STATUS_INDIVIDUAL_EXCLUDE_HIDDEN:
452  item->_errorString = tr("File/Folder is ignored because it's hidden.");
453  break;
454  case CYSNC_STATUS_FILE_LOCKED_OR_OPEN:
455  item->_errorString = QLatin1String("File locked"); // don't translate, internal use!
456  break;
457  case CSYNC_STATUS_INDIVIDUAL_STAT_FAILED:
458  item->_errorString = tr("Stat failed.");
459  break;
460  case CSYNC_STATUS_SERVICE_UNAVAILABLE:
461  item->_errorString = QLatin1String("Server temporarily unavailable.");
462  break;
463  case CSYNC_STATUS_STORAGE_UNAVAILABLE:
464  item->_errorString = QLatin1String("Directory temporarily not available on server.");
465  item->_status = SyncFileItem::SoftError;
466  _temporarilyUnavailablePaths.insert(item->_file);
467  break;
468  case CSYNC_STATUS_FORBIDDEN:
469  item->_errorString = QLatin1String("Access forbidden.");
470  item->_status = SyncFileItem::SoftError;
471  _temporarilyUnavailablePaths.insert(item->_file);
472  break;
473  case CSYNC_STATUS_PERMISSION_DENIED:
474  item->_errorString = QLatin1String("Directory not accessible on client, permission denied.");
475  item->_status = SyncFileItem::SoftError;
476  break;
477  default:
478  Q_ASSERT("Non handled error-status");
479  /* No error string */
480  }
481 
482  if (item->_instruction == CSYNC_INSTRUCTION_IGNORE && (utf8State.invalidChars > 0 || utf8State.remainingChars > 0)) {
483  item->_status = SyncFileItem::NormalError;
484  //item->_instruction = CSYNC_INSTRUCTION_ERROR;
485  item->_errorString = tr("Filename encoding is not valid");
486  }
487 
488  bool isDirectory = file->type == CSYNC_FTW_TYPE_DIR;
489 
490  if (file->etag && file->etag[0]) {
491  item->_etag = file->etag;
492  }
493  item->_size = file->size;
494 
495  if (!item->_inode) {
496  item->_inode = file->inode;
497  }
498 
499  switch( file->type ) {
500  case CSYNC_FTW_TYPE_DIR:
501  item->_type = SyncFileItem::Directory;
502  break;
503  case CSYNC_FTW_TYPE_FILE:
504  item->_type = SyncFileItem::File;
505  break;
506  case CSYNC_FTW_TYPE_SLINK:
507  item->_type = SyncFileItem::SoftLink;
508  break;
509  default:
510  item->_type = SyncFileItem::UnknownType;
511  }
512 
513  SyncFileItem::Direction dir = SyncFileItem::None;
514 
515  int re = 0;
516  switch(file->instruction) {
517  case CSYNC_INSTRUCTION_NONE: {
518  // Any files that are instruction NONE?
519  if (!isDirectory && file->other.instruction == CSYNC_INSTRUCTION_NONE) {
520  _hasNoneFiles = true;
521  }
522  // No syncing or update to be done.
523  return re;
524  }
525  case CSYNC_INSTRUCTION_UPDATE_METADATA:
526  dir = SyncFileItem::None;
527  // For directories, metadata-only updates will be done after all their files are propagated.
528  if (!isDirectory) {
529  item->_isDirectory = isDirectory;
530  emit syncItemDiscovered(*item);
531 
532  // Update the database now already: New remote fileid or Etag or RemotePerm
533  // Or for files that were detected as "resolved conflict".
534  // Or a local inode/mtime change
535 
536  // In case of "resolved conflict": there should have been a conflict because they
537  // both were new, or both had their local mtime or remote etag modified, but the
538  // size and mtime is the same on the server. This typically happens when the
539  // database is removed. Nothing will be done for those files, but we still need
540  // to update the database.
541 
542  // This metadata update *could* be a propagation job of its own, but since it's
543  // quick to do and we don't want to create a potentially large number of
544  // mini-jobs later on, we just update metadata right now.
545 
546  if (remote) {
547  QString filePath = _localPath + item->_file;
548 
549  // Even if the mtime is different on the server, we always want to keep the mtime from
550  // the file system in the DB, this is to avoid spurious upload on the next sync
551  item->_modtime = file->other.modtime;
552  // same for the size
553  item->_size = file->other.size;
554 
555  // If the 'W' remote permission changed, update the local filesystem
556  SyncJournalFileRecord prev = _journal->getFileRecord(item->_file);
557  if (prev.isValid() && prev._remotePerm.contains('W') != item->_remotePerm.contains('W')) {
558  const bool isReadOnly = !item->_remotePerm.contains('W');
559  FileSystem::setFileReadOnlyWeak(filePath, isReadOnly);
560  }
561 
562  _journal->setFileRecordMetadata(SyncJournalFileRecord(*item, filePath));
563  } else {
564  // The local tree is walked first and doesn't have all the info from the server.
565  // Update only outdated data from the disk.
566  _journal->updateLocalMetadata(item->_file, item->_modtime, item->_size, item->_inode);
567  }
568 
569  // Technically we're done with this item.
570  return re;
571  }
572  break;
573  case CSYNC_INSTRUCTION_RENAME:
574  dir = !remote ? SyncFileItem::Down : SyncFileItem::Up;
575  item->_renameTarget = renameTarget;
576  if (isDirectory)
577  _renamedFolders.insert(item->_file, item->_renameTarget);
578  break;
579  case CSYNC_INSTRUCTION_REMOVE:
580  _hasRemoveFile = true;
581  dir = !remote ? SyncFileItem::Down : SyncFileItem::Up;
582  break;
583  case CSYNC_INSTRUCTION_CONFLICT:
584  case CSYNC_INSTRUCTION_IGNORE:
585  case CSYNC_INSTRUCTION_ERROR:
586  dir = SyncFileItem::None;
587  break;
588  case CSYNC_INSTRUCTION_TYPE_CHANGE:
589  case CSYNC_INSTRUCTION_SYNC:
590  if (!remote) {
591  // An upload of an existing file means that the file was left unchanged on the server
592  // This counts as a NONE for detecting if all the files on the server were changed
593  _hasNoneFiles = true;
594  } else if (!isDirectory) {
595  auto difftime = std::difftime(file->modtime, file->other.modtime);
596  if (difftime < -3600*2) {
597  // We are going back on time
598  // We only increment if the difference is more than two hours to avoid clock skew
599  // issues or DST changes. (We simply ignore files that goes in the past less than
600  // two hours for the backup detection heuristics.)
601  _backInTimeFiles++;
602  qDebug() << file->path << "has a timestamp earlier than the local file";
603  } else if (difftime > 0) {
604  _hasForwardInTimeFiles = true;
605  }
606  }
607  dir = remote ? SyncFileItem::Down : SyncFileItem::Up;
608  break;
609  case CSYNC_INSTRUCTION_NEW:
610  case CSYNC_INSTRUCTION_EVAL:
611  case CSYNC_INSTRUCTION_STAT_ERROR:
612  default:
613  dir = remote ? SyncFileItem::Down : SyncFileItem::Up;
614  break;
615  }
616 
617  item->_direction = dir;
618  item->_isDirectory = isDirectory;
619  if (instruction != CSYNC_INSTRUCTION_NONE) {
620  // check for blacklisting of this item.
621  // if the item is on blacklist, the instruction was set to ERROR
622  checkErrorBlacklisting( *item );
623  }
624 
625  _progressInfo->adjustTotalsForFile(*item);
626 
627  _needsUpdate = true;
628 
629  item->log._etag = file->etag;
630  item->log._fileId = file->file_id;
631  item->log._instruction = file->instruction;
632  item->log._modtime = file->modtime;
633  item->log._size = file->size;
634

1120 void SyncEngine::checkForPermission()
1121 {
1122  bool selectiveListOk;
1123  auto selectiveSyncBlackList = _journal->getSelectiveSyncList(SyncJournalDb::SelectiveSyncBlackList, &selectiveListOk);
1124  std::sort(selectiveSyncBlackList.begin(), selectiveSyncBlackList.end());
1125 
1126  for (SyncFileItemVector::iterator it = _syncedItems.begin(); it != _syncedItems.end(); ++it) {
1127  if ((*it)->_direction != SyncFileItem::Up) {
1128  // Currently we only check server-side permissions
1129  continue;
1130  }
1131 
1132  // Do not propagate anything in the server if it is in the selective sync blacklist
1133  const QString path = (*it)->destination() + QLatin1Char('/');
1134 
1135  // if reading the selective sync list from db failed, lets ignore all rather than nothing.
1136  if (!selectiveListOk || std::binary_search(selectiveSyncBlackList.constBegin(), selectiveSyncBlackList.constEnd(),
1137  path)) {
1138  (*it)->_instruction = CSYNC_INSTRUCTION_IGNORE;
1139  (*it)->_status = SyncFileItem::FileIgnored;
1140  (*it)->_errorString = tr("Ignored because of the \"choose what to sync\" blacklist");
1141 
1142  if ((*it)->_isDirectory) {
1143  for (SyncFileItemVector::iterator it_next = it + 1; it_next != _syncedItems.end() && (*it_next)->_file.startsWith(path); ++it_next) {
1144  it = it_next;
1145  (*it)->_instruction = CSYNC_INSTRUCTION_IGNORE;
1146  (*it)->_status = SyncFileItem::FileIgnored;
1147  (*it)->_errorString = tr("Ignored because of the \"choose what to sync\" blacklist");
1148  }
1149  }
1150  continue;
1151  }
1152 
1153  switch((*it)->_instruction) {
1154  case CSYNC_INSTRUCTION_TYPE_CHANGE:
1155  case CSYNC_INSTRUCTION_NEW: {
1156  int slashPos = (*it)->_file.lastIndexOf('/');
1157  QString parentDir = slashPos <= 0 ? "" : (*it)->_file.mid(0, slashPos);
1158  const QByteArray perms = getPermissions(parentDir);
1159  if (perms.isNull()) {
1160  // No permissions set
1161  break;
1162  } else if ((*it)->_isDirectory && !perms.contains("K")) {
1163  qDebug() << "checkForPermission: ERROR" << (*it)->_file;
1164  (*it)->_instruction = CSYNC_INSTRUCTION_ERROR;
1165  (*it)->_status = SyncFileItem::NormalError;
1166  (*it)->_errorString = tr("Not allowed because you don't have permission to add subfolders to that folder");
1167 
1168  for (SyncFileItemVector::iterator it_next = it + 1; it_next != _syncedItems.end() && (*it_next)->destination().startsWith(path); ++it_next) {
1169  it = it_next;
1170  if ((*it)->_instruction == CSYNC_INSTRUCTION_RENAME) {
1171  // The file was most likely moved in this directory.
1172  // If the file was read only or could not be moved or removed, it should
1173  // be restored. Do that in the next sync by not considering as a rename
1174  // but delete and upload. It will then be restored if needed.
1175  _journal->avoidRenamesOnNextSync((*it)->_file);
1176  _anotherSyncNeeded = true;
1177  qDebug() << "Moving of " << (*it)->_file << " canceled because no permission to add parent folder";
1178  }
1179  (*it)->_instruction = CSYNC_INSTRUCTION_ERROR;
1180  (*it)->_status = SyncFileItem::SoftError;
1181  (*it)->_errorString = tr("Not allowed because you don't have permission to add parent folder");
1182  }
1183 
1184  } else if (!(*it)->_isDirectory && !perms.contains("C")) {
1185  qDebug() << "checkForPermission: ERROR" << (*it)->_file;
1186  (*it)->_instruction = CSYNC_INSTRUCTION_ERROR;
1187  (*it)->_status = SyncFileItem::NormalError;
1188  (*it)->_errorString = tr("Not allowed because you don't have permission to add files in that folder");
1189  }
1190  break;
1191  }
1192  case CSYNC_INSTRUCTION_SYNC: {
1193  const QByteArray perms = getPermissions((*it)->_file);
1194  if (perms.isNull()) {
1195  // No permissions set
1196  break;
1197  } if (!perms.contains("W")) {
1198  qDebug() << "checkForPermission: RESTORING" << (*it)->_file;
1199  (*it)->_instruction = CSYNC_INSTRUCTION_CONFLICT;
1200  (*it)->_direction = SyncFileItem::Down;
1201  (*it)->_isRestoration = true;
1202  // take the things to write to the db from the "other" node (i.e: info from server)
1203  (*it)->_modtime = (*it)->log._other_modtime;
1204  (*it)->_size = (*it)->log._other_size;
1205  (*it)->_fileId = (*it)->log._other_fileId;
1206  (*it)->_etag = (*it)->log._other_etag;
1207  (*it)->_errorString = tr("Not allowed to upload this file because it is read-only on the server, restoring");
1208  continue;
1209  }
1210  break;
1211  }
1212  case CSYNC_INSTRUCTION_REMOVE: {
1213  const QByteArray perms = getPermissions((*it)->_file);
1214  if (perms.isNull()) {
1215  // No permissions set
1216  break;
1217  }
