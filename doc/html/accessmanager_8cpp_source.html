57 QNetworkReply* AccessManager::createRequest(QNetworkAccessManager::Operation op, const QNetworkRequest& request, QIODevice* outgoingData)
58 {
59  QNetworkRequest newRequest(request);
60 
61  if (newRequest.hasRawHeader("cookie")) {
62  // This will set the cookie into the QNetworkCookieJar which will then override the cookie header
63  setRawCookie(request.rawHeader("cookie"), request.url());
64  }
65 
66  newRequest.setRawHeader(QByteArray("User-Agent"), Utility::userAgentString());
67 
68  // Some firewalls reject requests that have a "User-Agent" but no "Accept" header
69  newRequest.setRawHeader(QByteArray("Accept"), "*/*");
70 
71  QByteArray verb = newRequest.attribute(QNetworkRequest::CustomVerbAttribute).toByteArray();
72  // For PROPFIND (assumed to be a WebDAV op), set xml/utf8 as content type/encoding
73  // This needs extension
74  if (verb == "PROPFIND") {
75  newRequest.setHeader( QNetworkRequest::ContentTypeHeader, QLatin1String("text/xml; charset=utf-8"));
76  }
77  return QNetworkAccessManager::createRequest(op, newRequest, outgoingData);
78 }
