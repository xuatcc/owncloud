67 bool Utility::writeRandomFile( const QString& fname, int size )
68 {
69  int maxSize = 10*10*1024;
70  qsrand(QDateTime::currentMSecsSinceEpoch());
71 
72  if( size == -1 ) size = qrand() % maxSize;
73 
74  QString randString;
75  for( int i = 0; i < size; i++ ) {
76  int r = qrand() % 128;
77  randString.append(QChar(r));
78  }
79 
80  QFile file(fname);
81  if( file.open(QIODevice::WriteOnly | QIODevice::Text) ) {
82  QTextStream out(&file);
83  out << randString;
84  // optional, as QFile destructor will already do it:
85  file.close();
86  return true;
87  }
88  return false;
89

92 QString Utility::formatFingerprint( const QByteArray& fmhash, bool colonSeparated )
93 {
94  QByteArray hash;
95  int steps = fmhash.length()/2;
96  for (int i = 0; i < steps; i++) {
97  hash.append(fmhash[i*2]);
98  hash.append(fmhash[i*2+1]);
99  hash.append(' ');
100  }
101 
102  QString fp = QString::fromLatin1( hash.trimmed() );
103  if (colonSeparated) {
104  fp.replace(QChar(' '), QChar(':'));
105  }
106 
107  return fp;
108 }

115 QString Utility::octetsToString( qint64 octets )
116 {
117 #define THE_FACTOR 1024
118  static const qint64 kb = THE_FACTOR;
119  static const qint64 mb = THE_FACTOR * kb;
120  static const qint64 gb = THE_FACTOR * mb;
121 
122  QString s;
123  qreal value = octets;
124 
125  // Whether we care about decimals: only for GB and only
126  // if it's less than 10 GB.
127  bool round = true;
128 
129  // do not display terra byte with the current units, as when
130  // the MB, GB and KB units were made, there was no TB,
131  // see the JEDEC standard
132  // https://en.wikipedia.org/wiki/JEDEC_memory_standards
133  if (octets >= gb) {
134  s = QCoreApplication::translate("Utility", "%L1 GB");
135  value /= gb;
136  round = false;
137  } else if (octets >= mb) {
138  s = QCoreApplication::translate("Utility", "%L1 MB");
139  value /= mb;
140  } else if (octets >= kb) {
141  s = QCoreApplication::translate("Utility", "%L1 KB");
142  value /= kb;
143  } else {
144  s = QCoreApplication::translate("Utility", "%L1 B");
145  }
146 
147  if (value > 9.95)
148  round = true;
149 
150  if (round)
151  return s.arg(qRound(value));
152 
153  return s.arg(value, 0, 'g', 2);
154 }

207 qint64 Utility::freeDiskSpace(const QString &path)
208 {
209 #if defined(Q_OS_MAC) || defined(Q_OS_FREEBSD) || defined(Q_OS_FREEBSD_KERNEL) || defined(Q_OS_NETBSD) || defined(Q_OS_OPENBSD)
210  struct statvfs stat;
211  if (statvfs(path.toLocal8Bit().data(), &stat) == 0) {
212  return (qint64) stat.f_bavail * stat.f_frsize;
213  }
214 #elif defined(Q_OS_UNIX)
215  struct statvfs64 stat;
216  if (statvfs64(path.toLocal8Bit().data(), &stat) == 0) {
217  return (qint64) stat.f_bavail * stat.f_frsize;
218  }
219 #elif defined(Q_OS_WIN)
220  ULARGE_INTEGER freeBytes;
221  freeBytes.QuadPart = 0L;
222  if (GetDiskFreeSpaceEx( reinterpret_cast<const wchar_t *>(path.utf16()), &freeBytes, NULL, NULL )) {
223  return freeBytes.QuadPart;
224  }
225 #endif
226  return -1;
227 }

229 QString Utility::compactFormatDouble(double value, int prec, const QString& unit)
230 {
231  QLocale locale = QLocale::system();
232  QChar decPoint = locale.decimalPoint();
233  QString str = locale.toString(value, 'f', prec);
234  while (str.endsWith('0') || str.endsWith(decPoint)) {
235  if (str.endsWith(decPoint)) {
236  str.chop(1);
237  break;
238  }
239  str.chop(1);
240  }
241  if( !unit.isEmpty() )
242  str += (QLatin1Char(' ')+unit);
243  return str;
244 }

339 QString Utility::durationToDescriptiveString2(quint64 msecs)
340 {
341  int p = 0;
342  while (periods[p+1].name && msecs < periods[p].msec) {
343  p++;
344  }
345 
346  auto firstPart = periods[p].description(int(msecs / periods[p].msec));
347 
348  if (!periods[p+1].name) {
349  return firstPart;
350  }
351 
352  quint64 secondPartNum = qRound( double(msecs % periods[p].msec) / periods[p+1].msec);
353 
354  if (secondPartNum == 0) {
355  return firstPart;
356  }
357 
358  return QCoreApplication::translate("Utility", "%1 %2").arg(firstPart,
359  periods[p+1].description(secondPartNum));
360 }

459 // read the output of the owncloud --version command from the owncloud
460 // version that is on disk. This works for most versions of the client,
461 // because clients that do not yet know the --version flag return the
462 // version in the first line of the help output :-)
463 //
464 // This version only delivers output on linux, as Mac and Win get their
465 // restarting from the installer.
466 QByteArray Utility::versionOfInstalledBinary( const QString& command )
467 {
468  QByteArray re;
469  if( isLinux() ) {
470  QString binary(command);
471  if( binary.isEmpty() ) {
472  binary = qApp->arguments()[0];
473  }
474  QStringList params;
475  params << QLatin1String("--version");
476  QProcess process;
477  process.start(binary, params);
478  process.waitForFinished(); // sets current thread to sleep and waits for pingProcess end
479  re = process.readAllStandardOutput();
480  int newline = re.indexOf(QChar('\n'));
481  if( newline > 0 ) {
482  re.truncate( newline );
483  }
484  }
485  return re;
486 }

488 QString Utility::timeAgoInWords(const QDateTime& dt, const QDateTime& from)
489 {
490  QDateTime now = QDateTime::currentDateTimeUtc();
491 
492  if( from.isValid() ) {
493  now = from;
494  }
495 
496  if( dt.daysTo(now)>0 ) {
497  int dtn = dt.daysTo(now);
498  return QObject::tr("%n day(s) ago", "", dtn);
499  } else {
500  qint64 secs = dt.secsTo(now);
501  if( secs < 0 ) {
502  return QObject::tr("in the future");
503  }
504  if( floor(secs / 3600.0) > 0 ) {
505  int hours = floor(secs/3600.0);
506  return( QObject::tr("%n hour(s) ago", "", hours) );
507  } else {
508  int minutes = qRound(secs/60.0);
509  if( minutes == 0 ) {
510  if(secs < 5) {
511  return QObject::tr("now");
512  } else {
513  return QObject::tr("Less than a minute ago");
514  }
515  }
516  return( QObject::tr("%n minute(s) ago", "", minutes) );
517  }
518  }
519  return QObject::tr("Some time ago");
520 }

592 QUrl Utility::concatUrlPath(const QUrl &url, const QString &concatPath,
593  const QList< QPair<QString, QString> > &queryItems)
594 {
595  QString path = url.path();
596  if (! concatPath.isEmpty()) {
597  // avoid '//'
598  if (path.endsWith('/') && concatPath.startsWith('/')) {
599  path.chop(1);
600  } // avoid missing '/'
601  else if (!path.endsWith('/') && !concatPath.startsWith('/')) {
602  path += QLatin1Char('/');
603  }
604  path += concatPath; // put the complete path together
605  }
606 
607  QUrl tmpUrl = url;
608  tmpUrl.setPath(path);
609  if( queryItems.size() > 0 ) {
610  tmpUrl.setQueryItems(queryItems);
611  }
612  return tmpUrl;
613 }
