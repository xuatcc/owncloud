70 bool ExcludedFiles::isExcluded(
71  const QString& filePath,
72  const QString& basePath,
73  bool excludeHidden) const
74 {
75  if (!filePath.startsWith(basePath, Utility::fsCasePreserving() ? Qt::CaseInsensitive : Qt::CaseSensitive)) {
76  // Mark paths we're not responsible for as excluded...
77  return true;
78  }
79 
80  if( excludeHidden ) {
81  QString path = filePath;
82  // Check all path subcomponents, but to *not* check the base path:
83  // We do want to be able to sync with a hidden folder as the target.
84  while (path.size() > basePath.size()) {
85  QFileInfo fi(path);
86  if( fi.isHidden() || fi.fileName().startsWith(QLatin1Char('.')) ) {
87  return true;
88  }
89 
90  // Get the parent path
91  path = fi.absolutePath();
92  }
93  }
94 
95  QFileInfo fi(filePath);
96  csync_ftw_type_e type = CSYNC_FTW_TYPE_FILE;
97  if (fi.isDir()) {
98  type = CSYNC_FTW_TYPE_DIR;
99  }
100 
101  QString relativePath = filePath.mid(basePath.size());
102  if (relativePath.endsWith(QLatin1Char('/'))) {
103  relativePath.chop(1);
104  }
105 
106  return csync_excluded_no_ctx(*_excludesPtr, relativePath.toUtf8(), type) != CSYNC_NOT_EXCLUDED;
107 }
