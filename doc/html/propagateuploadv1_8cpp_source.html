37 void PropagateUploadFileV1::doStartUpload()
38 {
39  _chunkCount = std::ceil(_item->_size / double(chunkSize()));
40  _startChunk = 0;
41  _transferId = qrand() ^ _item->_modtime ^ (_item->_size << 16);
42 
43  const SyncJournalDb::UploadInfo progressInfo = _propagator->_journal->getUploadInfo(_item->_file);
44 
45  if (progressInfo._valid && Utility::qDateTimeToTime_t(progressInfo._modtime) == _item->_modtime ) {
46  _startChunk = progressInfo._chunk;
47  _transferId = progressInfo._transferid;
48  qDebug() << Q_FUNC_INFO << _item->_file << ": Resuming from chunk " << _startChunk;
49  }
50 
51  _currentChunk = 0;
52  _duration.start();
53 
54  emit progress(*_item, 0);
55  startNextChunk();
56 }

58 void PropagateUploadFileV1::startNextChunk()
59 {
60  if (_propagator->_abortRequested.fetchAndAddRelaxed(0))
61  return;
62 
63  if (! _jobs.isEmpty() && _currentChunk + _startChunk >= _chunkCount - 1) {
64  // Don't do parallel upload of chunk if this might be the last chunk because the server cannot handle that
65  // https://github.com/owncloud/core/issues/11106
66  // We return now and when the _jobs are finished we will proceed with the last chunk
67  // NOTE: Some other parts of the code such as slotUploadProgress also assume that the last chunk
68  // is sent last.
69  return;
70  }
71  quint64 fileSize = _item->_size;
72  auto headers = PropagateUploadFileCommon::headers();
73  headers["OC-Total-Length"] = QByteArray::number(fileSize);
74  headers["OC-Chunk-Size"]= QByteArray::number(quint64(chunkSize()));
75 
76  QString path = _item->_file;
77 
78  UploadDevice *device = new UploadDevice(&_propagator->_bandwidthManager);
79  qint64 chunkStart = 0;
80  qint64 currentChunkSize = fileSize;
81  bool isFinalChunk = false;
82  if (_chunkCount > 1) {
83  int sendingChunk = (_currentChunk + _startChunk) % _chunkCount;
84  // XOR with chunk size to make sure everything goes well if chunk size changes between runs
85  uint transid = _transferId ^ chunkSize();
86  qDebug() << "Upload chunk" << sendingChunk << "of" << _chunkCount << "transferid(remote)=" << transid;
87  path += QString("-chunking-%1-%2-%3").arg(transid).arg(_chunkCount).arg(sendingChunk);
88 
89  headers["OC-Chunked"] = "1";
90 
91  chunkStart = chunkSize() * quint64(sendingChunk);
92  currentChunkSize = chunkSize();
93  if (sendingChunk == _chunkCount - 1) { // last chunk
94  currentChunkSize = (fileSize % chunkSize());
95  if( currentChunkSize == 0 ) { // if the last chunk pretends to be 0, its actually the full chunk size.
96  currentChunkSize = chunkSize();
97  }
98  isFinalChunk = true;
99  }
100  } else {
101  // if there's only one chunk, it's the final one
102  isFinalChunk = true;
103  }
104 
105  if (isFinalChunk && !_transmissionChecksumType.isEmpty()) {
106  headers[checkSumHeaderC] = makeChecksumHeader(
107  _transmissionChecksumType, _transmissionChecksum);
108  }
109 
110  const QString fileName = _propagator->getFilePath(_item->_file);
111  if (! device->prepareAndOpen(fileName, chunkStart, currentChunkSize)) {
112  qDebug() << "ERR: Could not prepare upload device: " << device->errorString();
113 
114  // If the file is currently locked, we want to retry the sync
115  // when it becomes available again.
116  if (FileSystem::isFileLocked(fileName)) {
117  emit _propagator->seenLockedFile(fileName);
118  }
119  // Soft error because this is likely caused by the user modifying his files while syncing
120  abortWithError( SyncFileItem::SoftError, device->errorString() );
121  delete device;
122  return;
123  }
124 
125  // job takes ownership of device via a QScopedPointer. Job deletes itself when finishing
126  PUTFileJob* job = new PUTFileJob(_propagator->account(), _propagator->_remoteFolder + path, device, headers, _currentChunk, this);
127  _jobs.append(job);
128  connect(job, SIGNAL(finishedSignal()), this, SLOT(slotPutFinished()));
129  connect(job, SIGNAL(uploadProgress(qint64,qint64)), this, SLOT(slotUploadProgress(qint64,qint64)));
130  connect(job, SIGNAL(uploadProgress(qint64,qint64)), device, SLOT(slotJobUploadProgress(qint64,qint64)));
131  connect(job, SIGNAL(destroyed(QObject*)), this, SLOT(slotJobDestroyed(QObject*)));
132  job->start();
133  _propagator->_activeJobList.append(this);
134  _currentChunk++;
135 
136  bool parallelChunkUpload = true;
137 
138  if (_propagator->account()->capabilities().chunkingParallelUploadDisabled()) {
139  // Server may also disable parallel chunked upload for any higher version
140  parallelChunkUpload = false;
141  } else {
142  QByteArray env = qgetenv("OWNCLOUD_PARALLEL_CHUNK");
143  if (!env.isEmpty()) {
144  parallelChunkUpload = env != "false" && env != "0";
145  } else {
146  int versionNum = _propagator->account()->serverVersionInt();
147  if (versionNum < 0x080003) {
148  // Disable parallel chunk upload severs older than 8.0.3 to avoid too many
149  // internal sever errors (#2743, #2938)
150  parallelChunkUpload = false;
151  }
152  }
153  }
154 
155 
156  if (_currentChunk + _startChunk >= _chunkCount - 1) {
157  // Don't do parallel upload of chunk if this might be the last chunk because the server cannot handle that
158  // https://github.com/owncloud/core/issues/11106
159  parallelChunkUpload = false;
160  }
161 
162  if (parallelChunkUpload && (_propagator->_activeJobList.count() < _propagator->maximumActiveJob())
163  && _currentChunk < _chunkCount ) {
164  startNextChunk();
165  }
166  if (!parallelChunkUpload || _chunkCount - _currentChunk <= 0) {
167  emit ready();
168  }
169 }

171 void PropagateUploadFileV1::slotPutFinished()
172 {
173  PUTFileJob *job = qobject_cast<PUTFileJob *>(sender());
174  Q_ASSERT(job);
175  slotJobDestroyed(job); // remove it from the _jobs list
176 
177  qDebug() << Q_FUNC_INFO << job->reply()->request().url() << "FINISHED WITH STATUS"
178  << job->reply()->error()
179  << (job->reply()->error() == QNetworkReply::NoError ? QLatin1String("") : job->reply()->errorString())
180  << job->reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute)
181  << job->reply()->attribute(QNetworkRequest::HttpReasonPhraseAttribute);
182 
183  _propagator->_activeJobList.removeOne(this);
184 
185  if (_finished) {
186  // We have sent the finished signal already. We don't need to handle any remaining jobs
187  return;
188  }
189 
190  QNetworkReply::NetworkError err = job->reply()->error();
191 
192 #if QT_VERSION < QT_VERSION_CHECK(5, 4, 2)
193  if (err == QNetworkReply::OperationCanceledError && job->reply()->property("owncloud-should-soft-cancel").isValid()) { // Abort the job and try again later.
194  // This works around a bug in QNAM wich might reuse a non-empty buffer for the next request.
195  qDebug() << "Forcing job abort on HTTP connection reset with Qt < 5.4.2.";
196  _propagator->_anotherSyncNeeded = true;
197  abortWithError(SyncFileItem::SoftError, tr("Forcing job abort on HTTP connection reset with Qt < 5.4.2."));
198  return;
199  }
200 #endif
201 
202  if (err != QNetworkReply::NoError) {
203  _item->_httpErrorCode = job->reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
204  if(checkForProblemsWithShared(_item->_httpErrorCode,
205  tr("The file was edited locally but is part of a read only share. "
206  "It is restored and your edit is in the conflict file."))) {
207  return;
208  }
209  QByteArray replyContent = job->reply()->readAll();
210  qDebug() << replyContent; // display the XML error in the debug
211  QString errorString = errorMessage(job->errorString(), replyContent);
212 
213  if (job->reply()->hasRawHeader("OC-ErrorString")) {
214  errorString = job->reply()->rawHeader("OC-ErrorString");
215  }
216 
217  if (_item->_httpErrorCode == 412) {
218  // Precondition Failed: Maybe the bad etag is in the database, we need to clear the
219  // parent folder etag so we won't read from DB next sync.
220  _propagator->_journal->avoidReadFromDbOnNextSync(_item->_file);
221  _propagator->_anotherSyncNeeded = true;
222  }
223 
224  SyncFileItem::Status status = classifyError(err, _item->_httpErrorCode,
225  &_propagator->_anotherSyncNeeded);
226  abortWithError(status, errorString);
227  return;
228  }
229 
230  _item->_httpErrorCode = job->reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
231  // The server needs some time to process the request and provide us with a poll URL
232  if (_item->_httpErrorCode == 202) {
233  _finished = true;
234  QString path = QString::fromUtf8(job->reply()->rawHeader("OC-Finish-Poll"));
235  if (path.isEmpty()) {
236  done(SyncFileItem::NormalError, tr("Poll URL missing"));
237  return;
238  }
239  startPollJob(path);
240  return;
241  }
242 
243  // Check the file again post upload.
244  // Two cases must be considered separately: If the upload is finished,
245  // the file is on the server and has a changed ETag. In that case,
246  // the etag has to be properly updated in the client journal, and because
247  // of that we can bail out here with an error. But we can reschedule a
248  // sync ASAP.
249  // But if the upload is ongoing, because not all chunks were uploaded
250  // yet, the upload can be stopped and an error can be displayed, because
251  // the server hasn't registered the new file yet.
252  QByteArray etag = getEtagFromReply(job->reply());
253  bool finished = etag.length() > 0;
254 
255  // Check if the file still exists
256  const QString fullFilePath(_propagator->getFilePath(_item->_file));
257  if( !FileSystem::fileExists(fullFilePath) ) {
258  if (!finished) {
259  abortWithError(SyncFileItem::SoftError, tr("The local file was removed during sync."));
260  return;
261  } else {
262  _propagator->_anotherSyncNeeded = true;
263  }
264  }
265 
266  // Check whether the file changed since discovery.
267  if (! FileSystem::verifyFileUnchanged(fullFilePath, _item->_size, _item->_modtime)) {
268  _propagator->_anotherSyncNeeded = true;
269  if( !finished ) {
270  abortWithError(SyncFileItem::SoftError, tr("Local file changed during sync."));
271  // FIXME: the legacy code was retrying for a few seconds.
272  // and also checking that after the last chunk, and removed the file in case of INSTRUCTION_NEW
273  return;
274  }
275  }
276 
277  if (!finished) {
278  // Proceed to next chunk.
279  if (_currentChunk >= _chunkCount) {
280  if (!_jobs.empty()) {
281  // just wait for the other job to finish.
282  return;
283  }
284  _finished = true;
285  done(SyncFileItem::NormalError, tr("The server did not acknowledge the last chunk. (No e-tag was present)"));
286  return;
287  }
288 
289  // Deletes an existing blacklist entry on successful chunk upload
290  if (_item->_hasBlacklistEntry) {
291  _propagator->_journal->wipeErrorBlacklistEntry(_item->_file);
292  _item->_hasBlacklistEntry = false;
293  }
294 
295  SyncJournalDb::UploadInfo pi;
296  pi._valid = true;
297  auto currentChunk = job->_chunk;
298  foreach (auto *job, _jobs) {
299  // Take the minimum finished one
300  if (auto putJob = qobject_cast<PUTFileJob*>(job)) {
301  currentChunk = qMin(currentChunk, putJob->_chunk - 1);
302  }
303  }
304  pi._chunk = (currentChunk + _startChunk + 1) % _chunkCount ; // next chunk to start with
305  pi._transferid = _transferId;
306  pi._modtime = Utility::qDateTimeFromTime_t(_item->_modtime);
307  _propagator->_journal->setUploadInfo(_item->_file, pi);
308  _propagator->_journal->commit("Upload info");
309  startNextChunk();
310  return;
311  }
312 
313  // the following code only happens after all chunks were uploaded.
314  _finished = true;
315  // the file id should only be empty for new files up- or downloaded
316  QByteArray fid = job->reply()->rawHeader("OC-FileID");
317  if( !fid.isEmpty() ) {
318  if( !_item->_fileId.isEmpty() && _item->_fileId != fid ) {
319  qDebug() << "WARN: File ID changed!" << _item->_fileId << fid;
320  }
321  _item->_fileId = fid;
322  }
323 
324  _item->_etag = etag;
325 
326  _item->_responseTimeStamp = job->responseTimestamp();
327 
328  if (job->reply()->rawHeader("X-OC-MTime") != "accepted") {
329  // X-OC-MTime is supported since owncloud 5.0. But not when chunking.
330  // Normally Owncloud 6 always puts X-OC-MTime
331  qWarning() << "Server does not support X-OC-MTime" << job->reply()->rawHeader("X-OC-MTime");
332  // Well, the mtime was not set
333  done(SyncFileItem::SoftError, "Server does not support X-OC-MTime");
334  }
335 
336  // performance logging
337  _item->_requestDuration = _stopWatch.stop();
338  qDebug() << "*==* duration UPLOAD" << _item->_size
339  << _stopWatch.durationOfLap(QLatin1String("ContentChecksum"))
340  << _stopWatch.durationOfLap(QLatin1String("TransmissionChecksum"))
341  << _item->_requestDuration;
342  // The job might stay alive for the whole sync, release this tiny bit of memory.
343  _stopWatch.reset();
344 
345  finalize();
346 }

349 void PropagateUploadFileV1::slotUploadProgress(qint64 sent, qint64 total)
350 {
351  // Completion is signaled with sent=0, total=0; avoid accidentally
352  // resetting progress due to the sent being zero by ignoring it.
353  // finishedSignal() is bound to be emitted soon anyway.
354  // See https://bugreports.qt.io/browse/QTBUG-44782.
355  if (sent == 0 && total == 0) {
356  return;
357  }
358 
359  int progressChunk = _currentChunk + _startChunk - 1;
360  if (progressChunk >= _chunkCount)
361  progressChunk = _currentChunk - 1;
362 
363  // amount is the number of bytes already sent by all the other chunks that were sent
364  // not including this one.
365  // FIXME: this assumes all chunks have the same size, which is true only if the last chunk
366  // has not been finished (which should not happen because the last chunk is sent sequentially)
367  quint64 amount = progressChunk * chunkSize();
368 
369  sender()->setProperty("byteWritten", sent);
370  if (_jobs.count() > 1) {
371  amount -= (_jobs.count() -1) * chunkSize();
372  foreach (QObject *j, _jobs) {
373  amount += j->property("byteWritten").toULongLong();
374  }
375  } else {
376  // sender() is the only current job, no need to look at the byteWritten properties
377  amount += sent;
378  }
379  emit progress(*_item, amount);
380 }
