51 bool PropagateLocalRemove::removeRecursively(const QString& path)
52 {
53  bool success = true;
54  QString absolute = _propagator->_localDir + _item->_file + path;
55  QDirIterator di(absolute, QDir::AllEntries | QDir::Hidden | QDir::System | QDir::NoDotAndDotDot);
56 
57  QVector<QPair<QString, bool>> deleted;
58 
59  while (di.hasNext()) {
60  di.next();
61  const QFileInfo& fi = di.fileInfo();
62  bool ok;
63  // The use of isSymLink here is okay:
64  // we never want to go into this branch for .lnk files
65  bool isDir = fi.isDir() && !fi.isSymLink();
66  if (isDir) {
67  ok = removeRecursively(path + QLatin1Char('/') + di.fileName()); // recursive
68  } else {
69  QString removeError;
70  ok = FileSystem::remove(di.filePath(), &removeError);
71  if (!ok) {
72  _error += PropagateLocalRemove::tr("Error removing '%1': %2;").
73  arg(QDir::toNativeSeparators(di.filePath()), removeError) + " ";
74  qDebug() << "Error removing " << di.filePath() << ':' << removeError;
75  }
76  }
77  if (success && !ok) {
78  // We need to delete the entries from the database now from the deleted vector
79  foreach(const auto &it, deleted) {
80  _propagator->_journal->deleteFileRecord(_item->_originalFile + path + QLatin1Char('/') + it.first,
81  it.second);
82  }
83  success = false;
84  deleted.clear();
85  }
86  if (success) {
87  deleted.append(qMakePair(di.fileName(), isDir));
88  }
89  if (!success && ok) {
90  // This succeeded, so we need to delete it from the database now because the caller won't
91  _propagator->_journal->deleteFileRecord(_item->_originalFile + path + QLatin1Char('/') + di.fileName(),
92  isDir);
93  }
94  }
95  if (success) {
96  success = QDir().rmdir(absolute);
97  if (!success) {
98  _error += PropagateLocalRemove::tr("Could not remove folder '%1'")
99  .arg(QDir::toNativeSeparators(absolute)) + " ";
100  qDebug() << "Error removing folder" << absolute;
101  }
102  }
103  return success;
104 }

106 void PropagateLocalRemove::start()
107 {
108  if (_propagator->_abortRequested.fetchAndAddRelaxed(0))
109  return;
110 
111  QString filename = _propagator->_localDir + _item->_file;
112 
113  qDebug() << filename;
114 
115  if( _propagator->localFileNameClash(_item->_file)) {
116  done(SyncFileItem::NormalError, tr("Could not remove %1 because of a local file name clash")
117  .arg(QDir::toNativeSeparators(filename)));
118  return;
119  }
120 
121  if (_item->_isDirectory) {
122  if (QDir(filename).exists() && !removeRecursively(QString())) {
123  done(SyncFileItem::NormalError, _error);
124  return;
125  }
126  } else {
127  QString removeError;
128  if (FileSystem::fileExists(filename)
129  && !FileSystem::remove(filename, &removeError)) {
130  done(SyncFileItem::NormalError, removeError);
131  return;
132  }
133  }
134  emit progress(*_item, 0);
135  _propagator->_journal->deleteFileRecord(_item->_originalFile, _item->_isDirectory);
136  _propagator->_journal->commit("Local remove");
137  done(SyncFileItem::Success);
138 }

140 void PropagateLocalMkdir::start()
141 {
142  if (_propagator->_abortRequested.fetchAndAddRelaxed(0))
143  return;
144 
145  QDir newDir(_propagator->getFilePath(_item->_file));
146  QString newDirStr = QDir::toNativeSeparators(newDir.path());
147 
148  // When turning something that used to be a file into a directory
149  // we need to delete the file first.
150  QFileInfo fi(newDirStr);
151  if (_deleteExistingFile && fi.exists() && fi.isFile()) {
152  QString removeError;
153  if (!FileSystem::remove(newDirStr, &removeError)) {
154  done( SyncFileItem::NormalError,
155  tr("could not delete file %1, error: %2")
156  .arg(newDirStr, removeError));
157  return;
158  }
159  }
160 
161  if( Utility::fsCasePreserving() && _propagator->localFileNameClash(_item->_file ) ) {
162  qDebug() << "WARN: new folder to create locally already exists!";
163  done( SyncFileItem::NormalError, tr("Attention, possible case sensitivity clash with %1").arg(newDirStr) );
164  return;
165  }
166  emit _propagator->touchedFile(newDirStr);
167  QDir localDir(_propagator->_localDir);
168  if (!localDir.mkpath(_item->_file)) {
169  done( SyncFileItem::NormalError, tr("could not create folder %1").arg(newDirStr) );
170  return;
171  }
172 
173  // Insert the directory into the database. The correct etag will be set later,
174  // once all contents have been propagated, because should_update_metadata is true.
175  // Adding an entry with a dummy etag to the database still makes sense here
176  // so the database is aware that this folder exists even if the sync is aborted
177  // before the correct etag is stored.
178  SyncJournalFileRecord record(*_item, newDirStr);
179  record._etag = "_invalid_";
180  if (!_propagator->_journal->setFileRecord(record)) {
181  done(SyncFileItem::FatalError, tr("Error writing metadata to the database"));
182  return;
183  }
184  _propagator->_journal->commit("localMkdir");
185 
186  done(SyncFileItem::Success);
187 }

194 void PropagateLocalRename::start()
195 {
196  if (_propagator->_abortRequested.fetchAndAddRelaxed(0))
197  return;
198 
199  QString existingFile = _propagator->getFilePath(_item->_file);
200  QString targetFile = _propagator->getFilePath(_item->_renameTarget);
201 
202  // if the file is a file underneath a moved dir, the _item->file is equal
203  // to _item->renameTarget and the file is not moved as a result.
204  if (_item->_file != _item->_renameTarget) {
205  emit progress(*_item, 0);
206  qDebug() << "MOVE " << existingFile << " => " << targetFile;
207 
208  if (QString::compare(_item->_file, _item->_renameTarget, Qt::CaseInsensitive) != 0
209  && _propagator->localFileNameClash(_item->_renameTarget)) {
210  // Only use localFileNameClash for the destination if we know that the source was not
211  // the one conflicting (renaming A.txt -> a.txt is OK)
212 
213  // Fixme: the file that is the reason for the clash could be named here,
214  // it would have to come out the localFileNameClash function
215  done(SyncFileItem::NormalError, tr( "File %1 can not be renamed to %2 because of a local file name clash")
216  .arg(QDir::toNativeSeparators(_item->_file)).arg(QDir::toNativeSeparators(_item->_renameTarget)) );
217  return;
218  }
219 
220  emit _propagator->touchedFile(existingFile);
221  emit _propagator->touchedFile(targetFile);
222  QString renameError;
223  if (!FileSystem::rename(existingFile, targetFile, &renameError)) {
224  done(SyncFileItem::NormalError, renameError);
225  return;
226  }
227  }
228 
229  SyncJournalFileRecord oldRecord =
230  _propagator->_journal->getFileRecord(_item->_originalFile);
231  _propagator->_journal->deleteFileRecord(_item->_originalFile);
232 
233  // store the rename file name in the item.
234  const auto oldFile = _item->_file;
235  _item->_file = _item->_renameTarget;
236 
237  SyncJournalFileRecord record(*_item, targetFile);
238  record._path = _item->_renameTarget;
239  if (oldRecord.isValid()) {
240  record._contentChecksum = oldRecord._contentChecksum;
241  record._contentChecksumType = oldRecord._contentChecksumType;
242  }
243 
244  if (!_item->_isDirectory) { // Directories are saved at the end
245  if (!_propagator->_journal->setFileRecord(record)) {
246  done(SyncFileItem::FatalError, tr("Error writing metadata to the database"));
247  return;
248  }
249  } else {
250  if (!PropagateRemoteMove::adjustSelectiveSync(_propagator->_journal, oldFile, _item->_renameTarget)) {
251  done(SyncFileItem::FatalError, tr("Error writing metadata to the database"));
252  return;
253  }
254  }
255 
256  _propagator->_journal->commit("localRename");
257 
258  done(SyncFileItem::Success);
259 }
