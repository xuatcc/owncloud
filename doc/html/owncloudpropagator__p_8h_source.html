24 inline QByteArray parseEtag(const char *header) {
25  if (!header)
26  return QByteArray();
27  QByteArray arr = header;
28 
29  // Weak E-Tags can appear when gzip compression is on, see #3946
30  if (arr.startsWith("W/"))
31  arr = arr.mid(2);
32 
33  // https://github.com/owncloud/client/issues/1195
34  arr.replace("-gzip", "");
35 
36  if(arr.length() >= 2 && arr.startsWith('"') && arr.endsWith('"')) {
37  arr = arr.mid(1, arr.length() - 2);
38  }
39  return arr;
40 }

42 inline QByteArray getEtagFromReply(QNetworkReply *reply)
43 {
44  QByteArray ocEtag = parseEtag(reply->rawHeader("OC-ETag"));
45  QByteArray etag = parseEtag(reply->rawHeader("ETag"));
46  QByteArray ret = ocEtag;
47  if (ret.isEmpty()) {
48  ret = etag;
49  }
50  if (ocEtag.length() > 0 && ocEtag != etag) {
51  qDebug() << "Quite peculiar, we have an etag != OC-Etag [no problem!]" << etag << ocEtag;
52  }
53  return ret;
54 }

59 inline SyncFileItem::Status classifyError(QNetworkReply::NetworkError nerror,
60  int httpCode,
61  bool* anotherSyncNeeded = NULL) {
62  Q_ASSERT (nerror != QNetworkReply::NoError); // we should only be called when there is an error
63 
64  if (nerror > QNetworkReply::NoError && nerror <= QNetworkReply::UnknownProxyError) {
65  // network error or proxy error -> fatal
66  return SyncFileItem::FatalError;
67  }
68 
69  if (httpCode == 503) {
70  // "Service unavailable"
71  // Happens for maintenance mode and other temporary outages
72  return SyncFileItem::FatalError;
73  }
74 
75  if (httpCode == 412) {
76  // "Precondition Failed"
77  // Happens when the e-tag has changed
78  return SyncFileItem::SoftError;
79  }
80 
81  if (httpCode == 423) {
82  // "Locked"
83  // Should be temporary.
84  if (anotherSyncNeeded) { *anotherSyncNeeded = true; }
85  return SyncFileItem::SoftError;
86  }
87 
88  return SyncFileItem::NormalError;
89 }
