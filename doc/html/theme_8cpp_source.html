52 QString Theme::statusHeaderText( SyncResult::Status status ) const
53 {
54  QString resultStr;
55 
56  switch( status ) {
57  case SyncResult::Undefined:
58  resultStr = QCoreApplication::translate("theme", "Status undefined");
59  break;
60  case SyncResult::NotYetStarted:
61  resultStr = QCoreApplication::translate("theme", "Waiting to start sync");
62  break;
63  case SyncResult::SyncRunning:
64  resultStr = QCoreApplication::translate("theme", "Sync is running");
65  break;
66  case SyncResult::Success:
67  resultStr = QCoreApplication::translate("theme", "Sync Success");
68  break;
69  case SyncResult::Problem:
70  resultStr = QCoreApplication::translate("theme", "Sync Success, some files were ignored.");
71  break;
72  case SyncResult::Error:
73  resultStr = QCoreApplication::translate("theme", "Sync Error");
74  break;
75  case SyncResult::SetupError:
76  resultStr = QCoreApplication::translate("theme", "Setup Error" );
77  break;
78  case SyncResult::SyncPrepare:
79  resultStr = QCoreApplication::translate("theme", "Preparing to sync" );
80  break;
81  case SyncResult::SyncAbortRequested:
82  resultStr = QCoreApplication::translate("theme", "Aborting..." );
83  break;
84  case SyncResult::Paused:
85  resultStr = QCoreApplication::translate("theme", "Sync is paused");
86  break;
87  }

118 QIcon Theme::themeIcon( const QString& name, bool sysTray, bool sysTrayMenuVisible ) const
119 {
120  QString flavor;
121  if (sysTray) {
122  flavor = systrayIconFlavor(_mono, sysTrayMenuVisible);
123  } else {
124  flavor = QLatin1String("colored");
125  }
126 
127  QString key = name + "," + flavor;
128  QIcon & cached = _iconCache[key];
129  if (cached.isNull()) {
130  if(QIcon::hasThemeIcon(name)) {
131  // use from theme
132  return cached = QIcon::fromTheme(name);
133  }
134 
135  QList<int> sizes;
136  sizes <<16 << 22 << 32 << 48 << 64 << 128 << 256;
137  foreach (int size, sizes) {
138  QString pixmapName = QString::fromLatin1(":/client/theme/%1/%2-%3.png").arg(flavor).arg(name).arg(size);
139  if (QFile::exists(pixmapName)) {
140  QPixmap px(pixmapName);
141  // HACK, get rid of it by supporting FDO icon themes, this is really just emulating ubuntu-mono
142  if (qgetenv("DESKTOP_SESSION") == "ubuntu") {
143  QBitmap mask = px.createMaskFromColor(Qt::white, Qt::MaskOutColor);
144  QPainter p(&px);
145  p.setPen(QColor("#dfdbd2"));
146  p.drawPixmap(px.rect(), mask, mask.rect());
147  }
148  cached.addPixmap(px);
149  }
150  }
151  if (cached.isNull()) {
152  foreach (int size, sizes) {
153  QString pixmapName = QString::fromLatin1(":/client/resources/%1-%2.png").arg(name).arg(size);
154  if (QFile::exists(pixmapName)) {
155  cached.addFile(pixmapName);
156  }
157  }
158  }
159  }
160 
161  #ifdef Q_OS_MAC
162  #if QT_VERSION >= QT_VERSION_CHECK(5, 6, 0)
163  // This defines the icon as a template and enables automatic macOS color handling
164  // See https://bugreports.qt.io/browse/QTBUG-42109
165  cached.setIsMask(_mono && sysTray && !sysTrayMenuVisible);
166  #endif
167  #endif
168 
169  return cached;
170 }

172 QString Theme::hidpiFileName(const QString &fileName, QPaintDevice *dev)
173 {
174  #if QT_VERSION > QT_VERSION_CHECK(5, 0, 0)
175  qreal devicePixelRatio = dev ? dev->devicePixelRatio() :
176  qApp->primaryScreen()->devicePixelRatio();
177  if (devicePixelRatio <= 1.0) {
178  return fileName;
179  }
180  // try to find a 2x version
181 
182 
183 
184  const int dotIndex = fileName.lastIndexOf(QLatin1Char('.'));
185  if (dotIndex != -1) {
186  QString at2xfileName = fileName;
187  at2xfileName.insert(dotIndex, QStringLiteral("@2x"));
188  if (QFile::exists(at2xfileName)) {
189  return at2xfileName;
190  }
191  }
192  #endif
193  return fileName;
194 }

196 
197 #endif

349 QIcon Theme::syncStateIcon( SyncResult::Status status, bool sysTray, bool sysTrayMenuVisible ) const
350 {
351  // FIXME: Mind the size!
352  QString statusIcon;
353 
354  switch( status ) {

356  // this can happen if no sync connections are configured.
357  statusIcon = QLatin1String("state-information");
358  break;
359  case SyncResult::NotYetStarted:

363  case SyncResult::SyncAbortRequested:
364  case SyncResult::Paused:

373  break;

375  case SyncResult::SetupError:
376  // FIXME: Use state-problem once we have an icon.
377  default:
378  statusIcon = QLatin1String("state-error");
379  }
380 
381  return themeIcon( statusIcon, sysTray, sysTrayMenuVisible );
