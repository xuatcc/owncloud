87 bool parseChecksumHeader(const QByteArray& header, QByteArray* type, QByteArray* checksum)
88 {
89  if (header.isEmpty()) {
90  type->clear();
91  checksum->clear();
92  return true;
93  }
94 
95  const auto idx = header.indexOf(':');
96  if (idx < 0) {
97  return false;
98  }
99 
100  *type = header.left(idx);
101  *checksum = header.mid(idx + 1);
102  return true;
103 }

144 QByteArray ComputeChecksum::computeNow(const QString& filePath, const QByteArray& checksumType)
145 {
146  if( checksumType == checkSumMD5C ) {
147  return FileSystem::calcMd5(filePath);
148  } else if( checksumType == checkSumSHA1C ) {
149  return FileSystem::calcSha1(filePath);
150  }
151 #ifdef ZLIB_FOUND
152  else if( checksumType == checkSumAdlerC) {
153  return FileSystem::calcAdler32(filePath);
154  }
155 #endif
156  // for an unknown checksum or no checksum, we're done right now
157  if( !checksumType.isEmpty() ) {
158  qDebug() << "Unknown checksum type:" << checksumType;
159  }
160  return QByteArray();
161 }

179 void ValidateChecksumHeader::start(const QString& filePath, const QByteArray& checksumHeader)
180 {
181  // If the incoming header is empty no validation can happen. Just continue.
182  if( checksumHeader.isEmpty() ) {
183  emit validated(QByteArray(), QByteArray());
184  return;
185  }
186 
187  if( !parseChecksumHeader(checksumHeader, &_expectedChecksumType, &_expectedChecksum) ) {
188  qDebug() << "Checksum header malformed:" << checksumHeader;
189  emit validationFailed(tr("The checksum header is malformed."));
190  return;
191  }
192 
193  auto calculator = new ComputeChecksum(this);
194  calculator->setChecksumType(_expectedChecksumType);
195  connect(calculator, SIGNAL(done(QByteArray,QByteArray)),
196  SLOT(slotChecksumCalculated(QByteArray,QByteArray)));
197  calculator->start(filePath);
198 }

200 void ValidateChecksumHeader::slotChecksumCalculated(const QByteArray& checksumType,
201  const QByteArray& checksum)
202 {
203  if( checksumType != _expectedChecksumType ) {
204  emit validationFailed(tr("The checksum header contained an unknown checksum type '%1'").arg(
205  QString::fromLatin1(_expectedChecksumType)));
206  return;
207  }
208  if( checksum != _expectedChecksum ) {
209  emit validationFailed(tr("The downloaded file does not match the checksum, it will be resumed."));
210  return;
211  }
212  emit validated(checksumType, checksum);
213 }

220 const char* CSyncChecksumHook::hook(const char* path, uint32_t checksumTypeId, void *this_obj)
221 {
222  CSyncChecksumHook* checksumHook = static_cast<CSyncChecksumHook*>(this_obj);
223  QByteArray checksum = checksumHook->compute(QString::fromUtf8(path), checksumTypeId);
224  if (checksum.isNull()) {
225  return NULL;
226  }
227 
228  char* result = (char*)malloc(checksum.size() + 1);
229  memcpy(result, checksum.constData(), checksum.size());
230  result[checksum.size()] = 0;
231  return result;
232 }

234 QByteArray CSyncChecksumHook::compute(const QString& path, int checksumTypeId)
235 {
236  QByteArray checksumType = _journal->getChecksumType(checksumTypeId);
237  if (checksumType.isEmpty()) {
238  qDebug() << "Checksum type" << checksumTypeId << "not found";
239  return QByteArray();
240  }
241 
242  QByteArray checksum = ComputeChecksum::computeNow(path, checksumType);
243  if (checksum.isNull()) {
244  qDebug() << "Failed to compute checksum" << checksumType << "for" << path;
245  return QByteArray();
246  }
247 
248  return checksum;
249 }
