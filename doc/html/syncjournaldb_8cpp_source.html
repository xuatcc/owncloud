61 bool SyncJournalDb::maybeMigrateDb(const QString& localPath, const QString& absoluteJournalPath)
62 {
63  const QString oldDbName = localPath + QLatin1String(".csync_journal.db");
64  if( !FileSystem::fileExists(oldDbName) ) {
65  return true;
66  }
67 
68  const QString newDbName = absoluteJournalPath;
69 
70  // Whenever there is an old db file, migrate it to the new db path.
71  // This is done to make switching from older versions to newer versions
72  // work correctly even if the user had previously used a new version
73  // and therefore already has an (outdated) new-style db file.
74  QString error;
75 
76  if( FileSystem::fileExists( newDbName ) ) {
77  if( !FileSystem::remove(newDbName, &error) ) {
78  qDebug() << "Database migration: Could not remove db file" << newDbName
79  << "due to" << error;
80  return false;
81  }
82  }
83 
84  if( !FileSystem::rename(oldDbName, newDbName, &error) ) {
85  qDebug() << "Database migration: could not rename " << oldDbName
86  << "to" << newDbName << ":" << error;
87  return false;
88  }
89 
90  qDebug() << "Journal successfully migrated from" << oldDbName << "to" << newDbName;
91  return true;
92 }

158 static QString defaultJournalMode(const QString & dbPath)
159 {
160 #ifdef Q_OS_WIN
161  // See #2693: Some exFAT file systems seem unable to cope with the
162  // WAL journaling mode. They work fine with DELETE.
163  QString fileSystem = FileSystem::fileSystemForPath(dbPath);
164  qDebug() << "Detected filesystem" << fileSystem << "for" << dbPath;
165  if (fileSystem.contains("FAT")) {
166  qDebug() << "Filesystem contains FAT - using DELETE journal mode";
167  return "DELETE";
168  }
169 #else
170  Q_UNUSED(dbPath)
171 #endif
172  return "WAL";
173 }

254  "md5 VARCHAR(32)," /* This is the etag. Called md5 for compatibility */
255  // updateDatabaseStructure() will add
256  // fileid
257  // remotePerm
258  // filesize
259  // ignoredChildrenRemote
260  // contentChecksum
261  // contentChecksumTypeId
262  "PRIMARY KEY(phash)"
263  ");");
264

267  }

272  "etag VARCHAR(32),"
273  "errorcount INTEGER,"

297  "path VARCHAR(4096),"
298  "lastTryEtag VARCHAR[32],"
299  "lastTryModtime INTEGER[8],"
300  "retrycount INTEGER,"
301  "errorstring VARCHAR[4096],"

345  "major INTEGER(8),"
346  "minor INTEGER(8),"
347  "patch INTEGER(8),"
348  "custom VARCHAR(256)"
349  ");");
350  if (!createQuery.exec()) {
351  return sqlFail("Create table version", createQuery);

354  bool forceRemoteDiscovery = false;
355 
356  SqlQuery versionQuery("SELECT major, minor, patch FROM version;", _db);
357  if (!versionQuery.next()) {
358  // If there was no entry in the table, it means we are likely upgrading from 1.5
359  qDebug() << Q_FUNC_INFO << "possibleUpgradeFromMirall_1_5 detected!";
360  forceRemoteDiscovery = true;
361 
362  createQuery.prepare("INSERT INTO version VALUES (?1, ?2, ?3, ?4);");
363  createQuery.bindValue(1, MIRALL_VERSION_MAJOR);
364  createQuery.bindValue(2, MIRALL_VERSION_MINOR);
365  createQuery.bindValue(3, MIRALL_VERSION_PATCH);
366  createQuery.bindValue(4, MIRALL_VERSION_BUILD);
367  if (!createQuery.exec()) {
368  return sqlFail("Update version", createQuery);
369  }
370 
371  } else {
372  int major = versionQuery.intValue(0);
373  int minor = versionQuery.intValue(1);
374  int patch = versionQuery.intValue(2);
375 
376  if( major == 1 && minor == 8 && (patch == 0 || patch == 1) ) {
377  qDebug() << Q_FUNC_INFO << "possibleUpgradeFromMirall_1_8_0_or_1 detected!";
378  forceRemoteDiscovery = true;
379  }
380 
381  // There was a bug in versions <2.3.0 that could lead to stale
382  // local files and a remote discovery will fix them.
383  // See #5190 #5242.
384  if( major == 2 && minor < 3) {
385  qDebug() << Q_FUNC_INFO << "upgrade form client < 2.3.0 detected! forcing remote discovery";
386  forceRemoteDiscovery = true;
387  }
388 
389  // Not comparing the BUILD id here, correct?
390  if( !(major == MIRALL_VERSION_MAJOR && minor == MIRALL_VERSION_MINOR && patch == MIRALL_VERSION_PATCH) ) {
391  createQuery.prepare("UPDATE version SET major=?1, minor=?2, patch =?3, custom=?4 "
392  "WHERE major=?5 AND minor=?6 AND patch=?7;");
393  createQuery.bindValue(1, MIRALL_VERSION_MAJOR);
394  createQuery.bindValue(2, MIRALL_VERSION_MINOR);
395  createQuery.bindValue(3, MIRALL_VERSION_PATCH);
396  createQuery.bindValue(4, MIRALL_VERSION_BUILD);
397  createQuery.bindValue(5, major);
398  createQuery.bindValue(6, minor);
399  createQuery.bindValue(7, patch);
400  if (!createQuery.exec()) {
401  return sqlFail("Update version", createQuery);
402  }
403 
404  }
405  }

408

410  if( !rc ) {
411  qDebug() << "WARN: Failed to update the database structure!";
412  }
413 
414  /*
415  * If we are upgrading from a client version older than 1.5,
416  * we cannot read from the database because we need to fetch the files id and etags.
417  *
418  * If 1.8.0 caused missing data in the local tree, so we also don't read from DB
419  * to get back the files that were gone.
420  * In 1.8.1 we had a fix to re-get the data, but this one here is better
421  */
422  if (forceRemoteDiscovery) {
423  forceRemoteDiscoveryNextSyncLocked();
424  }

614 bool SyncJournalDb::updateMetadataTableStructure()
615 {
616  QStringList columns = tableColumns("metadata");
617  bool re = true;
618 
619  // check if the file_id column is there and create it if not
620  if( !checkConnect() ) {
621  return false;
622  }
623 
624  if( columns.indexOf(QLatin1String("fileid")) == -1 ) {
625  SqlQuery query(_db);
626  query.prepare("ALTER TABLE metadata ADD COLUMN fileid VARCHAR(128);");
627  if( !query.exec() ) {
628  sqlFail("updateMetadataTableStructure: Add column fileid", query);
629  re = false;
630  }
631 
632  query.prepare("CREATE INDEX metadata_file_id ON metadata(fileid);");
633  if( ! query.exec() ) {
634  sqlFail("updateMetadataTableStructure: create index fileid", query);
635  re = false;
636  }
637  commitInternal("update database structure: add fileid col");
638  }
639  if( columns.indexOf(QLatin1String("remotePerm")) == -1 ) {
640 
641  SqlQuery query(_db);
642  query.prepare("ALTER TABLE metadata ADD COLUMN remotePerm VARCHAR(128);");
643  if( !query.exec()) {
