80  friend bool operator<(const SyncFileItem& item1, const SyncFileItem& item2) {
81  // Sort by destination
82  auto d1 = item1.destination();
83  auto d2 = item2.destination();
84 
85  // But this we need to order it so the slash come first. It should be this order:
86  // "foo", "foo/bar", "foo-bar"
87  // This is important since we assume that the contents of a folder directly follows
88  // its contents
89 
90  auto data1 = d1.constData();
91  auto data2 = d2.constData();
92 
93  // Find the length of the largest prefix
94  int prefixL = 0;
95  auto minSize = std::min(d1.size(), d2.size());
96  while (prefixL < minSize && data1[prefixL] == data2[prefixL]) { prefixL++; }
97 
98  if (prefixL == d2.size())
99  return false;
100  if (prefixL == d1.size())
101  return true;
102 
103  if (data1[prefixL] == '/')
104  return true;
105  if (data2[prefixL] == '/')
106  return false;
107 
108  return data1[prefixL] < data2[prefixL];
109  }
