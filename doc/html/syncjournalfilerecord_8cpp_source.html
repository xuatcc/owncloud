36 SyncJournalFileRecord::SyncJournalFileRecord(const SyncFileItem &item, const QString &localFileName)
37  : _path(item._file), _modtime(Utility::qDateTimeFromTime_t(item._modtime)),
38  _type(item._type), _etag(item._etag), _fileId(item._fileId), _fileSize(item._size),
39  _remotePerm(item._remotePerm), _serverHasIgnoredFiles(item._serverHasIgnoredFiles),
40  _contentChecksum(item._contentChecksum),
41  _contentChecksumType(item._contentChecksumType)
42 {
43  // use the "old" inode coming with the item for the case where the
44  // filesystem stat fails. That can happen if the the file was removed
45  // or renamed meanwhile. For the rename case we still need the inode to
46  // detect the rename though.
47  _inode = item._inode;
48 
49 #ifdef Q_OS_WIN
50  /* Query the inode:
51  based on code from csync_vio_local.c (csync_vio_local_stat)
52  Get the Windows file id as an inode replacement. */
53 
54  HANDLE h = CreateFileW( (wchar_t*) FileSystem::longWinPath(localFileName).utf16(), 0, FILE_SHARE_READ, NULL, OPEN_EXISTING,
55  FILE_ATTRIBUTE_NORMAL+FILE_FLAG_BACKUP_SEMANTICS, NULL );
56 
57  if( h == INVALID_HANDLE_VALUE ) {
58  qWarning() << "Failed to query the 'inode' because CreateFileW failed for file " << localFileName;
59  } else {
60  BY_HANDLE_FILE_INFORMATION fileInfo;
61 
62  if( GetFileInformationByHandle( h, &fileInfo ) ) {
63  ULARGE_INTEGER FileIndex;
64  FileIndex.HighPart = fileInfo.nFileIndexHigh;
65  FileIndex.LowPart = fileInfo.nFileIndexLow;
66  FileIndex.QuadPart &= 0x0000FFFFFFFFFFFF;
67 
68  /* printf("Index: %I64i\n", FileIndex.QuadPart); */
69 
70  _inode = FileIndex.QuadPart;
71  } else {
72  qWarning() << "Failed to query the 'inode' for file " << localFileName;
73 
74  }
75  CloseHandle(h);
76  }
77 #else
78  struct stat sb;
79  if( stat(QFile::encodeName(localFileName).constData(), &sb) < 0) {
80  qWarning() << "Failed to query the 'inode' for file " << localFileName;
81  } else {
82  _inode = sb.st_ino;
83  }
84 #endif
85  qDebug() << Q_FUNC_INFO << localFileName << "Retrieved inode " << _inode << "(previous item inode: " << item._inode << ")";
86 
87 }

127 SyncJournalErrorBlacklistRecord SyncJournalErrorBlacklistRecord::update(
128  const SyncJournalErrorBlacklistRecord& old, const SyncFileItem& item)
129 {
130  SyncJournalErrorBlacklistRecord entry;
131  bool mayBlacklist =
132  item._errorMayBeBlacklisted // explicitly flagged for blacklisting
133  || (item._httpErrorCode != 0 // or non-local error
134 #ifdef OWNCLOUD_5XX_NO_BLACKLIST
135  && item._httpErrorCode / 100 != 5 // In this configuration, never blacklist error 5xx
136 #endif
137  );
138 
139  if (!mayBlacklist) {
140  qDebug() << "This error is not blacklisted " << item._httpErrorCode << item._errorMayBeBlacklisted;
141  return entry;
142  }
143 
144  static time_t minBlacklistTime(getMinBlacklistTime());
145  static time_t maxBlacklistTime(qMax(getMaxBlacklistTime(), minBlacklistTime));
146 
147  entry._retryCount = old._retryCount + 1;
148  entry._errorString = item._errorString;
149  entry._lastTryModtime = item._modtime;
150  entry._lastTryEtag = item._etag;
151  entry._lastTryTime = Utility::qDateTimeToTime_t(QDateTime::currentDateTime());
152  // The factor of 5 feels natural: 25s, 2 min, 10 min, ~1h, ~5h, ~24h
153  entry._ignoreDuration = old._ignoreDuration * 5;
154  entry._file = item._file;
155  entry._renameTarget = item._renameTarget;
156 
157  if( item._httpErrorCode == 403 ) {
158  qDebug() << "Probably firewall error: " << item._httpErrorCode << ", blacklisting up to 1h only";
159  entry._ignoreDuration = qMin(entry._ignoreDuration, time_t(60*60));
160 
161  } else if( item._httpErrorCode == 413 || item._httpErrorCode == 415 ) {
162  qDebug() << "Fatal Error condition" << item._httpErrorCode << ", maximum blacklist ignore time!";
163  entry._ignoreDuration = maxBlacklistTime;
164  }
165  entry._ignoreDuration = qBound(minBlacklistTime, entry._ignoreDuration, maxBlacklistTime);
166 
167  qDebug() << "blacklisting " << item._file
168  << " for " << entry._ignoreDuration
169  << ", retry count " << entry._retryCount;
170  return entry;
171 }
