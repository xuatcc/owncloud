184 void Logger::setLogFile(const QString & name)
185 {
186  QMutexLocker locker(&_mutex);
187 
188  // Setup CSYNC logging to forward to our own logger
189  csync_set_log_callback(csyncLogCatcher);
190  csync_set_log_level(11);
191 
192  if( _logstream ) {
193  _logstream.reset(0);
194  _logFile.close();
195  }
196 
197  if( name.isEmpty() ) {
198  return;
199  }
200 
201  bool openSucceeded = false;
202  if (name == QLatin1String("-")) {
203  openSucceeded = _logFile.open(1, QIODevice::WriteOnly);
204  } else {
205  _logFile.setFileName( name );
206  openSucceeded = _logFile.open(QIODevice::WriteOnly);
207  }
208 
209  if(!openSucceeded) {
210  locker.unlock(); // Just in case postGuiMessage has a qDebug()
211  postGuiMessage( tr("Error"),
212  QString(tr("<nobr>File '%1'<br/>cannot be opened for writing.<br/><br/>"
213  "The log output can <b>not</b> be saved!</nobr>"))
214  .arg(name));
215  return;
216  }
217 
218  _logstream.reset(new QTextStream( &_logFile ));
219 }

236 void Logger::enterNextLogFile()
237 {
238  if (!_logDirectory.isEmpty()) {
239  QDir dir(_logDirectory);
240  if (!dir.exists()) {
241  dir.mkpath(".");
242  }
243 
244  // Find out what is the file with the highest number if any
245  QStringList files = dir.entryList(QStringList("owncloud.log.*"),
246  QDir::Files);
247  QRegExp rx("owncloud.log.(\\d+)");
248  uint maxNumber = 0;
249  QDateTime now = QDateTime::currentDateTime();
250  foreach(const QString &s, files) {
251  if (rx.exactMatch(s)) {
252  maxNumber = qMax(maxNumber, rx.cap(1).toUInt());
253  if (_logExpire > 0) {
254  QFileInfo fileInfo = dir.absoluteFilePath(s);
255  if (fileInfo.lastModified().addSecs(60*60 * _logExpire) < now) {
256  dir.remove(s);
257  }
258  }
259  }
260  }
261 
262  QString filename = _logDirectory + "/owncloud.log." + QString::number(maxNumber+1);
263  setLogFile(filename);
264  }
265 }
