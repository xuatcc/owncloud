44 class HttpCredentialsAccessManager : public AccessManager {
45 public:
46  HttpCredentialsAccessManager(const HttpCredentials *cred, QObject* parent = 0)
47  : AccessManager(parent), _cred(cred) {}
48 protected:
49  QNetworkReply *createRequest(Operation op, const QNetworkRequest &request, QIODevice *outgoingData) Q_DECL_OVERRIDE {
50  QByteArray credHash = QByteArray(_cred->user().toUtf8()+":"+_cred->password().toUtf8()).toBase64();
51  QNetworkRequest req(request);
52  req.setRawHeader(QByteArray("Authorization"), QByteArray("Basic ") + credHash);
53  //qDebug() << "Request for " << req.url() << "with authorization" << QByteArray::fromBase64(credHash);
54  return AccessManager::createRequest(op, req, outgoingData);
55  }
56 private:
57  const HttpCredentials *_cred;
58 };

128 void HttpCredentials::fetchFromKeychain()
129 {
130  // User must be fetched from config file
131  fetchUser();
132  _certificatePath = _account->credentialSetting(QLatin1String(certifPathC)).toString();
133  _certificatePasswd = _account->credentialSetting(QLatin1String(certifPasswdC)).toString();
134 
135  auto settings = Utility::settingsWithGroup(Theme::instance()->appName());
136  const QString kck = keychainKey(_account->url().toString(), _user );
137 
138  QString key = QString::fromLatin1( "%1/data" ).arg( kck );
139  if( settings && settings->contains(key) ) {
140  // Clean the password from the config file if it is in there.
141  // we do not want a security problem.
142  settings->remove(key);
143  key = QString::fromLatin1( "%1/type" ).arg( kck );
144  settings->remove(key);
145  settings->sync();
146  }
147 
148  if (_ready) {
149  Q_EMIT fetched();
150  } else {
151  ReadPasswordJob *job = new ReadPasswordJob(Theme::instance()->appName());
152  settings->setParent(job); // make the job parent to make setting deleted properly
153  job->setSettings(settings.release());
154 
155  job->setInsecureFallback(false);
156  job->setKey(kck);
157  connect(job, SIGNAL(finished(QKeychain::Job*)), SLOT(slotReadJobDone(QKeychain::Job*)));
158  job->start();
159  }
160 }
161 bool HttpCredentials::stillValid(QNetworkReply *reply)
162 {
163  return ((reply->error() != QNetworkReply::AuthenticationRequiredError)
164  // returned if user or password is incorrect
165  && (reply->error() != QNetworkReply::OperationCanceledError
166  || !reply->property(authenticationFailedC).toBool()));
167 }

169 void HttpCredentials::slotReadJobDone(QKeychain::Job *job)
170 {
171  ReadPasswordJob *readJob = static_cast<ReadPasswordJob*>(job);
172  _password = readJob->textData();
173 
174  if( _user.isEmpty()) {
175  qDebug() << "Strange: User is empty!";
176  }
177 
178  QKeychain::Error error = job->error();
179 
180  if( !_password.isEmpty() && error == NoError ) {
181 
182  // All cool, the keychain did not come back with error.
183  // Still, the password can be empty which indicates a problem and
184  // the password dialog has to be opened.
185  _ready = true;
186  emit fetched();
187  } else {
188  // we come here if the password is empty or any other keychain
189  // error happend.
190 
191  _fetchErrorString = job->error() != EntryNotFound ? job->errorString() : QString();
192 
193  _password = QString();
194  _ready = false;
195  emit fetched();
196  }
197 }

199 void HttpCredentials::invalidateToken()
200 {
201  if (! _password.isEmpty()) {
202  _previousPassword = _password;
203  }
204  _password = QString();
205  _ready = false;
206 
207  // User must be fetched from config file to generate a valid key
208  fetchUser();
209 
210  const QString kck = keychainKey(_account->url().toString(), _user);
211  if( kck.isEmpty() ) {
212  qDebug() << "InvalidateToken: User is empty, bailing out!";
213  return;
214  }
215 
216  DeletePasswordJob *job = new DeletePasswordJob(Theme::instance()->appName());
217  auto settings = Utility::settingsWithGroup(Theme::instance()->appName());
218  settings->setParent(job); // make the job parent to make setting deleted properly
219  job->setSettings(settings.release());
220  job->setInsecureFallback(true);
221  job->setKey(kck);
222  job->start();
223 
224  // Also ensure the password is deleted from the deprecated place
225  // otherwise we'd possibly read and use it again and again.
226  DeletePasswordJob *job2 = new DeletePasswordJob(Theme::instance()->appName());
227  // no job2->setSettings() call here, to make it use the deprecated location.
228  job2->setInsecureFallback(true);
229  job2->setKey(kck);
230  job2->start();
231 
232  // clear the session cookie.
233  _account->clearCookieJar();
234 
235  // let QNAM forget about the password
236  // This needs to be done later in the event loop because we might be called (directly or
237  // indirectly) from QNetworkAccessManagerPrivate::authenticationRequired, which itself
238  // is a called from a BlockingQueuedConnection from the Qt HTTP thread. And clearing the
239  // cache needs to synchronize again with the HTTP thread.
240  QTimer::singleShot(0, this, SLOT(clearQNAMCache()));
241 }

258 void HttpCredentials::persist()
259 {
260  if (_user.isEmpty()) {
261  // We never connected or fetched the user, there is nothing to save.
262  return;
263  }
264  _account->setCredentialSetting(QLatin1String(userC), _user);
265  _account->setCredentialSetting(QLatin1String(certifPathC), _certificatePath);
266  _account->setCredentialSetting(QLatin1String(certifPasswdC), _certificatePasswd);
267  WritePasswordJob *job = new WritePasswordJob(Theme::instance()->appName());
268  auto settings = Utility::settingsWithGroup(Theme::instance()->appName());
269  settings->setParent(job); // make the job parent to make setting deleted properly
270  job->setSettings(settings.release());
271 
272  job->setInsecureFallback(false);
273  connect(job, SIGNAL(finished(QKeychain::Job*)), SLOT(slotWriteJobDone(QKeychain::Job*)));
274  job->setKey(keychainKey(_account->url().toString(), _user));
275  job->setTextData(_password);
276  job->start();
277 }

279 void HttpCredentials::slotWriteJobDone(QKeychain::Job *job)
280 {
281  delete job->settings();
282  switch (job->error()) {
283  case NoError:
284  break;
285  default:
286  qDebug() << "Error while writing password" << job->errorString();
287  }
288  WritePasswordJob *wjob = qobject_cast<WritePasswordJob*>(job);
289  wjob->deleteLater();
290 }

292 void HttpCredentials::slotAuthentication(QNetworkReply* reply, QAuthenticator* authenticator)
293 {
294  Q_UNUSED(authenticator)
295  // Because of issue #4326, we need to set the login and password manually at every requests
296  // Thus, if we reach this signal, those credentials were invalid and we terminate.
297  qDebug() << "Stop request: Authentication failed for " << reply->url().toString();
298  reply->setProperty(authenticationFailedC, true);
299  reply->close();
300 }
