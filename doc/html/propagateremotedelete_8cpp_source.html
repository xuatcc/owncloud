80 void PropagateRemoteDelete::slotDeleteJobFinished()
81 {
82  _propagator->_activeJobList.removeOne(this);
83 
84  Q_ASSERT(_job);
85 
86  qDebug() << Q_FUNC_INFO << _job->reply()->request().url() << "FINISHED WITH STATUS"
87  << _job->reply()->error()
88  << (_job->reply()->error() == QNetworkReply::NoError ? QLatin1String("") : _job->reply()->errorString());
89 
90  QNetworkReply::NetworkError err = _job->reply()->error();
91  const int httpStatus = _job->reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
92  _item->_httpErrorCode = httpStatus;
93 
94  if (err != QNetworkReply::NoError && err != QNetworkReply::ContentNotFoundError) {
95 
96  if( checkForProblemsWithShared(_item->_httpErrorCode,
97  tr("The file has been removed from a read only share. It was restored.")) ) {
98  return;
99  }
100 
101  SyncFileItem::Status status = classifyError(err, _item->_httpErrorCode,
102  &_propagator->_anotherSyncNeeded);
103  done(status, _job->errorString());
104  return;
105  }
106 
107  _item->_requestDuration = _job->duration();
108  _item->_responseTimeStamp = _job->responseTimestamp();
109 
110  // A 404 reply is also considered a success here: We want to make sure
111  // a file is gone from the server. It not being there in the first place
112  // is ok. This will happen for files that are in the DB but not on
113  // the server or the local file system.
114  if (httpStatus != 204 && httpStatus != 404) {
115  // Normally we expect "204 No Content"
116  // If it is not the case, it might be because of a proxy or gateway intercepting the request, so we must
117  // throw an error.
118  done(SyncFileItem::NormalError, tr("Wrong HTTP code returned by server. Expected 204, but received \"%1 %2\".")
119  .arg(_item->_httpErrorCode).arg(_job->reply()->attribute(QNetworkRequest::HttpReasonPhraseAttribute).toString()));
120  return;
121  }
122 
123  _propagator->_journal->deleteFileRecord(_item->_originalFile, _item->_isDirectory);
124  _propagator->_journal->commit("Remote Remove");
125  done(SyncFileItem::Success);
126 }
