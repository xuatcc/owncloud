184 class OWNCLOUDSYNC_EXPORT PropagateDirectory : public PropagatorJob {
185  Q_OBJECT
186 public:
187  // e.g: create the directory
188  QScopedPointer<PropagateItemJob>_firstJob;
189 
190  // all the sub files or sub directories.
191  QVector<PropagatorJob *> _subJobs;
192 
193  SyncFileItemPtr _item;
194 
195  int _jobsFinished; // number of jobs that have completed
196  int _runningNow; // number of subJobs running right now
197  SyncFileItem::Status _hasError; // NoStatus, or NormalError / SoftError if there was an error
198  int _firstUnfinishedSubJob;
199 
200  explicit PropagateDirectory(OwncloudPropagator *propagator, const SyncFileItemPtr &item = SyncFileItemPtr(new SyncFileItem))
201  : PropagatorJob(propagator)
202  , _firstJob(0), _item(item), _jobsFinished(0), _runningNow(0), _hasError(SyncFileItem::NoStatus), _firstUnfinishedSubJob(0)
203  { }
204 
205  virtual ~PropagateDirectory() {
206  qDeleteAll(_subJobs);
207  }
208 
209  void append(PropagatorJob *subJob) {
210  _subJobs.append(subJob);
211  }
212 
213  virtual bool scheduleNextJob() Q_DECL_OVERRIDE;
214  virtual JobParallelism parallelism() Q_DECL_OVERRIDE;
215  virtual void abort() Q_DECL_OVERRIDE {
216  if (_firstJob)
217  _firstJob->abort();
218  foreach (PropagatorJob *j, _subJobs)
219  j->abort();
220  }
221 
222  void increaseAffectedCount() {
223  _firstJob->_item->_affectedItems++;
224  }
225 
226  void finalize();
227 
228  qint64 committedDiskSpace() const Q_DECL_OVERRIDE;
229 
230 private slots:
231  bool possiblyRunNextJob(PropagatorJob *next) {
232  if (next->_state == NotYetStarted) {
233  connect(next, SIGNAL(finished(SyncFileItem::Status)), this, SLOT(slotSubJobFinished(SyncFileItem::Status)), Qt::QueuedConnection);
234  connect(next, SIGNAL(itemCompleted(const SyncFileItem &, const PropagatorJob &)),
235  this, SIGNAL(itemCompleted(const SyncFileItem &, const PropagatorJob &)));
236  connect(next, SIGNAL(progress(const SyncFileItem &,quint64)), this, SIGNAL(progress(const SyncFileItem &,quint64)));
237  connect(next, SIGNAL(ready()), this, SIGNAL(ready()));
238  _runningNow++;
239  }
240  return next->scheduleNextJob();
241  }
242 
243  void slotSubJobFinished(SyncFileItem::Status status);
244 };
