40 class TokenCredentialsAccessManager : public AccessManager {
41 public:
42  friend class TokenCredentials;
43  TokenCredentialsAccessManager(const TokenCredentials *cred, QObject* parent = 0)
44  : AccessManager(parent), _cred(cred) {}
45 protected:
46  QNetworkReply *createRequest(Operation op, const QNetworkRequest &request, QIODevice *outgoingData) {
47  if (_cred->user().isEmpty() || _cred->password().isEmpty()) {
48  qWarning() << Q_FUNC_INFO << "Empty user/password provided!";
49  }
50 
51  QNetworkRequest req(request);
52 
53  QByteArray credHash = QByteArray(_cred->user().toUtf8()+":"+_cred->password().toUtf8()).toBase64();
54  req.setRawHeader(QByteArray("Authorization"), QByteArray("Basic ") + credHash);
55 
56  // A pre-authenticated cookie
57  QByteArray token = _cred->_token.toUtf8();
58  if (token.length() > 0) {
59  setRawCookie(token, request.url());
60  }
61 
62  return AccessManager::createRequest(op, req, outgoingData);
63  }
64 private:
65  const TokenCredentials *_cred;
66 };

124 bool TokenCredentials::stillValid(QNetworkReply *reply)
125 {
126  return ((reply->error() != QNetworkReply::AuthenticationRequiredError)
127  // returned if user/password or token are incorrect
128  && (reply->error() != QNetworkReply::OperationCanceledError
129  || !reply->property(authenticationFailedC).toBool()));
130 }

152 void TokenCredentials::slotAuthentication(QNetworkReply* reply, QAuthenticator* authenticator)
153 {
154  Q_UNUSED(authenticator)
155  // we cannot use QAuthenticator, because it sends username and passwords with latin1
156  // instead of utf8 encoding. Instead, we send it manually. Thus, if we reach this signal,
157  // those credentials were invalid and we terminate.
158  qDebug() << "Stop request: Authentication failed for " << reply->url().toString();
159  reply->setProperty(authenticationFailedC, true);
160  reply->close();
161 }
