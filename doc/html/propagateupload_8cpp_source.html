127 bool PollJob::finished()
128 {
129  QNetworkReply::NetworkError err = reply()->error();
130  if (err != QNetworkReply::NoError) {
131  _item->_httpErrorCode = reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
132  _item->_status = classifyError(err, _item->_httpErrorCode);
133  _item->_errorString = reply()->errorString();
134 
135  if (reply()->hasRawHeader("OC-ErrorString")) {
136  _item->_errorString = reply()->rawHeader("OC-ErrorString");
137  }
138 
139  if (_item->_status == SyncFileItem::FatalError || _item->_httpErrorCode >= 400) {
140  if (_item->_status != SyncFileItem::FatalError
141  && _item->_httpErrorCode != 503) {
142  SyncJournalDb::PollInfo info;
143  info._file = _item->_file;
144  // no info._url removes it from the database
145  _journal->setPollInfo(info);
146  _journal->commit("remove poll info");
147 
148  }
149  emit finishedSignal();
150  return true;
151  }
152  start();
153  return false;
154  }
155 
156  bool ok = false;
157  QByteArray jsonData = reply()->readAll().trimmed();
158  qDebug() << Q_FUNC_INFO << ">" << jsonData << "<" << reply()->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
159  QVariantMap status = QtJson::parse(QString::fromUtf8(jsonData), ok).toMap();
160  if (!ok || status.isEmpty()) {
161  _item->_errorString = tr("Invalid JSON reply from the poll URL");
162  _item->_status = SyncFileItem::NormalError;
163  emit finishedSignal();
164  return true;
165  }
166 
167  if (status["unfinished"].isValid()) {
168  start();
169  return false;
170  }
171 
172  _item->_errorString = status["error"].toString();
173  _item->_status = _item->_errorString.isEmpty() ? SyncFileItem::Success : SyncFileItem::NormalError;
174  _item->_fileId = status["fileid"].toByteArray();
175  _item->_etag = status["etag"].toByteArray();
176  _item->_responseTimeStamp = responseTimestamp();
177 
178  SyncJournalDb::PollInfo info;
179  info._file = _item->_file;
180  // no info._url removes it from the database
181  _journal->setPollInfo(info);
182  _journal->commit("remove poll info");
183 
184  emit finishedSignal();
185  return true;
186 }

215 void PropagateUploadFileCommon::slotComputeContentChecksum()
216 {
217  if (_propagator->_abortRequested.fetchAndAddRelaxed(0)) {
218  return;
219  }
220 
221  const QString filePath = _propagator->getFilePath(_item->_file);
222 
223  // remember the modtime before checksumming to be able to detect a file
224  // change during the checksum calculation
225  _item->_modtime = FileSystem::getModTime(filePath);
226 
227  _stopWatch.start();
228 
229  QByteArray checksumType = contentChecksumType();
230 
231  // Maybe the discovery already computed the checksum?
232  if (_item->_contentChecksumType == checksumType
233  && !_item->_contentChecksum.isEmpty()) {
234  slotComputeTransmissionChecksum(checksumType, _item->_contentChecksum);
235  return;
236  }
237 
238  // Compute the content checksum.
239  auto computeChecksum = new ComputeChecksum(this);
240  computeChecksum->setChecksumType(checksumType);
241 
242  connect(computeChecksum, SIGNAL(done(QByteArray,QByteArray)),
243  SLOT(slotComputeTransmissionChecksum(QByteArray,QByteArray)));
244  connect(computeChecksum, SIGNAL(done(QByteArray,QByteArray)),
245  computeChecksum, SLOT(deleteLater()));
246  computeChecksum->start(filePath);
247 }

249 void PropagateUploadFileCommon::slotComputeTransmissionChecksum(const QByteArray& contentChecksumType, const QByteArray& contentChecksum)
250 {
251  _item->_contentChecksum = contentChecksum;
252  _item->_contentChecksumType = contentChecksumType;
253 
254  _stopWatch.addLapTime(QLatin1String("ContentChecksum"));
255  _stopWatch.start();
256 
257  // Reuse the content checksum as the transmission checksum if possible
258  const auto supportedTransmissionChecksums =
259  _propagator->account()->capabilities().supportedChecksumTypes();
260  if (supportedTransmissionChecksums.contains(contentChecksumType)) {
261  slotStartUpload(contentChecksumType, contentChecksum);
262  return;
263  }
264 
265  // Compute the transmission checksum.
266  auto computeChecksum = new ComputeChecksum(this);
267  if (uploadChecksumEnabled()) {
268  computeChecksum->setChecksumType(_propagator->account()->capabilities().uploadChecksumType());
269  } else {
270  computeChecksum->setChecksumType(QByteArray());
271  }
272 
273  connect(computeChecksum, SIGNAL(done(QByteArray,QByteArray)),
274  SLOT(slotStartUpload(QByteArray,QByteArray)));
275  connect(computeChecksum, SIGNAL(done(QByteArray,QByteArray)),
276  computeChecksum, SLOT(deleteLater()));
277  const QString filePath = _propagator->getFilePath(_item->_file);
278  computeChecksum->start(filePath);
279 }

281 void PropagateUploadFileCommon::slotStartUpload(const QByteArray& transmissionChecksumType, const QByteArray& transmissionChecksum)
282 {
283  // Remove ourselfs from the list of active job, before any posible call to done()
284  // When we start chunks, we will add it again, once for every chunks.
285  _propagator->_activeJobList.removeOne(this);
286 
287  _transmissionChecksum = transmissionChecksum;
288  _transmissionChecksumType = transmissionChecksumType;
289 
290  if (_item->_contentChecksum.isEmpty() && _item->_contentChecksumType.isEmpty()) {
291  // If the _contentChecksum was not set, reuse the transmission checksum as the content checksum.
292  _item->_contentChecksum = transmissionChecksum;
293  _item->_contentChecksumType = transmissionChecksumType;
294  }
295 
296  const QString fullFilePath = _propagator->getFilePath(_item->_file);
297 
298  if (!FileSystem::fileExists(fullFilePath)) {
299  done(SyncFileItem::SoftError, tr("File Removed"));
300  return;
301  }
302  _stopWatch.addLapTime(QLatin1String("TransmissionChecksum"));
303 
304  time_t prevModtime = _item->_modtime; // the _item value was set in PropagateUploadFile::start()
305  // but a potential checksum calculation could have taken some time during which the file could
306  // have been changed again, so better check again here.
307 
308  _item->_modtime = FileSystem::getModTime(fullFilePath);
309  if( prevModtime != _item->_modtime ) {
310  _propagator->_anotherSyncNeeded = true;
311  done(SyncFileItem::SoftError, tr("Local file changed during syncing. It will be resumed."));
312  return;
313  }
314 
315  quint64 fileSize = FileSystem::getSize(fullFilePath);
316  _item->_size = fileSize;
317 
318  // But skip the file if the mtime is too close to 'now'!
319  // That usually indicates a file that is still being changed
320  // or not yet fully copied to the destination.
321  if (fileIsStillChanging(*_item)) {
322  _propagator->_anotherSyncNeeded = true;
323  done(SyncFileItem::SoftError, tr("Local file changed during sync."));
324  return;
325  }
326 
327  doStartUpload();
328 }

376 qint64 UploadDevice::readData(char* data, qint64 maxlen) {
377  //qDebug() << Q_FUNC_INFO << maxlen << _read << _size << _bandwidthQuota;
378  if (_data.size() - _read <= 0) {
379  // at end
380  if (_bandwidthManager) {
381  _bandwidthManager->unregisterUploadDevice(this);
382  }
383  return -1;
384  }
385  maxlen = qMin(maxlen, _data.size() - _read);
386  if (maxlen == 0) {
387  return 0;
388  }
389  if (isChoked()) {
390  return 0;
391  }
392  if (isBandwidthLimited()) {
393  maxlen = qMin(maxlen, _bandwidthQuota);
394  if (maxlen <= 0) { // no quota
395  //qDebug() << "no quota";
396  return 0;
397  }
398  _bandwidthQuota -= maxlen;
399  }
400  std::memcpy(data, _data.data()+_read, maxlen);
401  _read += maxlen;
402  return maxlen;
403 }

519 QMap<QByteArray, QByteArray> PropagateUploadFileCommon::headers()
520 {
521  QMap<QByteArray, QByteArray> headers;
522  headers["OC-Async"] = "1";
523  headers["Content-Type"] = "application/octet-stream";
524  headers["X-OC-Mtime"] = QByteArray::number(qint64(_item->_modtime));
525 
526  if(_item->_file.contains(".sys.admin#recall#")) {
527  // This is a file recall triggered by the admin. Note: the
528  // recall list file created by the admin and downloaded by the
529  // client (.sys.admin#recall#) also falls into this category
530  // (albeit users are not supposed to mess up with it)
531 
532  // We use a special tag header so that the server may decide to store this file away in some admin stage area
533  // And not directly in the user's area (which would trigger redownloads etc).
534  headers["OC-Tag"] = ".sys.admin#recall#";
535  }
536 
537  if (!_item->_etag.isEmpty() && _item->_etag != "empty_etag"
538  && _item->_instruction != CSYNC_INSTRUCTION_NEW // On new files never send a If-Match
539  && _item->_instruction != CSYNC_INSTRUCTION_TYPE_CHANGE
540  && !_deleteExisting
541  ) {
542  // We add quotes because the owncloud server always adds quotes around the etag, and
543  // csync_owncloud.c's owncloud_file_id always strips the quotes.
544  headers["If-Match"] = '"' + _item->_etag + '"';
545  }
546  return headers;
547 }
