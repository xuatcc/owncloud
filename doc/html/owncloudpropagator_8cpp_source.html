104 static bool blacklistCheck(SyncJournalDb* journal, const SyncFileItem& item)
105 {
106  SyncJournalErrorBlacklistRecord oldEntry = journal->errorBlacklistEntry(item._file);
107  SyncJournalErrorBlacklistRecord newEntry = SyncJournalErrorBlacklistRecord::update(oldEntry, item);
108 
109  if (newEntry.isValid()) {
110  journal->updateErrorBlacklistEntry(newEntry);
111  // Also clear upload info if any so we don't resume from the same transfer-id if there was too many failures (#5344)
112  // (maybe the reason is that the state for this transfer id is broken on the server.)
113  if (newEntry._retryCount > 3) {
114  journal->setUploadInfo(item._file, SyncJournalDb::UploadInfo());
115  }
116  } else if (oldEntry.isValid()) {
117  journal->wipeErrorBlacklistEntry(item._file);
118  }
119 
120  return newEntry.isValid();
121 }

123 void PropagateItemJob::done(SyncFileItem::Status status, const QString &errorString)
124 {
125  _state = Finished;
126  if (_item->_isRestoration) {
127  if( status == SyncFileItem::Success || status == SyncFileItem::Conflict) {
128  status = SyncFileItem::Restoration;
129  } else {
130  _item->_errorString += tr("; Restoration Failed: %1").arg(errorString);
131  }
132  } else {
133  if( _item->_errorString.isEmpty() ) {
134  _item->_errorString = errorString;
135  }
136  }
137 
138  if( _propagator->_abortRequested.fetchAndAddRelaxed(0) &&
139  (status == SyncFileItem::NormalError || status == SyncFileItem::FatalError)) {
140  // an abort request is ongoing. Change the status to Soft-Error
141  status = SyncFileItem::SoftError;
142  }
143 
144  switch( status ) {
145  case SyncFileItem::SoftError:
146  case SyncFileItem::FatalError:
147  // do not blacklist in case of soft error or fatal error.
148  break;
149  case SyncFileItem::NormalError:
150  if (blacklistCheck(_propagator->_journal, *_item) && _item->_hasBlacklistEntry) {
151  // do not error if the item was, and continues to be, blacklisted
152  status = SyncFileItem::FileIgnored;
153  _item->_errorString.prepend(tr("Continue blacklisting:") + " ");
154  }
155  break;
156  case SyncFileItem::Success:
157  case SyncFileItem::Restoration:
158  if( _item->_hasBlacklistEntry ) {
159  // wipe blacklist entry.
160  _propagator->_journal->wipeErrorBlacklistEntry(_item->_file);
161  // remove a blacklist entry in case the file was moved.
162  if( _item->_originalFile != _item->_file ) {
163  _propagator->_journal->wipeErrorBlacklistEntry(_item->_originalFile);
164  }
165  }
166  break;
167  case SyncFileItem::Conflict:
168  case SyncFileItem::FileIgnored:
169  case SyncFileItem::NoStatus:
170  // nothing
171  break;
172  }
173 
174  _item->_status = status;
175 
176  emit itemCompleted(*_item, *this);
177  emit finished(status);
178 }

186 bool PropagateItemJob::checkForProblemsWithShared(int httpStatusCode, const QString& msg)
187 {
188  PropagateItemJob *newJob = NULL;
189 
190  if( httpStatusCode == 403 && _propagator->isInSharedDirectory(_item->_file )) {
191  if( !_item->_isDirectory ) {
192  SyncFileItemPtr downloadItem(new SyncFileItem(*_item));
193  if (downloadItem->_instruction == CSYNC_INSTRUCTION_NEW
194  || downloadItem->_instruction == CSYNC_INSTRUCTION_TYPE_CHANGE) {
195  // don't try to recover pushing new files
196  return false;
197  } else if (downloadItem->_instruction == CSYNC_INSTRUCTION_SYNC) {
198  // we modified the file locally, just create a conflict then
199  downloadItem->_instruction = CSYNC_INSTRUCTION_CONFLICT;
200 
201  // HACK to avoid continuation: See task #1448: We do not know the _modtime from the
202  // server, at this point, so just set the current one. (rather than the one locally)
203  downloadItem->_modtime = Utility::qDateTimeToTime_t(QDateTime::currentDateTime());
204  } else {
205  // the file was removed or renamed, just recover the old one
206  downloadItem->_instruction = CSYNC_INSTRUCTION_SYNC;
207  }
208  downloadItem->_direction = SyncFileItem::Down;
209  newJob = new PropagateDownloadFile(_propagator, downloadItem);
210  } else {
211  // Directories are harder to recover.
212  // But just re-create the directory, next sync will be able to recover the files
213  SyncFileItemPtr mkdirItem(new SyncFileItem(*_item));
214  mkdirItem->_instruction = CSYNC_INSTRUCTION_NEW;
215  mkdirItem->_direction = SyncFileItem::Down;
216  newJob = new PropagateLocalMkdir(_propagator, mkdirItem);
217  // Also remove the inodes and fileid from the db so no further renames are tried for
218  // this item.
219  _propagator->_journal->avoidRenamesOnNextSync(_item->_file);
220  _propagator->_anotherSyncNeeded = true;
221  }
222  if( newJob ) {
223  newJob->setRestoreJobMsg(msg);
224  _restoreJob.reset(newJob);
225  connect(_restoreJob.data(), SIGNAL(itemCompleted(const SyncFileItemPtr &, const PropagatorJob &)),
226  this, SLOT(slotRestoreJobCompleted(const SyncFileItemPtr &)));
227  QMetaObject::invokeMethod(newJob, "start");
228  }
229  return true;
230  }
231  return false;
232 }

252 PropagateItemJob* OwncloudPropagator::createJob(const SyncFileItemPtr &item) {
253  bool deleteExisting = item->_instruction == CSYNC_INSTRUCTION_TYPE_CHANGE;
254  switch(item->_instruction) {
255  case CSYNC_INSTRUCTION_REMOVE:
256  if (item->_direction == SyncFileItem::Down) return new PropagateLocalRemove(this, item);
257  else return new PropagateRemoteDelete(this, item);
258  case CSYNC_INSTRUCTION_NEW:
259  case CSYNC_INSTRUCTION_TYPE_CHANGE:
260  if (item->_isDirectory) {
261  if (item->_direction == SyncFileItem::Down) {
262  auto job = new PropagateLocalMkdir(this, item);
263  job->setDeleteExistingFile(deleteExisting);
264  return job;
265  } else {
266  auto job = new PropagateRemoteMkdir(this, item);
267  job->setDeleteExisting(deleteExisting);
268  return job;
269  }
270  } //fall through
271  case CSYNC_INSTRUCTION_SYNC:
272  case CSYNC_INSTRUCTION_CONFLICT:
273  if (item->_direction != SyncFileItem::Up) {
274  auto job = new PropagateDownloadFile(this, item);
275  job->setDeleteExistingFolder(deleteExisting);
276  return job;
277  } else {
278  PropagateUploadFileCommon *job = 0;
279  if (item->_size > chunkSize() && account()->capabilities().chunkingNg()) {
280  job = new PropagateUploadFileNG(this, item);
281  } else {
282  job = new PropagateUploadFileV1(this, item);
283  }
284  job->setDeleteExisting(deleteExisting);
285  return job;
286  }
287  case CSYNC_INSTRUCTION_RENAME:
288  if (item->_direction == SyncFileItem::Up) {
289  return new PropagateRemoteMove(this, item);
290  } else {
291  return new PropagateLocalRename(this, item);
292  }
293  case CSYNC_INSTRUCTION_IGNORE:
294  case CSYNC_INSTRUCTION_ERROR:
295  return new PropagateIgnoreJob(this, item);
296  default:
297  return 0;
298  }
299  return 0;
300 }

302 void OwncloudPropagator::start(const SyncFileItemVector& items)
303 {
304  Q_ASSERT(std::is_sorted(items.begin(), items.end()));
305 
306  /* This builds all the jobs needed for the propagation.
307  * Each directory is a PropagateDirectory job, which contains the files in it.
308  * In order to do that we loop over the items. (which are sorted by destination)
309  * When we enter a directory, we can create the directory job and push it on the stack. */
310 
311  _rootJob.reset(new PropagateDirectory(this));
312  QStack<QPair<QString /* directory name */, PropagateDirectory* /* job */> > directories;
313  directories.push(qMakePair(QString(), _rootJob.data()));
314  QVector<PropagatorJob*> directoriesToRemove;
315  QString removedDirectory;
316  foreach(const SyncFileItemPtr &item, items) {
317 
318  if (!removedDirectory.isEmpty() && item->_file.startsWith(removedDirectory)) {
319  // this is an item in a directory which is going to be removed.
320  PropagateDirectory *delDirJob = qobject_cast<PropagateDirectory*>(directoriesToRemove.first());
321 
322  if (item->_instruction == CSYNC_INSTRUCTION_REMOVE) {
323  // already taken care of. (by the removal of the parent directory)
324 
325  // increase the number of subjobs that would be there.
326  if( delDirJob ) {
327  delDirJob->increaseAffectedCount();
328  }
329  continue;
330  } else if (item->_isDirectory
331  && (item->_instruction == CSYNC_INSTRUCTION_NEW
332  || item->_instruction == CSYNC_INSTRUCTION_TYPE_CHANGE)) {
333  // create a new directory within a deleted directory? That can happen if the directory
334  // etag was not fetched properly on the previous sync because the sync was aborted
335  // while uploading this directory (which is now removed). We can ignore it.
336  if( delDirJob ) {
337  delDirJob->increaseAffectedCount();
338  }
339  continue;
340  } else if (item->_instruction == CSYNC_INSTRUCTION_IGNORE) {
341  continue;
342  } else if (item->_instruction == CSYNC_INSTRUCTION_RENAME) {
343  // all is good, the rename will be executed before the directory deletion
344  } else {
345  qWarning() << "WARNING: Job within a removed directory? This should not happen!"
346  << item->_file << item->_instruction;
347  }
348  }
349 
350  while (!item->destination().startsWith(directories.top().first)) {
351  directories.pop();
352  }
353 
354  if (item->_isDirectory) {
355  PropagateDirectory *dir = new PropagateDirectory(this, item);
356  dir->_firstJob.reset(createJob(item));
357 
358  if (item->_instruction == CSYNC_INSTRUCTION_TYPE_CHANGE
359  && item->_direction == SyncFileItem::Up) {
360  // Skip all potential uploads to the new folder.
361  // Processing them now leads to problems with permissions:
362  // checkForPermissions() has already run and used the permissions
363  // of the file we're about to delete to decide whether uploading
364  // to the new dir is ok...
365  foreach(const SyncFileItemPtr &item2, items) {
366  if (item2->destination().startsWith(item->destination() + "/")) {
367  item2->_instruction = CSYNC_INSTRUCTION_NONE;
368  _anotherSyncNeeded = true;
369  }
370  }
371  }
372 
373  if (item->_instruction == CSYNC_INSTRUCTION_REMOVE) {
374  // We do the removal of directories at the end, because there might be moves from
375  // these directories that will happen later.
376  directoriesToRemove.prepend(dir);
377  removedDirectory = item->_file + "/";
378 
379  // We should not update the etag of parent directories of the removed directory
380  // since it would be done before the actual remove (issue #1845)
381  // NOTE: Currently this means that we don't update those etag at all in this sync,
382  // but it should not be a problem, they will be updated in the next sync.
383  for (int i = 0; i < directories.size(); ++i) {
384  if (directories[i].second->_item->_instruction == CSYNC_INSTRUCTION_UPDATE_METADATA)
385  directories[i].second->_item->_instruction = CSYNC_INSTRUCTION_NONE;
386  }
387  } else {
388  PropagateDirectory* currentDirJob = directories.top().second;
389  currentDirJob->append(dir);
390  }
391  directories.push(qMakePair(item->destination() + "/" , dir));
392  } else if (PropagateItemJob* current = createJob(item)) {
393  if (item->_instruction == CSYNC_INSTRUCTION_TYPE_CHANGE) {
394  // will delete directories, so defer execution
395  directoriesToRemove.prepend(current);
396  removedDirectory = item->_file + "/";
397  } else {
398  directories.top().second->append(current);
399  }
400  }
401  }
402 
403  foreach(PropagatorJob* it, directoriesToRemove) {
404  _rootJob->append(it);
405  }
406 
407  connect(_rootJob.data(), SIGNAL(itemCompleted(const SyncFileItem &, const PropagatorJob &)),
408  this, SIGNAL(itemCompleted(const SyncFileItem &, const PropagatorJob &)));
409  connect(_rootJob.data(), SIGNAL(progress(const SyncFileItem &,quint64)), this, SIGNAL(progress(const SyncFileItem &,quint64)));
410  connect(_rootJob.data(), SIGNAL(finished(SyncFileItem::Status)), this, SLOT(emitFinished(SyncFileItem::Status)));
411  connect(_rootJob.data(), SIGNAL(ready()), this, SLOT(scheduleNextJob()), Qt::QueuedConnection);
412 
413  qDebug() << "Using QNAM/HTTP parallel code path";
414 
415  QTimer::singleShot(0, this, SLOT(scheduleNextJob()));
416 }

463 bool OwncloudPropagator::localFileNameClash( const QString& relFile )
464 {
465  bool re = false;
466  const QString file( _localDir + relFile );
467 
468  if( !file.isEmpty() && Utility::fsCasePreserving() ) {
469 #ifdef Q_OS_MAC
470  QFileInfo fileInfo(file);
471  if (!fileInfo.exists()) {
472  re = false;
473  qDebug() << Q_FUNC_INFO << "No valid fileinfo";
474  } else {
475  // Need to normalize to composited form because of
476  // https://bugreports.qt-project.org/browse/QTBUG-39622
477  const QString cName = fileInfo.canonicalFilePath().normalized(QString::NormalizationForm_C);
478  // qDebug() << Q_FUNC_INFO << "comparing " << cName << " with " << file;
479  bool equal = (file == cName);
480  re = (!equal && ! cName.endsWith(relFile, Qt::CaseSensitive) );
481  // qDebug() << Q_FUNC_INFO << "Returning for localFileNameClash: " << re;
482  }
483 #elif defined(Q_OS_WIN)
484  const QString file( _localDir + relFile );
485  qDebug() << "CaseClashCheck for " << file;
486  WIN32_FIND_DATA FindFileData;
487  HANDLE hFind;
488 
489  hFind = FindFirstFileW( (wchar_t*)file.utf16(), &FindFileData);
490  if (hFind == INVALID_HANDLE_VALUE) {
491  //qDebug() << "FindFirstFile failed " << GetLastError();
492  // returns false.
493  } else {
494  QString realFileName = QString::fromWCharArray( FindFileData.cFileName );
495  FindClose(hFind);
496 
497  if( ! file.endsWith(realFileName, Qt::CaseSensitive) ) {
498  qDebug() << Q_FUNC_INFO << "Detected case clash between" << file << "and" << realFileName;
499  re = true;
500  }
501  }
502 #else
503  // On Linux, the file system is case sensitive, but this code is useful for testing.
504  // Just check that there is no other file with the same name and different casing.
505  QFileInfo fileInfo(file);
506  const QString fn = fileInfo.fileName();
507  QStringList list = fileInfo.dir().entryList(QStringList() << fn);
508  if (list.count() > 1 || (list.count() == 1 && list[0] != fn)) {
509  re = true;
510  }
511 #endif
512  }
513  return re;
514 }

521 void OwncloudPropagator::scheduleNextJob()
522 {
523  // TODO: If we see that the automatic up-scaling has a bad impact we
524  // need to check how to avoid this.
525  // Down-scaling on slow networks? https://github.com/owncloud/client/issues/3382
526  // Making sure we do up/down at same time? https://github.com/owncloud/client/issues/1633
527 
528  if (_activeJobList.count() < maximumActiveJob()) {
529  if (_rootJob->scheduleNextJob()) {
530  QTimer::singleShot(0, this, SLOT(scheduleNextJob()));
531  }
532  } else if (_activeJobList.count() < hardMaximumActiveJob()) {
533  int likelyFinishedQuicklyCount = 0;
534  // NOTE: Only counts the first 3 jobs! Then for each
535  // one that is likely finished quickly, we can launch another one.
536  // When a job finishes another one will "move up" to be one of the first 3 and then
537  // be counted too.
538  for (int i = 0; i < maximumActiveJob() && i < _activeJobList.count(); i++) {
539  if (_activeJobList.at(i)->isLikelyFinishedQuickly()) {
540  likelyFinishedQuicklyCount++;
541  }
542  }
543  if (_activeJobList.count() < maximumActiveJob() + likelyFinishedQuicklyCount) {
544  qDebug() << "Can pump in another request! activeJobs =" << _activeJobList.count();
545  if (_rootJob->scheduleNextJob()) {
546  QTimer::singleShot(0, this, SLOT(scheduleNextJob()));
547  }
548  }
549  }
550 }

598 bool PropagateDirectory::scheduleNextJob()
599 {
600  if (_state == Finished) {
601  return false;
602  }
603 
604  if (_state == NotYetStarted) {
605  _state = Running;
606 
607  if (!_firstJob && _subJobs.isEmpty()) {
608  finalize();
609  return true;
610  }
611  }
612 
613  if (_firstJob && _firstJob->_state == NotYetStarted) {
614  return possiblyRunNextJob(_firstJob.data());
615  }
616 
617  if (_firstJob && _firstJob->_state == Running) {
618  return false;
619  }
620 
621  // cache the value of first unfinished subjob
622  bool stopAtDirectory = false;
623  int i = _firstUnfinishedSubJob;
624  int subJobsCount = _subJobs.count();
625  while (i < subJobsCount && _subJobs.at(i)->_state == Finished) {
626  _firstUnfinishedSubJob = ++i;
627  }
628 
629  for (int i = _firstUnfinishedSubJob; i < subJobsCount; ++i) {
630  if (_subJobs.at(i)->_state == Finished) {
631  continue;
632  }
633 
634  if (stopAtDirectory && qobject_cast<PropagateDirectory*>(_subJobs.at(i))) {
635  return false;
636  }
637 
638  if (possiblyRunNextJob(_subJobs.at(i))) {
639  return true;
640  }
641 
642  Q_ASSERT(_subJobs.at(i)->_state == Running);
643 
644  auto paral = _subJobs.at(i)->parallelism();
645  if (paral == WaitForFinished) {
646  return false;
647  }
648  if (paral == WaitForFinishedInParentDirectory) {
649  stopAtDirectory = true;
650  }
651  }

655 void PropagateDirectory::slotSubJobFinished(SyncFileItem::Status status)
656 {
657  if (status == SyncFileItem::FatalError ||
658  (sender() == _firstJob.data() && status != SyncFileItem::Success && status != SyncFileItem::Restoration)) {
659  abort();
660  _state = Finished;
661  emit finished(status);
662  return;
663  } else if (status == SyncFileItem::NormalError || status == SyncFileItem::SoftError) {
664  _hasError = status;
665  }
666  _runningNow--;
667  _jobsFinished++;
668 
669  int totalJobs = _subJobs.count();
670  if (_firstJob) {
671  totalJobs++;
672  }
673 
674  // We finished processing all the jobs
675  // check if we finished
676  if (_jobsFinished >= totalJobs) {
677  Q_ASSERT(!_runningNow); // how can we be finished if there are still jobs running now
678  finalize();
679  } else {
680  emit ready();
681  }

684 void PropagateDirectory::finalize()
685 {
686  bool ok = true;
687  if (!_item->isEmpty() && _hasError == SyncFileItem::NoStatus) {
688  if( !_item->_renameTarget.isEmpty() ) {
689  if(_item->_instruction == CSYNC_INSTRUCTION_RENAME
690  && _item->_originalFile != _item->_renameTarget) {
691  // Remove the stale entries from the database.
692  _propagator->_journal->deleteFileRecord(_item->_originalFile, true);
693  }
694 
695  _item->_file = _item->_renameTarget;
696  }
697 
698  // For new directories we always want to update the etag once
699  // the directory has been propagated. Otherwise the directory
700  // could appear locally without being added to the database.
701  if (_item->_instruction == CSYNC_INSTRUCTION_RENAME
702  || _item->_instruction == CSYNC_INSTRUCTION_NEW
703  || _item->_instruction == CSYNC_INSTRUCTION_UPDATE_METADATA) {
704  if (PropagateRemoteMkdir* mkdir = qobject_cast<PropagateRemoteMkdir*>(_firstJob.data())) {
705  // special case from MKDIR, get the fileId from the job there
706  if (_item->_fileId.isEmpty() && !mkdir->_item->_fileId.isEmpty()) {
707  _item->_fileId = mkdir->_item->_fileId;
708  }
709  }
710  SyncJournalFileRecord record(*_item, _propagator->_localDir + _item->_file);
711  ok = _propagator->_journal->setFileRecordMetadata(record);
712  if (!ok) {
713  _hasError = _item->_status = SyncFileItem::FatalError;
714  _item->_errorString = tr("Error writing metadata to the database");
715  qWarning() << "Error writing to the database for file" << _item->_file;
716  }
717  }
718  }
719  _state = Finished;
720  emit finished(_hasError == SyncFileItem::NoStatus ? SyncFileItem::Success : _hasError);
721 }

755 {

757  Q_ASSERT(job);

759  emit aborted(job->_item->_errorString);
760  deleteLater();
761  return;
762  } else if (job->_item->_status != SyncFileItem::Success) {
763  qDebug() << "There was an error with file " << job->_item->_file << job->_item->_errorString;
764  } else {

768  job->_item->_errorString = tr("Error writing metadata to the database");
769  emit aborted(job->_item->_errorString);
770  deleteLater();
771  return;
772  }
773  }

775  start();
