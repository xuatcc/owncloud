63 bool FileSystem::fileEquals(const QString& fn1, const QString& fn2)
64 {
65  // compare two files with given filename and return true if they have the same content
66  QFile f1(fn1);
67  QFile f2(fn2);
68  if (!f1.open(QIODevice::ReadOnly) || !f2.open(QIODevice::ReadOnly)) {
69  qDebug() << "fileEquals: Failed to open " << fn1 << "or" << fn2;
70  return false;
71  }
72 
73  if (getSize(fn1) != getSize(fn2)) {
74  return false;
75  }
76 
77  const int BufferSize = 16 * 1024;
78  char buffer1[BufferSize];
79  char buffer2[BufferSize];
80  do {
81  int r = f1.read(buffer1, BufferSize);
82  if (f2.read(buffer2, BufferSize) != r) {
83  // this should normally not happen: the files are supposed to have the same size.
84  return false;
85  }
86  if (r <= 0) {
87  return true;
88  }
89  if (memcmp(buffer1, buffer2, r) != 0) {
90  return false;
91  }
92  } while (true);
93  return false;
94 }

174 {

210 bool FileSystem::rename(const QString &originFileName,
211  const QString &destinationFileName,
212  QString *errorString)
213 {
214  bool success = false;
215  QString error;
216 #ifdef Q_OS_WIN
217  QString orig = longWinPath(originFileName);
218  QString dest = longWinPath(destinationFileName);
219 
220  if (isLnkFile(originFileName) || isLnkFile(destinationFileName)) {
221  success = MoveFileEx((wchar_t*)orig.utf16(),
222  (wchar_t*)dest.utf16(),
223  MOVEFILE_COPY_ALLOWED | MOVEFILE_WRITE_THROUGH);
224  if (!success) {
225  wchar_t *string = 0;
226  FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM,
227  NULL, ::GetLastError(), MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
228  (LPWSTR)&string, 0, NULL);
229 
230  error = QString::fromWCharArray(string);
231  LocalFree((HLOCAL)string);
232  }
233  } else
234 #endif
235  {
236  QFile orig(originFileName);
237  success = orig.rename(destinationFileName);
238  if (!success) {
239  error = orig.errorString();
240  }
241  }
242 
243  if (!success) {
244  qDebug() << "FAIL: renaming file" << originFileName
245  << "to" << destinationFileName
246  << "failed: " << error;
247  if (errorString) {
248  *errorString = error;
249  }
250  }
251  return success;
252 }

295 bool FileSystem::uncheckedRenameReplace(const QString& originFileName,
296  const QString& destinationFileName,
297  QString* errorString)
298 {
299 #ifndef Q_OS_WIN
300  bool success;
301  QFile orig(originFileName);
302 #if QT_VERSION < QT_VERSION_CHECK(5, 0, 0)
303  success = orig.fileEngine()->rename(destinationFileName);
304  // qDebug() << "Renaming " << tmpFile.fileName() << " to " << fn;
305 #else
306  // We want a rename that also overwites. QFile::rename does not overwite.
307  // Qt 5.1 has QSaveFile::renameOverwrite we could use.
308  // ### FIXME
309  success = true;
310  bool destExists = fileExists(destinationFileName);
311  if( destExists && !QFile::remove(destinationFileName) ) {
312  *errorString = orig.errorString();
313  qDebug() << Q_FUNC_INFO << "Target file could not be removed.";
314  success = false;
315  }
316  if( success ) {
317  success = orig.rename(destinationFileName);
318  }
319 #endif
320  if (!success) {
321  *errorString = orig.errorString();
322  qDebug() << "FAIL: renaming temp file to final failed: " << *errorString ;
323  return false;
324  }
325 
326 #else //Q_OS_WIN
327  // You can not overwrite a read-only file on windows.
328  if (!QFileInfo(destinationFileName).isWritable()) {
329  setFileReadOnly(destinationFileName, false);
330  }
331 
332  BOOL ok;
333  QString orig = longWinPath(originFileName);
334  QString dest = longWinPath(destinationFileName);
335 
336  ok = MoveFileEx((wchar_t*)orig.utf16(),
337  (wchar_t*)dest.utf16(),
338  MOVEFILE_REPLACE_EXISTING+MOVEFILE_COPY_ALLOWED+MOVEFILE_WRITE_THROUGH);
339  if (!ok) {
340  wchar_t *string = 0;
341  FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER|FORMAT_MESSAGE_FROM_SYSTEM,
342  NULL, ::GetLastError(), MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
343  (LPWSTR)&string, 0, NULL);
344 
345  *errorString = QString::fromWCharArray(string);
346  qDebug() << "FAIL: renaming temp file to final failed: " << *errorString;
347  LocalFree((HLOCAL)string);
348  return false;
349  }
350 #endif
351  return true;
352 }

354 bool FileSystem::openAndSeekFileSharedRead(QFile* file, QString* errorOrNull, qint64 seek)
355 {
356  QString errorDummy;
357  // avoid many if (errorOrNull) later.
358  QString& error = errorOrNull ? *errorOrNull : errorDummy;
359  error.clear();
360 
361 #ifdef Q_OS_WIN
362  //
363  // The following code is adapted from Qt's QFSFileEnginePrivate::nativeOpen()
364  // by including the FILE_SHARE_DELETE share mode.
365  //
366 
367  // Enable full sharing.
368  DWORD shareMode = FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE;
369 
370  int accessRights = GENERIC_READ;
371  DWORD creationDisp = OPEN_EXISTING;
372 
373  // Create the file handle.
374  SECURITY_ATTRIBUTES securityAtts = { sizeof(SECURITY_ATTRIBUTES), NULL, FALSE };
375  QString fName = longWinPath(file->fileName());
376 
377  HANDLE fileHandle = CreateFileW(
378  (const wchar_t*)fName.utf16(),
379  accessRights,
380  shareMode,
381  &securityAtts,
382  creationDisp,
383  FILE_ATTRIBUTE_NORMAL,
384  NULL);
385 
386  // Bail out on error.
387  if (fileHandle == INVALID_HANDLE_VALUE) {
388  error = qt_error_string();
389  return false;
390  }
391 
392  // Convert the HANDLE to an fd and pass it to QFile's foreign-open
393  // function. The fd owns the handle, so when QFile later closes
394  // the fd the handle will be closed too.
395  int fd = _open_osfhandle((intptr_t)fileHandle, _O_RDONLY);
396  if (fd == -1) {
397  error = "could not make fd from handle";
398  return false;
399  }
400  if (!file->open(fd, QIODevice::ReadOnly, QFile::AutoCloseHandle)) {
401  error = file->errorString();
402  return false;
403  }
404 
405  // Seek to the right spot
406  LARGE_INTEGER *li = reinterpret_cast<LARGE_INTEGER*>(&seek);
407  DWORD newFilePointer = SetFilePointer(fileHandle, li->LowPart, &li->HighPart, FILE_BEGIN);
408  if (newFilePointer == 0xFFFFFFFF && GetLastError() != NO_ERROR) {
409  error = qt_error_string();
410  return false;
411  }
412 
413  return true;
414 #else
415  if (!file->open(QFile::ReadOnly)) {
416  error = file->errorString();
417  return false;
418  }
419  if (!file->seek(seek)) {
420  error = file->errorString();
421  return false;
422  }
423  return true;
424 #endif
425 }

514 static QByteArray readToCrypto( const QString& filename, QCryptographicHash::Algorithm algo )
515 {
516  QFile file(filename);
517  const qint64 bufSize = qMin(BUFSIZE, file.size() + 1);
518  QByteArray buf(bufSize, Qt::Uninitialized);
519  QByteArray arr;
520  QCryptographicHash crypto( algo );
521 
522  if (file.open(QIODevice::ReadOnly)) {
523  qint64 size;
524  while (!file.atEnd()) {
525  size = file.read( buf.data(), bufSize );
526  if( size > 0 ) {
527  crypto.addData(buf.data(), size);
528  }
529  }
530  arr = crypto.result().toHex();
531  }
532  return arr;
533 }

566 QString FileSystem::makeConflictFileName(const QString &fn, const QDateTime &dt)
567 {
568  QString conflictFileName(fn);
569  // Add _conflict-XXXX before the extension.
570  int dotLocation = conflictFileName.lastIndexOf('.');
571  // If no extension, add it at the end (take care of cases like foo/.hidden or foo.bar/file)
572  if (dotLocation <= conflictFileName.lastIndexOf('/') + 1) {
573  dotLocation = conflictFileName.size();
574  }
575  QString timeString = dt.toString("yyyyMMdd-hhmmss");
576 
577  // Additional marker
578  QByteArray conflictFileUserName = qgetenv("CSYNC_CONFLICT_FILE_USERNAME");
579  if (conflictFileUserName.isEmpty())
580  conflictFileName.insert(dotLocation, "_conflict-" + timeString);
581  else
582  conflictFileName.insert(dotLocation, "_conflict_" + QString::fromUtf8(conflictFileUserName) + "-" + timeString);
583 
584  return conflictFileName;
585 }
