60 void ClientProxy::setupQtProxyFromConfig()
61 {
62  OCC::ConfigFile cfg;
63  int proxyType = QNetworkProxy::DefaultProxy;
64  QNetworkProxy proxy;
65 
66  // if there is no config file, default to system proxy.
67  if( cfg.exists() ) {
68  proxyType = cfg.proxyType();
69  proxy = proxyFromConfig(cfg);
70  }
71 
72  switch(proxyType) {
73  case QNetworkProxy::NoProxy:
74  qDebug() << "Set proxy configuration to use NO proxy";
75  QNetworkProxyFactory::setUseSystemConfiguration(false);
76  QNetworkProxy::setApplicationProxy(QNetworkProxy::NoProxy);
77  break;
78  case QNetworkProxy::DefaultProxy:
79  qDebug() << "Set proxy configuration to use system configuration";
80  QNetworkProxyFactory::setUseSystemConfiguration(true);
81  break;
82  case QNetworkProxy::Socks5Proxy:
83  proxy.setType(QNetworkProxy::Socks5Proxy);
84  qDebug() << "Set proxy configuration to SOCKS5" << printQNetworkProxy(proxy);
85  QNetworkProxyFactory::setUseSystemConfiguration(false);
86  QNetworkProxy::setApplicationProxy(proxy);
87  break;
88  case QNetworkProxy::HttpProxy:
89  proxy.setType(QNetworkProxy::HttpProxy);
90  qDebug() << "Set proxy configuration to HTTP" << printQNetworkProxy(proxy);
91  QNetworkProxyFactory::setUseSystemConfiguration(false);
92  QNetworkProxy::setApplicationProxy(proxy);
93  break;
94  default:
95  break;
96  }
97 }
