250 ProgressInfo::Estimates ProgressInfo::totalProgress() const
251 {
252  Estimates file = _fileProgress.estimates();
253  if (_sizeProgress._total == 0) {
254  return file;
255  }
256 
257  Estimates size = _sizeProgress.estimates();
258 
259  // Ideally the remaining time would be modeled as:
260  // remaning_file_sizes / transfer_speed
261  // + remaining_file_count * per_file_overhead
262  // + remaining_chunked_file_sizes / chunked_reassembly_speed
263  // with us estimating the three parameters in conjunction.
264  //
265  // But we currently only model the bandwidth and the files per
266  // second independently, which leads to incorrect values. To slightly
267  // mitigate this problem, we combine the two models depending on
268  // which factor dominates (essentially big-file-upload vs.
269  // many-small-files)
270  //
271  // If we have size information, we prefer an estimate based
272  // on the upload speed. That's particularly relevant for large file
273  // up/downloads, where files per second will be close to 0.
274  //
275  // However, when many *small* files are transfered, the estimate
276  // can become very pessimistic as the transfered amount per second
277  // drops significantly.
278  //
279  // So, if we detect a high rate of files per second or a very low
280  // transfer rate (often drops hugely during a sequence of deletes,
281  // for instance), we gradually prefer an optimistic estimate and
282  // assume the remaining transfer will be done with the highest speed
283  // we've seen.
284 
285  // Compute a value that is 0 when fps is <=L*max and 1 when fps is >=U*max
286  double fps = _fileProgress._progressPerSec;
287  double fpsL = 0.5;
288  double fpsU = 0.8;
289  double nearMaxFps =
290  qBound(0.0,
291  (fps - fpsL * _maxFilesPerSecond) /
292  ((fpsU - fpsL) * _maxFilesPerSecond),
293  1.0);
294 
295  // Compute a value that is 0 when transfer is >= U*max and
296  // 1 when transfer is <= L*max
297  double trans = _sizeProgress._progressPerSec;
298  double transU = 0.1;
299  double transL = 0.01;
300  double slowTransfer = 1.0 -
301  qBound(0.0,
302  (trans - transL * _maxBytesPerSecond) /
303  ((transU - transL) * _maxBytesPerSecond),
304  1.0);
305 
306  double beOptimistic = nearMaxFps * slowTransfer;
307  size.estimatedEta = (1.0 - beOptimistic) * size.estimatedEta
308  + beOptimistic * optimisticEta();
309 
310  return size;
311 }

333 void ProgressInfo::updateEstimates()
334 {
335  _sizeProgress.update();
336  _fileProgress.update();
337 
338  // Update progress of all running items.
339  QMutableHashIterator<QString, ProgressItem> it(_currentItems);
340  while (it.hasNext()) {
341  it.next();
342  it.value()._progress.update();
343  }
344 
345  _maxFilesPerSecond = qMax(_fileProgress._progressPerSec,
346  _maxFilesPerSecond);
347  _maxBytesPerSecond = qMax(_sizeProgress._progressPerSec,
348  _maxBytesPerSecond);
349 }

383 void ProgressInfo::Progress::update()
384 {
385  // A good way to think about the smoothing factor:
386  // If we make progress P per sec and then stop making progress at all,
387  // after N calls to this function (and thus seconds) the _progressPerSec
388  // will have reduced to P*smoothing^N.
389  // With a value of 0.9, only 4% of the original value is left after 30s
390  //
391  // In the first few updates we want to go to the correct value quickly.
392  // Therefore, smoothing starts at 0 and ramps up to its final value over time.
393  const double smoothing = 0.9 * (1.0 - _initialSmoothing);
394  _initialSmoothing *= 0.7; // goes from 1 to 0.03 in 10s
395  _progressPerSec = smoothing * _progressPerSec + (1.0 - smoothing) * (_completed - _prevCompleted);
396  _prevCompleted = _completed;
397 }
