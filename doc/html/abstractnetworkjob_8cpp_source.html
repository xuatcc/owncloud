43 AbstractNetworkJob::AbstractNetworkJob(AccountPtr account, const QString &path, QObject *parent)
44  : QObject(parent)
45  , _timedout(false)
46  , _followRedirects(true)
47  , _account(account)
48  , _ignoreCredentialFailure(false)
49  , _reply(0)
50  , _path(path)
51  , _redirectCount(0)
52 {
53  _timer.setSingleShot(true);
54  _timer.setInterval(OwncloudPropagator::httpTimeout() * 1000); // default to 5 minutes.
55  connect(&_timer, SIGNAL(timeout()), this, SLOT(slotTimeout()));
56 
57  connect(this, SIGNAL(networkActivity()), SLOT(resetTimeout()));
58 
59  // Network activity on the propagator jobs (GET/PUT) keeps all requests alive.
60  // This is a workaround for OC instances which only support one
61  // parallel up and download
62  if (_account) {
63  connect(_account.data(), SIGNAL(propagatorNetworkActivity()), SLOT(resetTimeout()));
64  }
65 }

156 void AbstractNetworkJob::slotFinished()
157 {
158  _timer.stop();
159 
160  if( _reply->error() == QNetworkReply::SslHandshakeFailedError ) {
161  qDebug() << "SslHandshakeFailedError: " << reply()->errorString() << " : can be caused by a webserver wanting SSL client certificates";
162  }
163 
164  if( _reply->error() != QNetworkReply::NoError ) {
165  qDebug() << Q_FUNC_INFO << _reply->error() << _reply->errorString()
166  << _reply->attribute(QNetworkRequest::HttpStatusCodeAttribute);
167  if (_reply->error() == QNetworkReply::ProxyAuthenticationRequiredError) {
168  qDebug() << Q_FUNC_INFO << _reply->rawHeader("Proxy-Authenticate");
169  }
170  emit networkError(_reply);
171  }
172 
173  // get the Date timestamp from reply
174  _responseTimestamp = _reply->rawHeader("Date");
175  _duration = _durationTimer.elapsed();
176 
177  if (_followRedirects) {
178  // ### the qWarnings here should be exported via displayErrors() so they
179  // ### can be presented to the user if the job executor has a GUI
180  QUrl requestedUrl = reply()->request().url();
181  QUrl redirectUrl = reply()->attribute(QNetworkRequest::RedirectionTargetAttribute).toUrl();
182  if (!redirectUrl.isEmpty()) {
183  _redirectCount++;
184  if (requestedUrl.scheme() == QLatin1String("https") &&
185  redirectUrl.scheme() == QLatin1String("http")) {
186  qWarning() << this << "HTTPS->HTTP downgrade detected!";
187  } else if (requestedUrl == redirectUrl || _redirectCount >= maxRedirects()) {
188  qWarning() << this << "Redirect loop detected!";
189  } else {
190  resetTimeout();
191  setReply(getRequest(redirectUrl));
192  setupConnections(reply());
193  return;
194  }
195  }
196  }
197 
198  AbstractCredentials *creds = _account->credentials();
199  if (!creds->stillValid(_reply) && ! _ignoreCredentialFailure) {
200  _account->handleInvalidCredentials();
201  }
202 
203  bool discard = finished();
204  if (discard) {
205  deleteLater();
206  }
207 }

265 QString extractErrorMessage(const QByteArray& errorResponse)
266 {
267  QXmlStreamReader reader(errorResponse);
268  reader.readNextStartElement();
269  if (reader.name() != "error") {
270  return QString::null;
271  }
272 
273  QString exception;
274  while (!reader.atEnd() && !reader.hasError()) {
275  reader.readNextStartElement();
276  if (reader.name() == QLatin1String("message")) {
277  QString message = reader.readElementText();
278  if (!message.isEmpty()) {
279  return message;
280  }
281  } else if (reader.name() == QLatin1String("exception")) {
282  exception = reader.readElementText();
283  }
284 
285  }
286  // Fallback, if message could not be found
287  return exception;
288 }
