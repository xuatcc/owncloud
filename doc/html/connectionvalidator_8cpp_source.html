63 void ConnectionValidator::checkServerAndAuth()
64 {
65  if( !_account ) {
66  _errors << tr("No ownCloud account configured");
67  reportResult( NotConfigured );
68  return;
69  }
70  qDebug() << "Checking server and authentication";
71 
72  _isCheckingServerAndAuth = true;
73 
74  // Lookup system proxy in a thread https://github.com/owncloud/client/issues/2993
75  if (ClientProxy::isUsingSystemDefault()) {
76  qDebug() << "Trying to look up system proxy";
77  ClientProxy::lookupSystemProxyAsync(_account->url(),
78  this, SLOT(systemProxyLookupDone(QNetworkProxy)));
79  } else {
80  // We want to reset the QNAM proxy so that the global proxy settings are used (via ClientProxy settings)
81  _account->networkAccessManager()->setProxy(QNetworkProxy(QNetworkProxy::DefaultProxy));
82  // use a queued invocation so we're as asynchronous as with the other code path
83  QMetaObject::invokeMethod(this, "slotCheckServerAndAuth", Qt::QueuedConnection);
84  }
85 }

87 void ConnectionValidator::systemProxyLookupDone(const QNetworkProxy &proxy) {
88  if (!_account) {
89  qDebug() << "Bailing out, Account had been deleted";
90  return;
91  }
92 
93  if (proxy.type() != QNetworkProxy::NoProxy) {
94  qDebug() << "Setting QNAM proxy to be system proxy" << printQNetworkProxy(proxy);
95  } else {
96  qDebug() << "No system proxy set by OS";
97  }
98  _account->networkAccessManager()->setProxy(proxy);
99 
100  slotCheckServerAndAuth();
101 }

104 void ConnectionValidator::slotCheckServerAndAuth()
105 {
106  CheckServerJob *checkJob = new CheckServerJob(_account, this);
107  checkJob->setTimeout(timeoutToUseMsec);
108  checkJob->setIgnoreCredentialFailure(true);
109  connect(checkJob, SIGNAL(instanceFound(QUrl,QVariantMap)), SLOT(slotStatusFound(QUrl,QVariantMap)));
110  connect(checkJob, SIGNAL(instanceNotFound(QNetworkReply*)), SLOT(slotNoStatusFound(QNetworkReply*)));
111  connect(checkJob, SIGNAL(timeout(QUrl)), SLOT(slotJobTimeout(QUrl)));
112  checkJob->start();
113 }

115 void ConnectionValidator::slotStatusFound(const QUrl&url, const QVariantMap &info)
116 {
117  // status.php was found.
118  qDebug() << "** Application: ownCloud found: "
119  << url << " with version "
120  << CheckServerJob::versionString(info)
121  << "(" << CheckServerJob::version(info) << ")";
122 
123  QString version = CheckServerJob::version(info);
124  _account->setServerVersion(version);
125 
126  // We cannot deal with servers < 5.0.0
127  if (version.contains('.') && version.split('.')[0].toInt() < 5) {
128  _errors.append( tr("The configured server for this client is too old") );
129  _errors.append( tr("Please update to the latest server and restart the client.") );
130  reportResult( ServerVersionMismatch );
131  return;
132  }
133 
134  // We attempt to work with servers >= 5.0.0 but warn users.
135  // Check usages of Account::serverVersionUnsupported() for details.
136 
137  // now check the authentication
138  if (_account->credentials()->ready())
139  QTimer::singleShot( 0, this, SLOT( checkAuthentication() ));
140  else
141  reportResult( CredentialsMissingOrWrong );
142 }

166 void ConnectionValidator::checkAuthentication()
167 {
168  AbstractCredentials *creds = _account->credentials();
169 
170  if (!creds->ready()) { // The user canceled
171  reportResult(UserCanceledCredentials);
172  }
173 
174  // simply GET the webdav root, will fail if credentials are wrong.
175  // continue in slotAuthCheck here :-)
176  qDebug() << "# Check whether authenticated propfind works.";
177  PropfindJob *job = new PropfindJob(_account, "/", this);
178  job->setTimeout(timeoutToUseMsec);
179  job->setProperties(QList<QByteArray>() << "getlastmodified");
180  connect(job, SIGNAL(result(QVariantMap)), SLOT(slotAuthSuccess()));
181  connect(job, SIGNAL(finishedWithError(QNetworkReply*)), SLOT(slotAuthFailed(QNetworkReply*)));
182  job->start();
183 }

185 void ConnectionValidator::slotAuthFailed(QNetworkReply *reply)
186 {
187  Status stat = Timeout;
188 
189  if( reply->error() == QNetworkReply::AuthenticationRequiredError ||
190  !_account->credentials()->stillValid(reply)) {
191  qDebug() << reply->error() << reply->errorString();
192  qDebug() << "******** Password is wrong!";
193  _errors << tr("The provided credentials are not correct");
194  stat = CredentialsMissingOrWrong;
195 
196  } else if( reply->error() != QNetworkReply::NoError ) {
197  _errors << errorMessage(reply->errorString(), reply->readAll());
198 
199  const int httpStatus =
200  reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
201  if ( httpStatus == 503 ) {
202  _errors.clear();
203  stat = ServiceUnavailable;
204  }
205  }
206 
207  reportResult( stat );
208 }
